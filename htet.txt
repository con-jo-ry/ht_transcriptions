	<l><pb n="154"/>atha vajrī sarvatantranidānaṃ nāmopāyaṃ yoginīnāṃ kathayām āsa—</l>
	<l>saṃvaraṃ cābhiṣekaṃ ca sandhyābhāṣāṃ tathaiva ca |</l>
	<l>ānandakṣaṇabhedaṃ ca tathānyaṃ bhojanādikam || 1 ||</l>
</lg>

<lg xml:id="HT2.3.2">
	<l>tatra saṃvaram āha—</l>
	<l>saṃvaraṃ sarvabuddhānām evaṃkāre pratiṣṭhitam |</l>
	<l>abhiṣekāj jñāyate samyag evaṃkāraṃ mahat sukham || 2 ||</l>
</lg>

<lg xml:id="HT2.3.3">
	<l><pb n="155"/>atha bhagavantaṃ vajrasattvaṃ yoginya evam āhuḥ—</l>
	<l>evaṃkāraṃ kim ucyate ḍākinīnān tu saṃvaram |</l>
	<l>deśayantu yathānyāyaṃ bhagavān śāstā jagadguruḥ || 3 ||</l>
</lg>

<lg xml:id="HT2.3.4">
	<l>bhagavān āha—</l>
	<l>ekārākṛti yad divyaṃ madhye vaṃkārabhūṣitam |</l>
	<l>ālayaḥ sarvasaukhyānāṃ buddharatnakaraṇḍakam || 4 ||</l>
</lg>

<lg xml:id="HT2.3.5">
	<l>ānandās tatra jāyante kṣaṇabhedena bheditāḥ |</l>
	<l>kṣaṇajñānāt sukhajñānam evaṃkāre pratiṣṭhitam || 5 ||</l>
</lg>

<lg xml:id="HT2.3.6">
	<l>*vicitraṃ ca vipākaṃ ca vimardo vilakṣaṇaṃ tathā |</l>
	<l>catuḥkṣaṇasamāgamyam evaṃ jānanti yoginaḥ* || 6 ||</l>
</lg>

<lg xml:id="HT2.3.7">
	<l><pb n="156"/>vicitraṃ vividhaṃ khyātaṃ āliṅgacumbanādikam |</l>
	<l>vipākaṃ tadviparyāsaṃ sukhajñānasya bhuñjanam || 7 ||</l>
</lg>

<lg xml:id="HT2.3.8">
	<l>vimardam ālocanaṃ proktaṃ sukhaṃ bhuktaṃ mayeti ca |</l>
	<l>vilakṣaṇaṃ tribhyo 'nyatra rāgārāgavivarjitam || 8 ||</l>
</lg>

<lg xml:id="HT2.3.9">
	<l>vicitre prathamānadaḥ paramānando vipākake |</l>
	<l>viramānando vimarde ca sahajānando vilakṣaṇe || 9 ||</l>
</lg>

<lg xml:id="HT2.3.10">
	<l>ācārya guhya prajñā ca caturthan tat punas tathā |</l>
	<l>ānandāḥ kramaśo jñeyāś catuḥsecanasaṃkhyayā || 10 ||</l>
</lg>

<lg xml:id="HT2.3.11">
	<l><pb n="157"/>hasitaśuddhyā tv ācārya īkṣaṇe guhyakas tathā |</l>
	<l>prajñā hi pāṇyāvāptau ca tat punar dvandvatantrake || 11 ||</l>
</lg>

<lg xml:id="HT2.3.12">
	<l>sekaṃ caturvidhaṃ khyātaṃ sattvānāṃ siddhihetave |</l>
	<l>sicyate snāpyate 'neneti sekas tenābhidhīyate || 12 ||</l>
</lg>

<lg xml:id="HT2.3.13">
	<l>pāṇibhyāṃ tu samāliṅgya prajñāṃ vai ṣoḍaśābdikām |</l>
	<l>ghaṇṭāvajrasamāyogād ācāryasecanaṃ matam || 13 ||</l>
</lg>

<lg xml:id="HT2.3.14">
	<l>cāruvaktrā viśālākṣī rūpayauvanamaṇḍitā |</l>
	<l>jyeṣṭhānāmikābhyāṃ ca śiṣyavaktre nipātayet || 14 ||</l>
</lg>

<lg xml:id="HT2.3.15">
	<l><pb n="158"/>*kāritavyaṃ tu tatraiva samarasaṃ śiṣyagocaram* |</l>
	<l>prajñāṃ pūjayec chāstā arcayitvā samarpayet || 15 ||</l>
</lg>

<lg xml:id="HT2.3.16">
	<l>śāstā brūyāt mahāsattva gṛhṇa mudrāṃ sukhāvahām |</l>
	<l>jñātvā śiṣyaṃ mahadbhūtaṃ nirīrṣyaṃ krodhavarjitam || 16 ||</l>
</lg>

<lg xml:id="HT2.3.17">
	<l>śāstā tam ājñāpayati kunduruṃ kuru vajradhṛk |</l>
	<l>śiṣyakṛtyaṃ pravakṣyāmy abhiṣekam anunāyayed || 17 ||</l>
</lg>

<lg xml:id="HT2.3.18">
	<l>mudrāyuktaṃ guruṃ dṛṣṭvā stutipūjāṃ kared yathā |</l>
	<l>he bhagavan mahāśānta vajrayogaikatatpara || 18 ||</l>
</lg>

<lg xml:id="HT2.3.19">
	<l>mudrāprasādhakābhedyavajrayogasamudbhava |</l>
	<l>yathā yūyaṃ mahātmāno mamāpi kuru tad vibho || 19 ||</l>
</lg>

<lg xml:id="HT2.3.20">
	<l><pb n="159"/>saṃsārapaṅkasaṃghāṭe magno 'haṃ trāhy aśaraṇam |</l>
	<l>miṣṭānnapānakhādyaṃ ca madanaṃ balaṃ mahattaram || 20 ||</l>
</lg>

<lg xml:id="HT2.3.21">
	<l>dhūpan naivedyaṃ mālyaṃ ca ghaṇṭādhvajavilepanaiḥ |</l>
	<l>ābhiḥ pūjādibhiḥ śiṣyaḥ pūjayed vajradhāriṇam || 21 ||</l>
</lg>

<lg xml:id="HT2.3.22">
	<l>paramānande tu saṃprāpte nānātvavarjite kṣaṇe |</l>
	<l>śāstā brūyāt mahāsattva dhāraṇīyaṃ mahat sukham || 22 ||</l>
</lg>

<lg xml:id="HT2.3.23">
	<l>yāvad ā bodhiparyantaṃ sattvārthaṃ kuru vajradhṛk |</l>
	<l>ity evaṃ vadate vajrī śiṣyaṃ dīkṣākriyācayam || 23 ||</l>
</lg>

<lg xml:id="HT2.3.24">
	<l><pb n="160"/>etad eva mahājñānaṃ sarvadehe vyavasthitam |</l>
	<l>advayaṃ dvayarūpaṃ ca bhāvābhāvātmakaṃ prabhum || 24 ||</l>
</lg>

<lg xml:id="HT2.3.25">
	<l>sthiracalaṃ vyāpya saṃtiṣṭhet māyārūpi ca bhāti ca |</l>
	<l>maṇḍalacakrādyupāyena sātatyaṃ yāti niścayam || 25 ||</l>
</lg>

<p xml:id="HT2.3.26">
	<l>atha sarvayoginīnāṃ bhaginīnāṃ mṛṣitvā vajragarbho bhagavantam evam āha—</l>
	<l>maṇḍalacakraṃ kim ucyeta sarvabuddhātmakaṃ puram |</l>
	<l>deśayantu yathānyāyaṃ bhagavanto bhrāntir me 'bhūt || 26 ||</l>
</p>

<lg xml:id="HT2.3.27">
	<l><pb n="161"/>bhagavān āha—</l>
	<l>maṇḍalaṃ sāram ity uktaṃ bodhicittaṃ mahat sukham |</l>
	<l>ādānaṃ tat karotīti maṇḍalaṃ malanaṃ matam || 27 ||</l>
</lg>

<lg xml:id="HT2.3.28">
	<l>cakraṃ nivahaṃ khadhātvākhyaṃ viṣayādīnāṃ viśodhanam |</l>
	<l>bolakakkolayogena tasya saukhyaṃ pratīyate || 28 ||</l>
</lg>

<lg xml:id="HT2.3.29">
	<l><pb n="162"/>vajragarbha āha—</l>
	<l>kena samayena sthātavyaṃ kena saṃvareṇeti |</l>
	<l>bhagavān āha—</l>
	<l>prāṇinaś ca tvayā ghātyā vaktavyaṃ ca mṛṣāvacaḥ |</l>
	<l>adattaṃ ca tvayā grāhyaṃ sevanaṃ parayoṣitaḥ || 29 ||</l>
</lg>

<lg xml:id="HT2.3.30">
	<l>ekacittaṃ prāṇivadhaṃ prāṇacittaṃ yato matam |</l>
	<l>lokān uttārayiṣyāmi mṛṣāvādaṃ ca śabditam |</l>
	<l>yoṣicchukram adattaṃ ca paradārāḥ svābhasundarī || 30 ||</l>
</lg>

<lg xml:id="HT2.3.31">
	<l><pb n="163"/>atha sarvayoginyo bhagavantam e<cb/>vam āhuḥ |</l>
	<l>ke punas te viṣayāḥ kānīndriyāṇi kim āyatanaṃ | katame skandhāḥ | ke punas te dhātavaḥ | eṣāṃ kiṃ svabhāvam || 31 ||</l>
</lg>

<lg xml:id="HT2.3.32">
	<l>bhagavān āha—ṣaḍ viṣayāḥ |</l>
	<l>rūpa śabdas tathā gandho rasa sparśas tathaiva ca |</l>
	<l>dharmadhātusvabhāvaś ca ṣaḍ ete viṣayā matāḥ || 32 ||</l>
</lg>

<lg xml:id="HT2.3.33">
	<l>indriyāṇi ca ṣaṭ |</l>
	<l><pb n="164"/>cakṣuḥ śrotraṃ ca ghrāṇaṃ ca jihvā kāyo manas tathā |</l>
	<l>mohavajrādibhir yuktāḥ ṣaḍ etānīndryāṇi ca || 33 ||</l>
</lg>

<lg xml:id="HT2.3.34">
	<l>viṣayavaiṣayikābhyāṃ ca dvādaśāyatanaṃ bhavet |</l>
	<l>pañcaskandhāś ca rūpādyā vijñānāntā mahākṛpa || 34 ||</l>
</lg>

<lg xml:id="HT2.3.35">
	<l>indriyaṃ viṣayaṃ caiva indriyavijñānam eva ca |</l>
	<l>dhātavo 'ṣṭādaśākhyātā yoginīnāṃ tu bodhaye || 35 ||</l>
</lg>

<lg xml:id="HT2.3.36">
	<l>*svabhāvam ādyanutpannaṃ na satyaṃ na mṛṣā tathā |</l>
	<l>udakacandropamaṃ sarvaṃ yoginyo jānatecchayā* || 36 ||</l>
</lg>

<p xml:id="HT2.3.37">
	<pb n="165"/>tad yathā kāṇḍaṃ ca mathanīyaṃ ca puruṣahastavyāyāmaṃ ca pratītyākasmād agnir upajāyate | asāv agnir na kāṇḍe tiṣṭhati | na mathanīye na puruṣahastavyāyāme | sarvākārataḥ parigaveṣyamānaḥ, ekasminn api nāsti | sa cāgnir na satyaṃ na mṛṣā || evaṃ sarvadharmān yoginyo manasikuruta || 37 ||
</p>

<p xml:id="HT2.3.38">
	atha nairātmyayoginīpramukhāḥ sarvavajraḍākinyaḥ pañcāmṛtaṃ gṛhītvā samayadravyaṃ ca bhagavantaṃ vajrasattvaṃ pūjayanti | kunduruyogenānurāgayanti | pibayanti ca vajrāmṛtarasam || 38 ||
</p>

<lg xml:id="HT2.3.39">
	<l><pb n="166"/>tataḥ paścād bhagavāṃs tuṣṭe sati adhiṣṭhānaṃ darśayati |</l>
	<l>bho bho vajraḍākinyaḥ—</l>
	<l>mayā guptīkṛtaṃ tattvaṃ sarvair buddhair namaskṛtam |</l>
	<l>vajrapūjāprabhāvena kathayāmi śṛṇutecchayā || 39 ||</l>
</lg>

<p xml:id="HT2.3.40">
	atha utsāhaprāptāḥ sarvā devyo dakṣiṇaṃ jānumaṇḍalaṃ pṛthivyām pratiṣṭhāpya yena bhagavāṃs tenāñjaliṃ praṇamya bhagavato bhāṣitaṃ śṛṇvanti || 40 ||
</p>

<lg xml:id="HT2.3.41">
	<l>bhagavān āha—</l>
	<l>khānaṃ pānaṃ yathāprāptaṃ gamyāgamyaṃ na varjayet |</l>
	<l>snānaṃ śaucaṃ na kurvīta grāmyadharmaṃ na varjayet || 41 ||</l>
</lg>

<lg xml:id="HT2.3.42">
	<l>mantraṃ naiva japed dhīmān dhyānaṃ naivāvalambayet |</l>
	<l>nidrātyāgaṃ na kurvīta nendriyāṇāṃ nivāraṇam || 42 ||</l>
</lg>

<lg xml:id="HT2.3.43">
	<l>bhakṣaṇīyaṃ balaṃ sarvaṃ pañcavarṇaṃ samācaret |</l>
	<l>ramate sarvayoṣitā nirviśaṅkena cetasā || 43 ||</l>
</lg>

<lg xml:id="HT2.3.44">
	<l><pb n="167"/>mitrasnehaṃ na kurvīta dviṣṭe dveṣaṃ tathā na ca |</l>
	<l>na vandayed imān devān kāṣṭhapāṣāṇamṛṇmayān |</l>
	<l>*satataṃ devatāmūrtyā sthātavyaṃ yoginā yataḥ* || 44 ||</l>
</lg>

<lg xml:id="HT2.3.45">
	<l>ḍombacaṇḍālacarmārahaḍḍikādyān tu duḥspṛśān |</l>
	<l>brahmakṣatrivaiśyaśūdrādyān ātmadeham iva spṛśet || 45 ||</l>
</lg>

<lg xml:id="HT2.3.46">
	<l>pañcāmṛtaṃ guḍaṃ madyaṃ viṣaṃ nimbaṃ prasūtajam |</l>
	<l>amlamadhurakaṣāyādi tiktalavaṇakaṭukas tathā || 46 ||</l>
</lg>

<lg xml:id="HT2.3.47">
	<l>pūtisurabhi jalāsṛg bodhicittena bhakṣayet |</l>
	<l>nābhaktaṃ vidyate kiñcid advayajñānacetasā || 47 ||</l>
</lg>

<lg xml:id="HT2.3.48">
	<l>svayambhūkusumaṃ prāpya padmabhāṇḍe niveśayet |</l>
	<l>śleṣmasiṅghāṇakānāṃ tu miśrīkṛtya pibed vratī || 48 ||</l>
</lg>

<lg xml:id="HT2.3.49">
	<l>kaupīnaṃ viśvavarṇaṃ ca mṛcchārair bhūṣaṇaṃ tathā |</l>
	<l>puṣpaṃ pretālaye prāpya bandhayen mūrddhajaṃ varam || 49 ||</l>
</lg>

<lg xml:id="HT2.3.50">
	<l><pb n="168"/>atha vajragarbha āha—</l>
	<l>indriyāṇy aviśuddhāni ṣaṭsaṃkhyayākṛtāni vai |</l>
	<l>śuddhiḥ sarvaviṣayasya bhagavatā kathitā purā || 50 ||</l>
</lg>

<lg xml:id="HT2.3.51">
	<l>bhagavān āha—</l>
	<l>cakṣuṣo mohavajrā tu śrotrayor dveṣavajrikā |</l>
	<l>ghrāṇe mātsaryakī khyātā vaktre ca rāgavajrikā || 51 ||</l>
</lg>

<lg xml:id="HT2.3.52">
	<l>sparśe īrṣyāvajrā ca mano nairātmyayoginī |</l>
	<l>kavacam ebhir mahāsattva indriyāṇāṃ viśuddhaye || 52 ||</l>
</lg>

<lg xml:id="HT2.3.53">
	<l>vajragarbha uvāca—</l>
	<l>sandhyābhāṣaṃ kim ucyeta bhagavān bobrūta niścitam |</l>
	<l>yoginīnāṃ mahāsamayaṃ śrāvakādyair na chidritam || 53 ||</l>
</lg>

<lg xml:id="HT2.3.54">
	<l><pb n="169"/>hasitaṃ cekṣaṇābhyāṃ tu āliṅgaṃ dvandvakais tathā |</l>
	<l>tantreṇāpi caturṇāṃ ca saṃdhyābhāṣaṃ na śabditam || 54 ||</l>
</lg>

<lg xml:id="HT2.3.55">
	<l>bhagavān āha—</l>
	<l>vakṣyāmy ahaṃ vajragarbha śṛṇu tvam ekacetasā |</l>
	<l>sandhyābhāṣaṃ mahābhāṣaṃ samayasaṃketavistaram || 55 ||</l>
</lg>

<lg xml:id="HT2.3.56">
	<l>madanaṃ madyaṃ balaṃ māṃsaṃ malayajaṃ milanaṃ matam |</l>
	<l>gatiḥ kheṭaḥ śavaḥ śrāyo 'sthyābharaṇaṃ niraṃśukam || 56 ||</l>
</lg>

<lg xml:id="HT2.3.57">
	<l>āgatiḥ preṅkhaṇaṃ proktaṃ kṛpīṭaṃ ḍamarukaṃ matam |</l>
	<l>abhavyaṃ dunduraṃ khyātaṃ bhavyaṃ kāliñjaraṃ matam || 57 ||</l>
</lg>

<lg xml:id="HT2.3.58">
	<l><pb n="170"/>asparśaṃ ḍiṇḍimaṃ proktaṃ kapālaṃ padmabhājanam |</l>
	<l>bhakṣaṃ tṛptikaraṃ jñeyaṃ vyañjanaṃ mālatīndhanam || 58 ||</l>
</lg>

<lg xml:id="HT2.3.59">
	<l>gūthaṃ catuḥsamaṃ proktaṃ mūtraṃ kastūrikā smṛtā |</l>
	<l>svayaṃbhū sihlakaṃ jñeyaṃ śukraṃ karpūrakaṃ matam || 59 ||</l>
</lg>

<lg xml:id="HT2.3.60">
	<l>mahāmāṃsaṃ sālijaṃ proktaṃ dvīndriyayogaṃ kundurum |</l>
	<l>vajraṃ bolakaṃ khyātaṃ padmaṃ kakkolakaṃ matam || 60 ||</l>
</lg>

<lg xml:id="HT2.3.61">
	<l>kulaṃ pañcavidhaṃ khyātaṃ varṇabhedena bheditam |</l>
	<l>sandhyābhāṣata evaṃ syur buddhāś ca pañcakaulikāḥ || 61 ||</l>
</lg>

<lg xml:id="HT2.3.62">
	<l>ḍombī vajrakulī khyātā naṭī padmakulī tathā |</l>
	<l>caṇḍālī ratnakulī caiva dvijā tāthāgatī matā || 62 ||</l>
</lg>

<lg xml:id="HT2.3.63">
	<l>rajakī karmakulī caiva etā mudrāḥ susiddhidāḥ |</l>
	<l>āsāṃ śukraṃ bhaved vajraṃ pūjayitvā pibed vratī || 63 ||</l>
</lg>

<lg xml:id="HT2.3.64">
	<l><pb n="171"/>vajragarbha mahāsattva yan mayā kathitaṃ tvayi |</l>
	<l>tat sarvaṃ sādaraṃ grāhyaṃ sandhyābhāṣaṃ mahādbhutam || 64 ||</l>
</lg>

<lg xml:id="HT2.3.65">
	<l>yo 'bhiṣikto 'tra hevajre na vadet saṃdhyābhāṣayā |</l>
	<l>samayavidrohanaṃ tasya jāyate nātra saṃśayaḥ || 65 ||</l>
</lg>

<lg xml:id="HT2.3.66">
	<l>ity upadravacauraiś ca grahajvaraviṣādibhiḥ |</l>
	<l>mṛyate 'sau yadi buddho 'pi sandhyābhāṣān na bhāṣayet || 66 ||</l>
</lg>

<lg xml:id="HT2.3.67">
	<l>svasamayavidāṃ prāpya yadi na bhāṣed idaṃ vacaḥ |</l>
	<l>tadā kṣobhaṃ prakurvanti yoginyaś catuḥpīṭhajāḥ || 67 ||</l>
</lg>
