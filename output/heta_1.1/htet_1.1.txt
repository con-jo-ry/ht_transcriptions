<div n="1" type="paṭala">

<p xml:id="HT1.1.0">
	<pb n="1"/>oṃ namaḥ śrīhevajrāya ||
</p>

<p xml:id="HT1.1.1">
	evaṃ mayā śrutam | ekasmin samaye bhagavān sarvatathāgatakāyavākcittahṛdayavajrayoṣidbhageṣu vijahāra || 1 ||
</p>

<p xml:id="HT1.1.2">
	<pb n="4"/>tatra bhagavān āha—sarvatathāgatakāyavākcittahṛdayaṃ bhaṭṭārakaṃ guhyātiguhyataraṃ iti | aho vajragarbha sādhu sādhu mahākṛpa mahābodhisattva [bodhisattvasya] vajrasattvasya mahāsattvasya samayasattvasya [jñānasattvasya] hṛdayaṃ hevajrākhyaṃ śṛṇu || 2 ||
</p>

<lg xml:id="HT1.1.3">
	<l><pb n="6"/>vajragarbha uvāca—</l>
	<l>vajrasattvo bhavet kasmāt mahāsattvo bhavet katham |</l>
	<l>samayasattvo bhavet kena kathayatu bhagavān mayi || 3 ||</l>
</lg>

<lg xml:id="HT1.1.4">
	<l>bhagavān āha—</l>
	<l>abhedyaṃ vajram ity uktam sattvaṃ tribhavasyaikatā |</l>
	<l>anayā prajñayā yuktyā vajrasattva iti smṛtaḥ || 4 ||</l>
</lg>

<lg xml:id="HT1.1.5">
	<l><pb n="7"/>mahājñānarasaiḥ pūrṇo mahāsattvo nigadyate |</l>
	<l>nityaṃ samayapravṛttatvāt samayasattvo 'bhidhīyate || 5 ||</l>
</lg>

<lg xml:id="HT1.1.5a">
	<l>bodhicaryāsamāsena bodhisattvo nigadyate |</l>
	<l>prajñājñānasamāyogāj jñānasattvas tathāgataḥ ||</l>
</lg>

<lg xml:id="HT1.1.6">
	<l><pb n="8"/>vajragarbha uvāca—</l>
	<l>hevajraṃ tu bhavet kena īdṛśaṃ nāmasaṃgraham |</l>
	<l>hekāreṇa kim ākhyātaṃ vajreṇāpi kim ucyate || 6 ||</l>
</lg>

<lg xml:id="HT1.1.7">
	<l><pb n="9"/>bhagavān āha—</l>
	<l>hekāreṇa mahākaruṇā vajraṃ prajñā ca bhaṇyate |</l>
	<l>prajñopāyātmakaṃ tantraṃ tan me nigaditaṃ śṛṇu || 7 |</l>
</lg>

<lg xml:id="HT1.1.8">
	<l><pb n="10"/>dṛṣṭyākrṣṭi mahācchomaṃ sāmarthyaṃ bahuvidhaṃ viduḥ |</l>
	<l>stambhanoccāṭanaṃ caiva sainyastambhābhicārukam || 8 ||</l>
</lg>

<lg xml:id="HT1.1.9">
	<l>yoginīnāṃ yathānyāyam utpattisthitikāraṇam |</l>
	<l>sāmarthyaṃ jñānavijñānaṃ devatānāṃ yathodayam || 9 ||</l>
</lg>

<lg xml:id="HT1.1.10">
	<l>prathamaṃ tāvad bhaved ekaṃ herukotpattikāraṇam |</l>
	<l>bhāvenaiva vimucyante vajragarbha mahākṛpa || 10 ||</l>
</lg>

<lg xml:id="HT1.1.11">
	<l><pb n="13"/>badhyante bhāvabandhena mucyante tatparijñayā |</l>
	<l>bhāvaṃ bhāvyaṃ bhavet prājña abhāvaṃ ca parijñayā |</l>
	<l>tadvac chrīherukaṃ bhāvyaṃ abhāvaṃ ca parijñayā || 11 ||</l>
</lg>

<lg xml:id="HT1.1.12">
	<l><pb n="14"/>dehasthaṃ ca mahājñānaṃ sarvasaṃkalpavarjitam |</l>
	<l>vyāpakaḥ sarvavastūnāṃ dehastho 'pi na dehajaḥ || 12 ||</l>
</lg>

<p xml:id="HT1.1.13">
	<l><pb n="16"/>vajragarbha āha—he bhagavan ! vajradehe katamā nāḍyaḥ |</l>
	<l>bhagavān āha—dvātriṃśan nāḍyaḥ | dvātriṃśad bodhicittāvahā mahāsukhasthāne sravante | tāsāṃ madhye tisro nāḍyaḥ pradhānāḥ | lalanā rasanā avadhūtī ceti || 13 ||</l>
</p>

<lg xml:id="HT1.1.14">
	<l><pb n="19"/>lalanā prajñāsvabhāvena rasanopāyasaṃsthitā |</l>
	<l>avadhūtī madhyadeśe grāhyagrāhakavarjitā || 14 ||</l>
</lg>

<lg xml:id="HT1.1.15">
	<l><pb n="20"/>akṣobhyāvahā lalanā rasanā raktavāhinī |</l>
	<l>prajñācandrāvahākhyātāvadhūtī sā prakīrtitā || 15 ||</l>
</lg>

<lg xml:id="HT1.1.16">
	<l>abhedyā sūkṣmarūpā ca divyā vāmā tu vāminī |</l>
	<l>kūrmajā bhāvakī sekā doṣā viṣṭā ca mātarī || 16 ||</l>
</lg>

<lg xml:id="HT1.1.17">
	<l>śavarī śītadā coṣmā lalanāvadhūtī rasanā |</l>
	<l>pravaṇā kṛṣṇavarṇā ca surūpiṇī sāmānyā hetudāyikā || 17 ||</l>
</lg>

<lg xml:id="HT1.1.18">
	<l><pb n="21"/>viyogā premaṇī siddhā pāvakī sumanās tathā |</l>
	<l>traivṛttā kāminī gehā caṇḍikā māradārikā || 18 ||</l>
</lg>
     
<p xml:id="HT1.1.19">
	vajragarbha uvāca—etā dvātriṃśan nāḍyo bhagavan kīdṛśāḥ || 19 ||
</p>

<lg xml:id="HT1.1.20">
	<l>bhagavān āha—</l>
	<l>tribhavapariṇatāḥ sarvā grāhyagrāhakavarjitāḥ |</l>
	<l>athavā sarvopāyena bhāvalakṣaṇakalpitāḥ || 20 ||</l>
</lg>

<p xml:id="HT1.1.21">
	<pb n="22"/>saṃvarabhedaś ca kathyate—āli-kāli-candra-sūrya-prajñopāya-dharma-saṃbhoga-nirmāṇa-mahāsukha-kāya-vāk-cittam | evaṃ mayā || 21 ||
</p>

<lg xml:id="HT1.1.22">
	<l><pb n="23"/>ekāreṇa locanā devī vaṃkāreṇa māmakī smṛtā |</l>
	<l>makāreṇa pāṇḍurā ca yākāreṇa tāraṇī smṛtā || 22 ||</l>
</lg>
     
<p xml:id="HT1.1.23">
	nirmāṇacakre padmaṃ catuḥṣaṣṭidalam | dharmacakre aṣṭadalam | saṃbhogacakre ṣoḍaśadalam | mahāsukhacakre dvātriṃśaddalam | cakrasaṃkhyākrameṇa vyavasthāpanam || 23 ||
</p>

<p xml:id="HT1.1.24">
	<pb n="24"/>catvāraḥ kṣaṇāḥ—vicitra-vipāka-vimarda-vilakṣaṇāś ceti || 24 ||
</p>

<p xml:id="HT1.1.25">
	catvāry aṅgāni—sevā-upasevā-sādhanā-mahāsādhanāś ceti || 25 ||
</p>

<p xml:id="HT1.1.26">
	caturāryasatyāni—duḥkha-samudaya-nirodha-mārgāś ceti || 26 ||
</p>

<p xml:id="HT1.1.27">
	catvāri tattvāni—ātmatattvaṃ-mantratattvaṃ-devatātattvaṃ-jñānatattvaṃ ceti || 27 ||
</p>

<p xml:id="HT1.1.28">
	catvāra ānandāḥ—ānandaḥ paramānando viramānandaḥ sahajānandaś ceti || 28 ||
</p>

<p xml:id="HT1.1.29">
	catvāro nikāyāḥ—sthāvarī sarvāstivādaḥ saṃvidī mahāsaṅghī ceti || 29 ||
</p>

<p xml:id="HT1.1.30">
	<pb n="25"/>candra-sūrya āli-kāli ṣoḍaśasaṃkrāntiś catuḥṣaṣṭidaṇḍo dvātriṃśannāḍī catvāraḥ praharā evaṃ sarve catvāraḥ || 30 ||
</p>

<lg xml:id="HT1.1.31">
	<l><pb n="27"/>caṇḍālī jvalitā nābhau dahati pañcatathāgatān |</l>
	<l>dahati ca locanādīḥ dagdhe haṃ sravate śaśī || 31 ||</l>
</lg>

<p xml:id="HT1.1.col">
	iti sarvatathāgatakāyavākcittahevajrakulapaṭalaḥ prathamaḥ |
</p>
