<div n="1" type="paṭala">

<p xml:id="HT1.1.0">
</p>

<p xml:id="HT1.1.1">
	evaṃ mayā śrutam | ekasmin samaye bhagavān sarvatathāgatakāyavākcittahṛdayavajrayoṣidbhageṣu vijahāra || 1 ||
</p>

<p xml:id="HT1.1.2">
	tatra bhagavān āha sarvatathāgatakāyavākcittahṛdayaṃ bhaṭṭārakaṃ guhyātiguhyataram | aho vajragarbha sādhu sādhu mahākṛpa mahābodhisattva vajrasattvasya mahāsattvasya mahāsamayasattvasya hṛdayaṃ hevajrākhyaṃ śṛṇu || 2 ||
</p>

<lg xml:id="HT1.1.3">
	<l>vajragarbha uvāca—</l>
	<l>vajrasattvo bhavet kasmān mahāsattvo bhavet katham |</l>
	<l>samayasattvo bhavet kena kathayatu bhagavān mayi || 3 ||</l>
</lg>

<lg xml:id="HT1.1.4">
	<l>bhagavān āha—</l>
	<l>abhedyaṃ vajram ity uktam sattvaṃ tribhavasyaikatā |</l>
	<l>anayā prajñayā yuktyā vajrasattva iti smṛtaḥ || 4 ||</l>
</lg>

<lg xml:id="HT1.1.5">
	<l>mahājñānarasaiḥ pūrṇo mahāsattveti nigadyate |</l>
	<l>nityaṃ samayapravṛttatvāt samayasattvo 'bhidhīyate || 5 ||</l>
</lg>

<lg xml:id="HT1.1.5a">
</lg>

<lg xml:id="HT1.1.6">
	<l>vajragarbha uvāca—</l>
	<l>hevajraṃ tu bhavet kena īdṛśaṃ nāmasaṃgraham |</l>
	<l>hekāreṇa kim ākhyātaṃ vajreṇāpi tathā kimu || 6 ||</l>
</lg>

<lg xml:id="HT1.1.7">
	<l>bhagavān āha—</l>
	<l>hekāreṇa mahākaruṇā vajraṃ prajñā ca bhaṇyate |</l>
	<l>prajñopāyātmakaṃ tantraṃ tan me nigaditaṃ śṛṇu || 7 ||</l>
</lg>

<lg xml:id="HT1.1.8">
	<l>dṛṣṭyākrṣṭimahācchomaṃ sāmarthyaṃ bahuvidhaṃ viduḥ |</l>
	<l>stambhanoccāṭanaṃ caiva sainyastambhābhicārukam || 8 ||</l>
</lg>

<lg xml:id="HT1.1.9">
	<l>yoginīnāṃ yathānyāyam utpattisthitikāraṇam |</l>
	<l>sāmarthyaṃ jñānavijñānaṃ devatānāṃ yathodayam || 9 ||</l>
</lg>

<lg xml:id="HT1.1.10">
	<l>prathamaṃ tāvad bhaved ekaṃ herukotpattikāraṇam |</l>
	<l>bhāvenaiva vimucyante vajragarbha mahākṛpa || 10 ||</l>
</lg>

<lg xml:id="HT1.1.11">
	<l>badhyante bhāvabandhena mucyante tatparijñayā |</l>
	<l>bhāvaṃ bhāvyaṃ bhavet prājña abhāvaṃ ca parijñayā |</l>
	<l>tadvad dherukaṃ bhāvyam abhāvaṃ ca parijñayā || 11 ||</l>
</lg>

<lg xml:id="HT1.1.12">
	<l>dehasthaṃ mahājñānaṃ sarvasaṃkalpavarjitam |</l>
	<l>vyāpakaḥ sarvavastūnāṃ dehastho 'pi na dehajaḥ || 12 ||</l>
</lg>

<lg xml:id="HT1.1.13">
	<l>vajragarbha uvāca—he bhagavan | vajradehe katamā nāḍyaḥ |</l>
	<l>bhagavān āha—dvātriṃśan nāḍyaḥ | dvātriṃśad bodhicittāvahā mahāsukhasthāne sravantyaḥ | tāsāṃ madhye trīṇi nāḍyaḥ pradhānāḥ lalanā rasanāvadhūtī ceti || 13 ||</l>
</lg>

<lg xml:id="HT1.1.14">
	<l>lalanā prajñāsvabhāvena rasanopāyena saṃsthitā |</l>
	<l>avadhūtī madhyadeśe tu grāhyagrāhakavarjitā || 14 ||</l>
</lg>

<lg xml:id="HT1.1.15">
	<l>akṣobhyāvahā lalanā rasanā raktavāhā tathā |</l>
	<l>prajñācandrāvahākhyātāvadhūtī sā prakīrtitā || 15 ||</l>
</lg>

<lg xml:id="HT1.1.16">
	<l>abhedyā sūkṣmarūpā ca divyā vāmā tu vāmanī |</l>
	<l>kūrmajā bhāvakī sekā doṣā viṣṭā ca mātarā || 16 ||</l>
</lg>

<lg xml:id="HT1.1.17">
	<l>sarvarī śītadā coṣmā lalanāvadhūtī rasanā |</l>
	<l>pravaṇā kṛṣṇavarṇā ca sāmānyā hetudāyikā || 17 ||</l>
</lg>

<lg xml:id="HT1.1.18">
	<l>viyogā premaṇī siddhā pāvakī sumanās tathā |</l>
	<l>tṛvṛttā kāminī gehā caṇḍikā māradārikā || 18 ||</l>
</lg>
     
<p xml:id="HT1.1.19">
	vajragarbha uvāca—etā dvātriṃśan nāḍyo bhagavan kīdṛśāḥ || 19 ||
</p>

<lg xml:id="HT1.1.20">
	<l>bhagavān āha—</l>
	<l>tribhavapariṇatāḥ sarvā grāhyagrāhakavarjitāḥ |</l>
	<l>athavā sarvopāyena bhāvalakṣaṇakalpitāḥ || 20 ||</l>
</lg>

<p xml:id="HT1.1.21">
	saṃvarabhedaś ca kathyate | ālikāli prajñopāya | dharmasambhoganirmāṇa kāyavākcittam || 21 ||
</p>

<lg xml:id="HT1.1.22">
	<l>evaṃ mayā | ekāreṇa locanādevī vaṃkāreṇa māmakī smṛtā |</l>
	<l>makāreṇa pāṇḍarādevī yākāreṇa tāriṇī smṛtā || 22 ||</l>
</lg>
     
<p xml:id="HT1.1.23">
	nirmāṇacakre padmaṃ catuḥṣaṭidalaṃ | dharmacakre aṣṭadalam | sambhogacakre ṣoḍaśadalam | mahāsukhacakre dvātriṃśaddalaṃ | cakrasaṃkhyākrameṇa vyavasthāpanam || 23 ||
</p>

<p xml:id="HT1.1.24">
	catvāraḥ kṣaṇāḥ | vicitra vipāka vimarda vilakṣaṇa || 24 ||
</p>

<p xml:id="HT1.1.25">
	
</p>

<p xml:id="HT1.1.26">
	caturāryasatyāni | duḥkha samudaya nirodha mārga || 26 ||
</p>

<p xml:id="HT1.1.27">
	catvāni tattvāni | ātmatattvaṃ mantratattvaṃ devatātattvaṃ jñānatattvaṃ ceti || 27 ||
</p>

<p xml:id="HT1.1.28">
	catvāra ānandāḥ | ānandaḥ paramānando viramānandaḥ sahajānandaś ceti || 28 ||
</p>

<p xml:id="HT1.1.29">
	catvāro nikāyāḥ | sthāvarī sarvāstivādaḥ saṃvidī mahāsaṅghī ceti || 29 ||
</p>

<p xml:id="HT1.1.30">
	candrasūrya ālikāli | ṣoḍaśasaṃkrāntiś catuḥṣaṣṭidaṇḍo dvātriṃśannāḍī catvāraḥ praharāḥ | evaṃ sarve catvāraḥ || 30 ||
</p>

<lg xml:id="HT1.1.31">
	<l>caṇḍālī jvalitā nābhau dahati pañcatathāgatān |</l>
	<l>dahati locanādīnāṃ dagdhe haṃ sravate śaśī || 31 ||</l>
</lg>

<p xml:id="HT1.1.col">
	vajrakulapaṭalaḥ prathamaḥ ||
</p>
