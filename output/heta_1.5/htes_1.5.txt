<div n="5" type="paṭala">
<p xml:id="HT1.5.1">
	atha tattvapaṭalaṃ vyākhyāsyāmaḥ | svarūpeṇa

	<lg>
		<l>nāsti rūpaṃ na draṣṭā ca na śabdo nāpi śrotā ca ||</l>
		<l>na gandho nāpi ghrātā ca na raso nāpi rāsakaḥ ||</l>
		<l>na sparśo nāpi spraṣṭā ca na cittaṃ nāpi caittikaṃ || (1)</l>
	</lg>
</p>

<lg xml:id="HT1.5.2">
	<l><pb n="16"/>jananīṃ bhaginīṃ caiva pūjayed yogavit sadā ||</l>
	<l>naṭīṃ ca rajakīṃ vajrāṃ caṇḍālīṃ brāhmaṇīṃ tathā ||</l>
	<l>prajñopāyavidhānena pūjayet tattvavatsalaḥ || (2)</l>
</lg>

<lg xml:id="HT1.5.3">
	<l>sevitavyāḥ prayatnena yathā bhedo na jāyate ||</l>
	<l>agupte kriyate duḥkhaṃ vyāḍacaurāgnibhūcaraiḥ || (3)</l>
</lg>

<lg xml:id="HT1.5.4">
	<l>mudrāḥ pañcakulānīti kathyate mokṣahetunā ||</l>
	<l>vajreṇa mudryate 'nena mudrā tenābhidhīyate || (4)</l>
</lg>

<lg xml:id="HT1.5.5">
	<l>vajra padma tathā karma tathāgata ratnaiva ca ||</l>
	<l>kulāni pañcavidhāny āhur uttamāni mahākṛpa || (5)</l>
</lg>

<lg xml:id="HT1.5.6">
	<l>vajre ḍombī bhaven mudrā padme nartī tathaiva ca ||</l>
	<l>karmaṇi rajaky āhyātā brāhmaṇī ca tathāgate || (6)</l>
</lg>

<lg xml:id="HT1.5.7">
	<l>ratne caṇḍālinī jñeyā pañcamudrā viniścitāḥ ||</l>
	<l>tathāgatakulaṃ caitat saṃkṣepenābhidhīyate || (7)</l>
</lg>

<lg xml:id="HT1.5.8">
	<l>tathatāyāṃ gataḥ śrīmān āgataś ca tathaiva ca ||</l>
	<l>anayā prajñayā yuktyā tathāgato 'bhidhīyate || (8)</l>
</lg>

<lg xml:id="HT1.5.9">
	<l>kulāni ṣaḍvidhāny āhuḥ saṃkṣepeṇa tu pañcadhā ||</l>
	<l>paścāc ca traividhyaṃ yānti kāyavakcittabhedena || (9)</l>
</lg>

<lg xml:id="HT1.5.10">
	<l>kulānāṃ pañcabhūtānāṃ pañcaskandhasvarūpiṇāṃ ||</l>
	<l>kulyate gaṇyate 'nena kulam ity abhidhīyate || (10)</l>
</lg>

<lg xml:id="HT1.5.11">
	<l>nāsti bhāvako na bhāvo 'sti mantran nāsti na devatā ||</l>
	<l>saṃsthitau mantradevau ca niḥprapañcasvabhāvataḥ || (11)</l>
</lg>

<lg xml:id="HT1.5.12">
	<l>vairocanākṣobhyāmoghaś ca ratnārolic ca sātvikaḥ ||</l>
	<l>brahmā viṣṇuḥ śivaḥ sarvo vibuddhas tattvam ucyate || (12)</l>
</lg>

<lg xml:id="HT1.5.13">
	<l>brahmā nirvṛtito buddhaḥ viṣanād viṣṇur ucyate ||</l>
	<l>śivaḥ sadā sukalyāṇāt sarvaḥ sarvātmani sthitaḥ || (13)</l>
</lg>

<lg xml:id="HT1.5.14">
	<l>satsukhatvena tattvaṃ ca vibuddho bodhanāt rateḥ ||</l>
	<l>dehe saṃbhavatīty asmād devateti nigadyate || (14)</l>
</lg>

<lg xml:id="HT1.5.15">
	<l>bhago 'syāstīti buddhasya bhagavān iti kathyate ||</l>
	<l>bhagāni ṣaḍvidhāny āhur aiśvaryādiguṇākhilāḥ ||</l>
	<l>athavā kleśādimārāṇāṃ bhañjanād bhagavān iti || (15)</l>
</lg>

<lg xml:id="HT1.5.16">
	<l>jananī bhaṇyate prajñā janayati yasmāj jagat ||</l>
	<l>bhaginīti tathā prajñā vibhāgaṃ darśayed yathā || (16)</l>
</lg>

<lg xml:id="HT1.5.17">
	<l>rajakīti duhitā ca nartakī ca prakathyate ||</l>
	<l>rañjanāt sarvasattvānāṃ rajakīti tathā smṛtā || (17)</l>
</lg>

<lg xml:id="HT1.5.18">
	<l>guṇasya duhanāt prajñā duhitā ca nigadyate ||</l>
	<l>nartakī bhaṇyate prajñā cañcalatvān mahākṛpa ||</l>
	<l><pb n="18"/>asparśā bhagavatī yasmāt tasmād ḍombī prakathyate || (18)</l>
</lg>

<lg xml:id="HT1.5.19">
	<l>jalpanaṃ japam ākhyātam ālikālyoḥ prajalpanāt |</l>
	<l>maṇḍalaṃ pādalekhaḥ syān malanād maṇḍalam ucyate || (19)</l>
</lg>

<lg xml:id="HT1.5.20">
	<l>karasphoṭo bhaven mudrā aṅgulyāmoṭanaṃ tathā |</l>
	<l>tad dhyeyaṃ cintitaṃ yac ca dhyeyaṃ yasmād vicintanaṃ || (20)</l>
</lg>

<lg xml:id="HT1.5.21">
	<l>pitari prāptaṃ yat saukhyaṃ tat sukhaṃ bhujyate svayaṃ ||</l>
	<l>maraṇaṃ yena sukhena tat sukhaṃ dhyānam ucyate || (21)</l>
</lg>

<p xml:id="HT1.5.col">
	tattvapaṭalaḥ pañcamaḥ ||
</p>
