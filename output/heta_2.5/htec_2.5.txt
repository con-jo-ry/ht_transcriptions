	<l>atha vajrī mahārājā hevajraḥ sarvadaḥ prabhuḥ |</l>

</lg>

<lg xml:id="HT2.5.2">
	<l>sukhāvatyāṃ samāsīnaḥ sarvākārasvarūpataḥ |</l>
	<l>cittavajrasya bījena niṣpanno maṇḍaleśvaraḥ || 2 ||</l>
</lg>

<lg xml:id="HT2.5.3">
	<l>ṣoḍaśabhujam aṣṭāsyaṃ catuścaraṇaṃ bhayānakam |</l>
	<l>kapālamālinaṃ vīraṃ nairātmyāśliṣṭakandharam |</l>
	<l>pañcamudrādharaṃ devaṃ nairātmyā pṛcchate svayam || 3 ||</l>
</lg>

<lg xml:id="HT2.5.4">
	<l>asmaccakraṃ tvayā kathitaṃ pañcadaśaparivāritam |</l>
	<l>tvadīyaṃ maṇḍalaṃ kīdṛk prāg na jñātaṃ mayā prabho || 4 ||</l>
</lg>

<lg xml:id="HT2.5.5">
	<l>cumbayitvā tu nairātmyāṃ kṣiptvā vajraṃ kapālake |</l>
	<l>mardayitvā stanau devo maṇḍalaṃ saṃprakāśate || 5 ||</l>
</lg>

<lg xml:id="HT2.5.6">
	<l>cakraṃ pūrvaṃ yathā kathitaṃ hārārdhahāraśobhitam |</l>
	<l>catuṣkoṇaṃ caturdvāraṃ vajrasūtrair alaṃkṛtam || 6 ||</l>
</lg>

<lg xml:id="HT2.5.7">
	<l>tatra madhye 'haṃ vidyāt tvayā sārdhaṃ varānane |</l>
	<l>mahārāgānurāgena sahajānandasvarūpataḥ || 7 ||</l>
</lg>

<lg xml:id="HT2.5.8">
	<l>aṣṭāsyaṃ catuścaraṇaṃ bhujaṣoḍaśabhūṣitam |</l>
	<l>caturmārasamākrāntaṃ bhayasyāpi bhayānakam || 8 ||</l>
</lg>

<lg xml:id="HT2.5.9">
	<l>muṇḍamālākṛtaṃ hāraṃ sūryasthaṃ tāṇḍavānvitam |</l>
	<l>viśvavajradharaṃ mūrdhni kṛṣṇavarṇaṃ bhayānakam || 9 ||</l>
</lg>

<lg xml:id="HT2.5.10">
	<l>hūṁkāraṃ sphārayet svamukhād bhasmoddhūlitavigraham |</l>
	<l>ratidvandvasamāpannaṃ nairātmyāsahasaṃpuṭam || 10 ||</l>
</lg>

<lg xml:id="HT2.5.11">
	<l>nistaraṅgasukhāvāptaṃ nistaraṅgasvarūpiṇam |</l>
	<l>mūlamukhaṃ mahākṛṣṇaṃ dakṣiṇaṃ kundasannibham || 11 ||</l>
</lg>

<lg xml:id="HT2.5.12">
	<l>vāmaṃ raktaṃ mahābhīmam ūrdhvāsyaṃ vikarālinam |</l>
	<l>caturviṃśatinetrāḍhyaṃ śeṣāsyā bhṛṅgasannibhāḥ || 12 ||</l>
</lg>

<lg xml:id="HT2.5.13">
	<l>tvayā mayā pure ramye krīḍatā ratinirbharaiḥ |</l>
	<l>niḥsṛtā indradig gaurī pūrvadvāre susaṃsthitā || 13 ||</l>
</lg>

<lg xml:id="HT2.5.14">
	<l>manthamanthānayogena caurikā niḥsṛtā punaḥ |</l>
	<l>niḥsṛtya dakṣiṇe dvāre caurī sā dvārapālikā || 14 ||</l>
</lg>

<lg xml:id="HT2.5.15">
	<l>bolakakkolayogena vetālī niḥsṛtā punaḥ |</l>
	<l>niḥsṛtya paścime dvāre niṣaṇṇā* mārabhañjanī || 15 ||</l>
</lg>

<lg xml:id="HT2.5.16">
	<l>mahādvandvasamāpattau niḥsṛtā ghasmarī punaḥ |</l>
	<l>niḥsṛtya uttare dvāre niṣaṇṇā* ghorarūpiṇī || 16 ||</l>
</lg>

<lg xml:id="HT2.5.17">
	<l>dvayor gharṣaṇasaṃyogān niḥsṛtā pukkasī punaḥ |</l>
	<l>niḥsṛtyaiśānakoṇe ca niṣaṇṇā raudrarūpiṇī || 17 ||</l>
</lg>

<lg xml:id="HT2.5.18">
	<l>punar manthānayogena śabarī pāvakakoṇake |</l>
	<l>caṇḍālī rākṣasāyāṃ ḍombī mārutakoṇake || 18 ||</l>
</lg>

<lg xml:id="HT2.5.19">
	<l>tato vajrī mahārāgād drutaṃ bhūtaṃ savidyayā |</l>
	<l>codayanti tato devyo nānāgītopahārataḥ || 19 ||</l>
</lg>

<lg xml:id="HT2.5.20"> 
	<l>uṭṭha bharāḍo karuṇamaṇu pukkasi mahu paritāhi |</l>
	<l>mahāsuhajoeṃ kāma mahu chaḍḍahi suṇṇasahāvi || 20 ||</l>
</lg>

<lg xml:id="HT2.5.21">
	<l>tojjha vihuṇṇe marami haüṃ uṭṭhahi tuhu hevajja |</l>
	<l>chaḍḍahi sunnasahāvaḍā savariha sihyāu kajja || 21 ||</l>
</lg>

<lg xml:id="HT2.5.22">
	<l>loa nimantia suraapahu suṇṇe acchasi kīsa |</l>
	<l>hauṃ caṇḍāli viṇṇanami taï viṇṇa ḍahami na dīsa || 22 ||</l>
</lg>

<lg xml:id="HT2.5.23">
	<l>indī ālī uṭṭha tuhuṃ hauṃ jānāmi ttuha cittaḥ |</l>
	<l>ambhe ḍombī cheamaṇḍa mā kara karuṇavicchittaḥ || 23 ||</l>
</lg>

<lg xml:id="HT2.5.24"> 
	<l>hastyaśvakharagāvoṣṭramanujaśarabhotukas tathā |</l>
	<l>dakṣiṇāṣṭakapāleṣu kramair jñeyā dvipādayaḥ || 24 ||</l>
</lg>

<lg xml:id="HT2.5.25">
	<l>pṛthivī varuṇa vāyuś ca tejaś candrārka eva ca |</l>
	<l>antako dhanadaś caiva vāmāṣṭakapālake || 25 ||</l>
</lg>

<lg xml:id="HT2.5.26">
	<l>śṛṅgāravīrabībhatsaraudrahāsyabhayānakaiḥ |</l>
	<l>karuṇādbhutaśāntaiś ca navanāṭyarasair yutaḥ || 26 ||</l>
</lg>

<lg xml:id="HT2.5.27">
	<l>aṃhūṃbhyāṃ ca mahāvajrī utthito dravamūrtitaḥ |</l>
	<l>caraṇān sphārayan bhūmau tarjayañ ca surāsurān || 27 ||</l>
</lg>

<lg xml:id="HT2.5.28">
	<l>gaṁ caṁ vaṁ ghaṁ puṁ śaṁ caṁ ḍaṁ bījais tu utsṛjed āsām |</l>
	<l>adhipatiratibījābhyāṃ hūṃaṃbhyāṃ jvālākarālanīlābhyām || 28 ||</l>
</lg>

<lg xml:id="HT2.5.29">
	<l>mātṛcakre pure ramye bhāvayed īdṛśaṃ prabhum |</l>
	<l>kṛṣṇavarṇaṃ mahāghoraṃ nairātmyāsukhadāyakam || 29 ||</l>
</lg>

<lg xml:id="HT2.5.30">
	<l>gauryā hi dakṣiṇe kartry avasavye rohitas tathā |</l>
	<l>kṛpīṭaṃ dakṣiṇe cauryā vāme pāṇau varāhakam || 30 ||</l>
</lg>

<lg xml:id="HT2.5.31">
	<l>vettālyā dakṣiṇe kūrmaṃ vāme padmabhājanam |</l>
	<l>ghasmaryā dakṣiṇe sarpam vāmena yogapātrikā || 31 ||</l>
</lg>

<lg xml:id="HT2.5.32">
	<l>pukkasyā dakṣiṇe siṃhaṃ vāme parśus tathā |</l>
	<l>śabaryā dakṣiṇe bhikṣur vāme khikkhirikā tathā || 32 ||</l>
</lg>

<lg xml:id="HT2.5.33">
	<l>caṇḍālyā dakṣiṇe cakraṃ vāmena lāṅgalas tathā |</l>
	<l>ḍombyā dakṣiṇe vajram avasavye tarjanī tathā || 33 ||</l>
</lg>

<lg xml:id="HT2.5.34">
	<l>ardhaparyaṅkanāṭyasthā gauryādyā dvibhujā matāḥ |</l>
	<l>trinetrā ūrdhvakeśāś ca pañcamudrāvibhūṣitāḥ || 34 ||</l>
</lg>

<lg xml:id="HT2.5.35">
	<l>kṛṣṇavarṇā bhaved gaurī caurī māñjiṣṭhasannibhā |</l>
	<l>vettālī taptahemābhā ghasmarī maraktopamā || 35 ||</l>
</lg>

<lg xml:id="HT2.5.36">
	<l>pukkasī indranīlābhā śavarī candramaṇiprabhā |</l>
	<l>caṇḍālī ca nabhaḥśyāmā ḍombī karcūrā matā || 36 ||</l>
</lg>

<lg xml:id="HT2.5.37">
	<l>brahmendropendrarudrāś ca vaivasvato vittanāyakaḥ |</l>
	<l>nairṛtir vemacitrī ca gauryādīnāṃ tu viṣṭaram || 37 ||</l>
</lg>

<lg xml:id="HT2.5.38">
	<l>bolakaṃ cūṣayitvā tu bhagavantaṃ pūjya bhaktitaḥ |</l>
	<l>nairātmyā pṛcchate mantraṃ gāḍhāliṅganacumbanaiḥ || 38 ||</l>
</lg>

<lg xml:id="HT2.5.39">
	<l>strīṇāṃ vaśyakaraṃ mantraṃ duṣṭānāṃ tarjanaṃ tathā |</l>
	<l>nāgākṣepakaraṃ caiva devāsuravimardanam || 39 ||</l>
</lg>

<lg xml:id="HT2.5.40">
	<l>tad ahaṃ kathayāmy eṣa śṛṇu devi sukhaṃdade |</l>
	<l>buddheṣu bodhisattveṣu mayā nānyatra deśitam || 40 ||</l>
</lg>

<lg xml:id="HT2.5.41">
	<l>asya mantrasya yad udbhūtaṃ vajrasattvena yat kṛtam |</l>
	<l>bibhemi sutarāṃ devi uparodhāt tvayi kathyate || 41 ||</l>
</lg>

<lg xml:id="HT2.5.42">
	<l>maṇḍalaṃ vartayitvā tu jvālāmālākarālinam |</l>
	<l>abhiṣekaṃ vajragarbhasya dātuṃ kṛṣṭā tilottamā || 42 ||</l>
</lg>

<lg xml:id="HT2.5.43">
	<l>ayutajāpena spaṣṭena dīrghanādena cāruṇā |</l>
	<l>hevajrayogayuktenākṛṣyante sarvayoṣitaḥ || 43 ||</l>
</lg>

<lg xml:id="HT2.5.44">
	<l>lakṣajāpena yogātmā sarvakarma karoty asau |</l>
	<l>hekāravajrayogena nirviśaṅkena cetasā || 44 ||</l>
</lg>

<lg xml:id="HT2.5.45">
	<l>vedānām ādimaṃ caiva ardhacandrendubhūṣitam |</l>
	<l>paścād aṣṭānanāyeti piṅgordhvakeśavartmane || 45 ||</l>
</lg>

<p xml:id="HT2.5.46">
	caturviṃśatinetrāya tadanu ṣoḍaśabhujāya kṛṣṇajīmūtavapuṣe kapālamālānaikadhāriṇe adhmātakrūracittāya ardhendudaṃṣṭriṇe || 46 ||
</p>
 
<p xml:id="HT2.5.47">
	māraya māraya kāraya kāraya garjaya garjaya tarjaya tarjaya śoṣaya śoṣaya saptasāgarān bandha bandha nāgāṣṭakān gṛhṇa gṛhṇa śatrūn ha hā hi hī hu hū he hai ho hau haṃ haḥ phaṭ svāhā || 47 ||
</p>

<lg xml:id="HT2.5.48">
	<l>tatra tuṣṭā tu sā devī manthamanthānayogataḥ |</l>

</lg>

<lg xml:id="HT2.5.49">
	<l>śāstā tatra mahājñānī maṇḍalaṃ likhate svayam |</l>

</lg>

<lg xml:id="HT2.5.50">

<lg xml:id="HT2.5.53">
