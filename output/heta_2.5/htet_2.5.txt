<lg xml:id="HT2.5.1">
	<l><pb n="198"/>*atha vajrī mahārājā hevajraḥ sarvadaḥ prabhuḥ |</l>

</lg>

<lg xml:id="HT2.5.2">

	<l>cittavajrasya bījena niṣpanno maṇḍaleśvaraḥ* || 2 ||</l>
</lg>

<lg xml:id="HT2.5.3">
	<l>ṣoḍaśabhujam aṣṭāsyaṃ catuścaraṇaṃ bhayānakam |</l>
	<l>kapālamālinaṃ vīraṃ nairātmyāśliṣṭakandharam |</l>
	<l>pañcamudrādharaṃ devaṃ nairātmyā pṛcchati svayam || 3 ||</l>
</lg>

<lg xml:id="HT2.5.4">
	<l>asmaccakraṃ tvayā kathitaṃ pañcadaśaparivāritam |</l>
	<l>tvadīyaṃ maṇḍalaṃ kīdṛk prāg na jñātaṃ mayā prabho || 4 ||</l>
</lg>

<lg xml:id="HT2.5.5">
	<l>cumbayitvā tu nairātmyāṃ kṣiptvā vajraṃ kapālake |</l>

</lg>

<lg xml:id="HT2.5.6">
	<l><pb n="199"/>cakraṃ pūrvaṃ yathā kathitaṃ hārārdhahāraśobhitam |</l>
	<l>catuṣkoṇaṃ caturdvāraṃ vajrasūtrair alaṃkṛtam || 6 ||</l>
</lg>

<lg xml:id="HT2.5.7">
	<l>tatra madhye 'haṃ vidyate tvayā sārdhaṃ varānane |</l>
	<l>mahārāgānurāgena sahajānandasvarūpataḥ || 7 ||</l>
</lg>

<lg xml:id="HT2.5.8">
	<l>aṣṭāsyaṃ catuścaraṇaṃ bhujaṣoḍaśabhūṣitam |</l>
	<l>caturmārasamākrāntaṃ bhayasyāpi bhayānakam || 8 ||</l>
</lg>

<lg xml:id="HT2.5.9">
	<l>muṇḍamālākṛtahāraṃ sūryasthaṃ tāṇḍavānvitam |</l>

</lg>

<lg xml:id="HT2.5.10">
	<l>hūṁ kāraṃ sphārayen mukhād bhasmoddhūlitavigraham |</l>
	<l>ratidvandvasamāpannaṃ nairātmyā saha saṃyutam || 10 ||</l>
</lg>

<lg xml:id="HT2.5.11">
	<l><pb n="200"/>nistaraṅgasukhāvāptaṃ nistaraṅgasvarūpiṇam |</l>
	<l>mūlamukhaṃ mahākṛṣṇaṃ dakṣiṇaṃ kundasannibham || 11 ||</l>
</lg>

<lg xml:id="HT2.5.12">
	<l>vāmaṃ raktaṃ mahābhīmaṃ mūrdhāsyaṃ vikarālinam |</l>

</lg>

<lg xml:id="HT2.5.13">
	<l>tvayā mayā pure ramye krīḍatā ratinirbharaiḥ |</l>

</lg>

<lg xml:id="HT2.5.14">

</lg>

<lg xml:id="HT2.5.15">
	<l>bolakakkolayogena vetālī niḥsṛtā punaḥ |</l>

</lg>

<lg xml:id="HT2.5.16">
	<l>mahādvandvasamāpattau niḥsṛtā ghasmarī punaḥ |</l>

</lg>

<lg xml:id="HT2.5.17">

</lg>

<lg xml:id="HT2.5.18">
	<l>punar manthānayogena śavarī pāvakakoṇake |</l>

</lg>

<lg xml:id="HT2.5.19">

</lg>

<lg xml:id="HT2.5.20"> 
	<l>uṭṭha bharāḍo karuṇamaṇḍa pukkasī mahuṃ paritāhiṃ |</l>

</lg>

<lg xml:id="HT2.5.21">
	<l><pb n="202"/>tohyā vihuṇṇe marami hahuṃ uṭṭhehiṃ tuhuṃ hevajja |</l>
	<l>chaḍḍahi sunnasabhāvaḍā śavaria sihyāu kajja || 21 ||</l>
</lg>

<lg xml:id="HT2.5.22">
	<l>loa nimantia suraapahu suṇṇe acchasi kīsa |</l>
	<l>hauṃ caṇḍāli viṇṇanami taï viṇṇa ḍahami na dīsa || 22 ||</l>
</lg>

<lg xml:id="HT2.5.23">
	<l>indī ālī uṭṭha tuhuṃ hauṃ jānāmi ttuha cittaḥ |</l>

</lg>

<lg xml:id="HT2.5.24"> 
	<l><pb n="203"/>hastyaśvakharagāvuṣṭramanujasarabhautukas tathā |</l>
	<l>dakṣiṇāṣṭakapāleṣu kramair jñeyā dvipādayaḥ || 24 ||</l>
</lg>

<lg xml:id="HT2.5.25">
	<l>pṛthivī varuṇa vāyuś ca tejaś candrārka eva ca |</l>
	<l>antako dhanadaś caiva vāmakāṣṭakapālake || 25 ||</l>
</lg>

<lg xml:id="HT2.5.26">
	<l>śṛṅgāravīrabībhatsaraudra hāsya bhayānakaiḥ |</l>
	<l>karuṇādbhutaśāntaiś ca ravanāṭyarasair yutam || 26 ||</l>
</lg>

<lg xml:id="HT2.5.27">
	<l>aṃ hūṃ bhyāṃ ca mahāvajrī utthito dravamūrtitaḥ |</l>
	<l>caraṇān sphārayan bhūmau tarjayañ ca surāsurān || 27 ||</l>
</lg>

<lg xml:id="HT2.5.28">
	<l><pb n="205"/>gaṁ caṁ vaṁ ghaṁ paṁ śaṁ laṁ ḍaṁ bījais tu sṛjed āsām |</l>
	<l>adhipatiratibījābhyāṃ hūṃ-aṃ-bhyāṃ jvālākarālanīlābhyām || 28 ||</l>
</lg>

<lg xml:id="HT2.5.29">
	<l>mātṛ cakre pure ramye bhāvayed īdṛśaṃ prabhum |</l>
	<l>kṛṣṇavarṇamahāghoraṃ nairātmyāsukhadāyakam || 29 ||</l>
</lg>

<lg xml:id="HT2.5.30">
	<l>gauryā hi dakṣiṇe kartry avasavye rohitas tathā |</l>
	<l>kṛpīṭaṃ dakṣiṇe cauryā vāme pāṇau varāhakam || 30 ||</l>
</lg>

<lg xml:id="HT2.5.31">
	<l>vetālyā dakṣiṇe kūrmaṃ vāme padmabhājanam |</l>

</lg>

<lg xml:id="HT2.5.32">
	<l>pukkasyā dakṣiṇe siṃhaṃ vāme paraśus tathā |</l>

</lg>

<lg xml:id="HT2.5.33">
	<l><pb n="206"/>caṇḍālyā dakṣiṇe cakraṃ vāmena lāṅgalaṃ tathā |</l>

</lg>

<lg xml:id="HT2.5.34">

</lg>

<lg xml:id="HT2.5.35">

</lg>

<lg xml:id="HT2.5.36">

</lg>

<lg xml:id="HT2.5.37">
	<l>brahmendropendrarudrāś ca vaivasvata vināyakaḥ |</l>

</lg>

<lg xml:id="HT2.5.38">
	<l><pb n="207"/>bolakaṃ bhūṣayitvā tu bhagavantaṃ pūjya bhaktitaḥ |</l>

</lg>

<lg xml:id="HT2.5.39">

</lg>

<lg xml:id="HT2.5.40">
	<l>tad ahaṃ kathayāmy eṣa śṛṇu devi sukhaṃdade |</l>
	<l>buddheṣu bodhisattveṣu mayā nānyatra deśitam || 40 ||</l>
</lg>

<lg xml:id="HT2.5.41">
	<l>asya mantrasya yad bhūtaṃ vajrasattvena yat kṛtam |</l>

</lg>

<lg xml:id="HT2.5.42">
	<l>maṇḍalaṃ vartayitvā tu jvālāmālākarālinam |</l>

</lg>

<lg xml:id="HT2.5.43">

<lg xml:id="HT2.5.44">

</lg>

<lg xml:id="HT2.5.45">
	<l>vedānām ādimaṃ caivārdhendubindubhūṣitam |</l>

</lg>

<p xml:id="HT2.5.46">

	adhyāntakrūracittāya ardhendudaṃṣṭriṇe || 46 ||
</p>
 
<p xml:id="HT2.5.47">

</p>
