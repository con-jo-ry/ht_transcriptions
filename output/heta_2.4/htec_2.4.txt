<lg xml:id="HT2.4.76">
	<l>caṇḍālaveṇukārādyā māraṇārthārthacittakāḥ |</l>
	<l>te 'pi hevajram āgamya sidhyante nātra saṃśayaḥ || 76 ||</l>
</lg>

<lg xml:id="HT2.4.77">
	<l>ajñānenāvṛtā bālā imāṃ gatim ajānakāḥ |</l>
	<l>saṃsaranti ca te mūḍhāḥ ṣaḍgatau bhavacārake || 77 ||</l>
</lg>

<lg xml:id="HT2.4.78">
	<l>upāyaṃ prāpya hevajram vajragarbha mahāmahaḥ |</l>
	<l>viśodhayanti ye viṣayān lapsyante te hy anuttaram || 78 ||</l>
</lg>

<lg xml:id="HT2.4.79">
	<l>vajragarbha āha—</l>
	<l>pṛthivī pukkasī khyātā katham akṣobhyamudraṇaṃ |</l>
	<l>mohaṃ yasmāt kakkhaṭatvaṃ kāyo vairocano mataḥ || 79 ||</l>
</lg>

<lg xml:id="HT2.4.80">
	<l>bhagavān āha |</l>
	<l>kāyaṃ vihāya cittasya nānyatra laḍitaṃ bhavet |</l>
	<l>tasmād vairocanaḥ cittaṃ kāyaṃ cittena mudrayet || 80 ||</l>
</lg>

<lg xml:id="HT2.4.81">
	<l>vajragarbha āha—</l>
	<l>abdhātuḥ śavarī khyātā akṣobhyaṃ dravarūpakam |</l>
	<l>śavarī akṣobhyamudreṇa mudraṇaṃ yujyate prabho || 81 ||</l>
</lg>

<lg xml:id="HT2.4.82">
	<l>bhagavān āha—</l>
	<l>cittaṃ vihāya kāyasya sthitir anyā na dṛśyate |</l>
	<l>tasmāc cittaṃ bhaven mohaṃ cittaṃ mohena mudrayet || 82 ||</l>
</lg>

<lg xml:id="HT2.4.83">
	<l>vajragarbha āha—</l>
	<l>tejaś caṇḍālinī khyātā kathaṃ ratnena mudraṇam |</l>
	<l>yujyate rāgamudreṇa caṇḍālyā nānyamudraṇam || 83 ||</l>
</lg>

<lg xml:id="HT2.4.84">
	<l>bhagavān āha—</l>
	<l>rāgaṃ raktaṃ yataḥ khyātaṃ raktaṃ ca ratnasambhavam |</l>
	<l>tejo raktasvarūpatvād rāgaṃ piśunena mudrayet || 84 ||</l>
</lg>

<lg xml:id="HT2.4.85">
	<l>vajragarbha āha—</l>
	<l>yasmād ḍombinī vāyur amoghaṃ vāyurūpakam |</l>
	<l>ḍombiny amoghamudreṇa mudraṇaṃ yujyate prabho || 85 ||</l>
</lg>

<lg xml:id="HT2.4.86">
	<l>bhagavān āha—</l>
	<l>rāgaṃ hitvā īrṣyāyā na syād anyatra saṃbhavaḥ |</l>
	<l>tasmād rāgasya mudreṇa ḍombinīṃ mudrayed budhaḥ || 86 ||</l>
</lg>

<lg xml:id="HT2.4.87">
	<l>rūpaṃ yasmāt kakkhaṭatvaṃ gauryā vairocano mataḥ |</l>
	<l>pūrvoktenaiva nyāyena gaurī citteśenaiva mudrayet || 87 ||</l>
</lg>

<lg xml:id="HT2.4.88">
	<l>caurīṃ tenaiva nyāyena vettālīṃ ca tathaiva ca |</l>
	<l>ghasmarīṃ ca tayā yuktyā mudraṇam aviparītataḥ || 88 ||</l>
</lg>

<lg xml:id="HT2.4.89">
	<l>samāpattau sthite deve hevajre vajradhāriṇi |</l>
	<l>tatra pṛcchati nairātmyā sattvārthāya mahābalim || 89 ||</l>
</lg>

<lg xml:id="HT2.4.90">
	<l>evaṃkāre samāsīno vajrasattvo diśed balim |</l>
	<l>sattvānāṃ prāṇarakṣāya vighnād vināyakād api || 90 ||</l>
</lg>

<lg xml:id="HT2.4.91">
	<l>oṁ inda jama jala jakkha bhuda vahni vāyu rakkha |</l>
	<l>canda sujja māda bappa talapātāle aṭṭhasappa sāhā || 91 ||</l>
</lg>

<lg xml:id="HT2.4.92">
	<l>evaṃ baliṃ bhuñja jiṅgha phulladhūpa maṃsa viṅgha |</l>
	<l>ambha kajja savva sādha khanti kuṇi pheḍa gāda || 92 ||</l>
</lg>

<p xml:id="HT2.4.93">
	oṁ akāro mukhaṃ sarvadharmāṇām ādyanutpannatvāt | oṁ āḥ hūṁ phaṭ svāhā || 93 ||
</p>

<lg xml:id="HT2.4.94">
	<l>anena balinā yadi sarvabhūtān pūjāṃ prakurvanti śubhāya yoginaḥ |</l>
	<l>bhavet tadā teṣu sukhaṃ tv anāvilaṃ devāś ca tuṣyanti jagatsubhūtayaḥ || 94 ||</l>
</lg>

<lg xml:id="HT2.4.95">
	<l>vaśyābhicāraṃ ripusainyanāśanam uccāṭanamāraṇākarṣaṇaṃ ca |</l>
	<l>śāntisukhaṃ pauṣṭikaṃ bhavec ca dadyāt baliṃ yadīha bhūtagaṇāya śaśvat || 95 ||</l>
</lg>

<lg xml:id="HT2.4.96">
	<l>vajragarbha āha—</l>
	<l>khecarī kena mudreṇa bhūcarī kasya mudrataḥ |</l>
	<l>karttavyaṃ mudraṇaṃ bhagavan prāg na jñātaṃ mayā prabho || 96 ||</l>
</lg>

<lg xml:id="HT2.4.97">
	<l>bhagavān āha—</l>
	<l>triguhyaṃ cakramadhye tu kāyavākcittabhedataḥ |</l>
	<l>adhordhvamadhyamaṃ sthānaṃ cakramadhye vyavasthitam || 97 ||</l>
</lg>

<lg xml:id="HT2.4.98">
	<l>bhūcarī kāyamudrī syād adhomukhī kāyavajriṇī |</l>
	<l>khecarī rāgamudrī ca ūrdhvamukhī vāgvajriṇī || 98 ||</l>
</lg>

<lg xml:id="HT2.4.99">
	<l>cittavajrī ca nairātmyā cittaṃ nairātmyarūpakam |</l>
	<l>cittaṃ madhyamakaṃ sthānaṃ nairātmyā tena madhyajā || 99 ||</l>
</lg>

<lg xml:id="HT2.4.100">
	<l>kulāni ṣaḍvidhāny āhur vistareṇa prakāśanāt |</l>
	<l>trividhaṃ pañcavidhaṃ caiva kathyate śṛṇu yoginī || 100 ||</l>
</lg>

<p xml:id="HT2.4.101">
	akṣobhya-vairocana-ratnasambhava-amitaprabha-amoghasiddhi-vajrasattvaḥ | dveṣa-moha-piśuna-rāga-īrṣyā-saukhyam | śuddhyānayānukramato hi bhāvyāḥ | 101 ||
</p>

<lg xml:id="HT2.4.102">	
	<l>vihāya vajrasattvākhyaṃ paścāt pañcavidhaṃ kulam |</l>
	<l>tad anu yāti traividhyaṃ mohadveṣarāgakaiḥ || 102 ||</l>
</lg>

<lg xml:id="HT2.4.103">
	<l>kulam ekaṃ tu citteśam akṣobhyadveṣarūpiṇam |</l>
	<l>dveṣavajra prabhāvo 'yaṃ kulaṃ ṣaṭ pañcakaṃ matam || 103 ||</l>
</lg>

	hevajrasarvatantramudraṇapiṇḍārtho nāma caturthaḥ paṭalaḥ ||
