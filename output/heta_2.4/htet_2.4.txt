	<l>na buddho labhate 'nyatra lokadhātuṣu kutracit |</l>
	<l>cittam eva hi saṃbuddho na buddho 'nyatra darśitaḥ || 75 ||</l>
</lg>

<lg xml:id="HT2.4.76">
	<l>caṇḍālaceṇḍakārādyā māraṇārthārthacittakāḥ |</l>
	<l>te 'pi hevajram āgamya sidhyante nātra saṃśayaḥ || 76 ||</l>
</lg>

<lg xml:id="HT2.4.77">
	<l>ajñānenāvṛtā bālā imāṃ gatim ajānakāḥ |</l>
	<l>saṃsaranti ca te mūḍhāḥ ṣaḍgatau bhavacārake || 77 ||</l>
</lg>

<lg xml:id="HT2.4.78">
	<l><pb n="191"/>upāyaṃ prāpya hevajram vajragarbha mahākṛpa |</l>
	<l>viśodhayanti viṣayān lapsyante te hy anuttaram || 78 ||</l>
</lg>

<lg xml:id="HT2.4.79">
	<l>vajragarbha āha—</l>
	<l>pṛthivī pukkasī khyātā katham akṣobhyamudraṇam |</l>
	<l>mohaṃ yasmāt kakkhāṭatvaṃ kāyo vairocano mataḥ |</l>
	<l>*pukkasī mohamudraṇaṃ mudraṇaṃ yujyate prabho* || 79 ||</l>
</lg>

<lg xml:id="HT2.4.80">
	<l>bhagavān āha—</l>
	<l>kāyaṃ vihāya cittasya nānyatra lalitaṃ bhavet |</l>
	<l>tasmād vairocanaś cittaṃ kāyaṃ cittena mudrayet || 80 ||</l>
</lg>

<lg xml:id="HT2.4.81">
	<l><pb n="192"/>vajragarbha āha—</l>
	<l>abdhātuḥ śavarī khyātā akṣobhyo dravarūpakaḥ |</l>
	<l>śavarī akṣobhyamudreṇa mudraṇaṃ yujyate prabho || 81 ||</l>
</lg>

<lg xml:id="HT2.4.82">
	<l>bhagavān āha—</l>
	<l>cittaṃ vihāya kāyasya sthitir anyā na dṛśyate |</l>
	<l>tasmāc cittaṃ bhavet mohaṃ cittaṃ mohena mudrayet || 82 ||</l>
</lg>

<lg xml:id="HT2.4.83">
	<l>vajragarbha āha—</l>
	<l>tejaś caṇḍālinī khyātā kathaṃ ratnena mudraṇam |</l>
	<l>yujyate rāgamudreṇa caṇḍālyā nānyamudraṇam || 83 ||</l>
</lg>

<lg xml:id="HT2.4.84">
	<l>bhagavān āha—</l>
	<l>rāgo raktaṃ yataḥ khyātaṃ raktaṃ ca ratnasambhavaḥ |</l>
	<l>tejo raktasvabhāvatvād rāgaṃ piśunena mudrayet || 84 ||</l>
</lg>

<lg xml:id="HT2.4.85">
	<l>vajragarbha āha—</l>
	<l>yasmād ḍombinī vāyur amogho vāyurūpakaḥ |</l>
	<l>ḍombiny amoghamudreṇa mudraṇaṃ yujyate prabho || 85 ||</l>
</lg>

<lg xml:id="HT2.4.86">
	<l><pb n="193"/>bhagavān āha—</l>
	<l>rāgaṃ hitvā īrṣyāyā na syād anyatra saṃbhavaḥ |</l>
	<l>tasmād rāgasya mudreṇa ḍombinīṃ mudrayed budhaḥ || 86 ||</l>
</lg>

<lg xml:id="HT2.4.87">
	<l>rūpaṃ yasmāt kakkhāṭatvaṃ gauryā vairocano mataḥ |</l>
	<l>pūrvoktenaiva nyāyena citteśenaiva mudrayet || 87 ||</l>
</lg>

<lg xml:id="HT2.4.88">
	<l>caurīṃ tenaiva nyāyena vetālīñ ca tathaiva ca |</l>
	<l>ghasmarīṃ ca tayā yuktyā mudraṇam aviparītataḥ || 88 ||</l>
</lg>

<lg xml:id="HT2.4.89">
	<l>samāpattau sthite deve hevajre vajradhāriṇi |</l>
	<l>tatra pṛcchati nairātmyā sattvārthāya mahābalim || 89 ||</l>
</lg>

<lg xml:id="HT2.4.90">
	<l>evaṃkāre samāsīno vajrasattvo diśed balim |</l>
	<l>sattvānāṃ prāṇarakṣāya vighnād vināyakād api || 90 ||</l>
</lg>

<lg xml:id="HT2.4.91">
	<l><pb n="194"/>inda jama jala jakkha bhuta vahni vāyu rakkha |</l>
	<l>canda sujja māda bappa talapātāle aṭṭhasappa svāhā || 91 ||</l>
</lg>

<lg xml:id="HT2.4.92">
	<l>idaṃ baliṃ bhuñja jighra phulla-dhūpa-māṃsa-viṅgha |</l>
	<l>ambha kajja savva sādha khanti kuṇi pheḍa gāda || 92 ||</l>
</lg>

<p xml:id="HT2.4.93">
	oṃ akāro mukhaṃ sarvadharmāṇām ādyanutpannatvāt | oṃ āḥ hūṁ phaṭ svāhā || 93 ||
</p>

<lg xml:id="HT2.4.94">
	<l><pb n="195"/>anena balinā yadi sarvabhūtān pūjāṃ prakurvanti śubhāya yoginaḥ |</l>
	<l>bhavet tadā teṣu sukham anāvilaṃ devāś ca tuṣyanti jagatsubhūtayaḥ || 94 ||</l>
</lg>

<lg xml:id="HT2.4.95">
	<l>vaśyābhicāraripusainyanāśanam uccāṭanamāraṇākarṣaṇaṃ ca</l>
	<l>śāntisukhaṃ pauṣṭikaṃ bhavec ca dadyāt baliṃ yadīha bhūtagaṇāya śāśvataḥ || 95 ||</l>
	<!-- not recognised like as verse -->
</lg>

<lg xml:id="HT2.4.96">
	<l>vajragarbha āha—</l>
	<l>khecarī kena mudreṇa bhūcarī kasya mudrataḥ |</l>
	<l>karttavyaṃ mudraṇaṃ kathaṃ prāg na jñātaṃ mayā prabho || 96 ||</l>
</lg>

<lg xml:id="HT2.4.97">
	<l><pb n="196"/>bhagavān āha—</l>
	<l>triguhyaṃ cakramadhye tu kāyavākcittabhedataḥ |</l>
	<l>adhordhvamadhyamaṃ sthānaṃ cakramadhye vyavasthitam || 97 ||</l>
</lg>

<lg xml:id="HT2.4.98">
	<l>bhūcarī kāyamudrī syād adhomukhī kāyavajriṇī |</l>
	<l>khecarī rāgamudrī ca ūrdhvamukhī vāgvajriṇī || 98 ||</l>
</lg>

<lg xml:id="HT2.4.99">
	<l>cittavajrī ca nairātmyā cittaṃ nairātmyarūpakam |</l>
	<l>cittaṃ madhyamakaṃ sthānaṃ nairātmyā tena madhyajā || 99 ||</l>
</lg>

<lg xml:id="HT2.4.100">
	<l>kulāni ṣaḍvidhāny āhur vistareṇa prakāśayet |</l>
	<l>trividhaṃ pañcavidhaṃ caiva kathyate śṛṇu yoginī || 100 ||</l>
</lg>

<p xml:id="HT2.4.101">
	akṣobhya-vairocana-ratnasambhava-amitaprabha-amoghasiddhi-vajrasattvaḥ | dveṣa-moha-piśuna-rāga-īrṣyā-saukhyam | śuddhyānayānukramato hi bhāvyāḥ || 101 ||
</p>

	<l><pb n="197"/>vihāya vajrasattvākhyaṃ paścāt pañcavidhaṃ kulam |</l>
