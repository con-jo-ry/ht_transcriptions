<div n="6" type="paṭala">
<lg xml:id="HT1.6.1">
	<l>ataḥ paraṃ pravakṣyāmi caryāṃ pāraṅgatāṃ varām |</l>
	<l>gamyate yena siddhāntaṃ hevajre siddhihetunā || 1 ||</l>
</lg>

<lg xml:id="HT1.6.2">
	<l>bhāvakena vidhartavyaṃ karṇayor divyakuṇḍalam |</l>
	<l>śirasi cakrī vidhartavyā hastayo rucakadvayam || 2 ||</l>
</lg>

<lg xml:id="HT1.6.3">
	<l>kaṭyāṃ mekhalaṃ caiva pādayor nūpuraṃ tathā |</l>
	<l>bāhumūle ca keyūraṃ grīvāyām asthimālikā || 3 ||</l>
</lg>

<lg xml:id="HT1.6.4">
	<l>paridhānaṃ vyāghracarmaṃ ca bhakṣaṇaṃ daśārdhāmṛtam |</l>
	<l>herukayogasya puṃso viharaṇaṃ pañcavarṇasu || 4 ||</l>
</lg>

<lg xml:id="HT1.6.5">
	<l>pañcavarṇasamāyuktam ekavarṇaṃ tu kalpitam |</l>
	<l>anekenaikavarṇena yasmād bhedo na lakṣyate || 5 ||</l>
</lg>

<lg xml:id="HT1.6.6">
	<l>ekavṛkṣe śmaśāne vā bhāvanā kathyate śubhā |</l>
	<l>mātṛgṛhe tathā rātrau 'thavā vijane prāntare || 6 ||</l>
</lg>

<lg xml:id="HT1.6.7">
	<l>kiñcid ūṣme tu saṃprāpte caryāṃ kartuṃ yadīṣyate |</l>
	<l>siddhiṃ gantuṃ yadīcchāsti caryayā tv anayā caret || 7 ||</l>
</lg>

<lg xml:id="HT1.6.8">
	<l>cāruvaktrāṃ viśālākṣīṃ svābhiṣiktāṃ kṛpāvatīm |</l>
	<l>vajrakanyām imāṃ gṛhya caryāṃ kartuṃ tu budhyate || 8 ||</l>
</lg>

<p xml:id="HT1.6.9">
	vajrakulābhāvāt sveṣṭadevatāyāḥ kulena kriyate | athavānyakulodbhavāṃ bodhibījanikṣepeṇa saṃskṛtām gṛhya || 9 ||
</p>

<lg xml:id="HT1.6.10">
	<l>yadi gītaṃ gīyata ānandāt tarhi vajrānvitaṃ param |</l>
	<l>yady ānande samutpanne nṛtyate mokṣahetunā |</l>
	<l>tarhi vajrapadair nāṭyaṃ kurute yogī samāhitaḥ || 10 ||</l>
</lg>

<lg xml:id="HT1.6.11">
	<l>akṣobhyaś cakrirūpeṇāmitābhaḥ kuṇḍalātmakaḥ |</l>
	<l>ratneśaḥ kaṇṭhamālāyāṃ haste vairocanaḥ smṛtaḥ || 11 ||</l>
</lg>

<lg xml:id="HT1.6.12">
	<l>mekhalāyāṃ sthito 'moghaḥ prajñā khaṭvāṅgarūpiṇī |</l>
	<l>ḍamarukopāyarūpeṇa yogī dveṣaviśuddhitaḥ || 12 ||</l>
</lg>

<lg xml:id="HT1.6.13">
	<l>mantraviśuddhyā sthitā gītā nartanā bhāvanā smṛtā |</l>
	<l>tasmād gītaṃ ca nāṭyaṃ ca kurute yogī sadā sadā || 13 ||</l>
</lg>

<lg xml:id="HT1.6.14">
	<l>bhakṣitavyaṃ tu bhaiṣajyaṃ pātavyaṃ vāri nityatām |</l>
	<l>jarāmṛtyur na bādheta rakṣābhūtaḥ sadā bhavet || 14 ||</l>
</lg>

<lg xml:id="HT1.6.15">
	<l>cauryakeśakṛtā mukuṭī tatra hūṁbhava yojayet |</l>
	<l>pañcabuddhakapālāni dhartavyaṃ yogacaryayā || 15 ||</l>
</lg>

<lg xml:id="HT1.6.16">
	<l>pañcāṅgulakapālakhaṇḍaṃ kṛtvā mukuṭyāṃ dhriyate sadā |</l>
	<l>kacaḍorī dviveḍhā ca prajñopāyasvabhāvataḥ |</l>
	<l>bhasma keśapavitraṃ ca yogī vibharti caryayā || 16 ||</l>
</lg>

<lg xml:id="HT1.6.17">
	<l>jāpaṃ ḍamarukāśabdaṃ prajñā khaṭvāṃgabhāvanā |</l>
	<l>jāpyaṃ bhāvyaṃ bhaved etad vajrakāpālacaryayā || 17 ||</l>
</lg>

<lg xml:id="HT1.6.18">
	<l>lobhaṃ mohaṃ bhayaṃ krodhaṃ vrīḍāṃ kāryaṃ ca varjayet |</l>
	<l>nidrām ātmānam utsṛjya caryā kriyate na saṃśayaḥ || 18 ||</l>
</lg>

<lg xml:id="HT1.6.19">
	<l>śarīraṃ dānaṃ dattvā paścāc caryāṃ samārabhet |</l>
	<l>bhāgābhāgavicāreṇa tasmād dānaṃ na dīyate || 19 ||</l>
</lg>

<lg xml:id="HT1.6.20">
	<l>bhakṣyabhojyaṃ tathā pānaṃ yathāprāptaṃ tu bhakṣayet |</l>
	<l>graham atra na kartavyaṃ miṣṭāmiṣṭavikalpataḥ || 20 ||</l>
</lg>

<lg xml:id="HT1.6.21">
	<l>bhakṣyābhakṣyavicāraṃ tu peyāpeyaṃ tathaiva ca |</l>
	<l>gamyāgamyaṃ tathā mantrī vikalpaṃ naiva kārayet || 21 ||</l>
</lg>

<lg xml:id="HT1.6.22">
	<l>siddhilabdho 'pi yaḥ śiṣyaḥ samyagjñānāvabhāsakaḥ |</l>
	<l>abhivandayati guruṃ siddho 'vīcyās tyājyahetunā || 22 ||</l>
</lg>

<lg xml:id="HT1.6.23">
	<l>śikṣādīkṣāvinirmukto lajjākāryaṃ tathaiva ca |</l>
	<l>sarvabhāvasvabhāvena vicared yogī mahākṛpaḥ || 23 ||</l>
</lg>

<lg xml:id="HT1.6.24">
	<l>homayāgatapotīto mantradhyānavivarjitaḥ |</l>
	<l>samayasaṃvaravinirmuktaś caryāṃ kurute suyogavān || 24 ||</l>
</lg>

<lg xml:id="HT1.6.25">
	<l>śakratulyo 'pi yo daityaḥ purato bhavati niścitam |</l>
	<l>bhayaṃ tatra na kurvīta siṃharūpeṇa paryaṭet || 25 ||</l>
</lg>

<lg xml:id="HT1.6.26">
	<l>karuṇā pīyate nityaṃ sarvasattvārthahetunā |</l>
	<l>yogapānarato yogī nānyapānena majjanam || 26 ||</l>
</lg>

<p xml:id="HT1.6.col">
	caryāpaṭalaḥ ṣaṣṭḥaḥ || ||
</p>
