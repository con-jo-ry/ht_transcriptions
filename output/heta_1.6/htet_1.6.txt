	<l><pb n="60"/>ataḥ paraṃ pravakṣyāmi caryāṃ pāraṅgatāṃ varām |</l>

</lg>

<lg xml:id="HT1.6.2">
	<l>bhāvakena vidhartavyaṃ karṇayor divyakuṇḍalam |</l>
	<l>śirasi cakrī dhartavyā hastayo rucakadvayam || 2 ||</l>
</lg>

<lg xml:id="HT1.6.3">
	<l>kaṭyāṃ vā mekhalaṃ caiva pādayor nūpuraṃ tathā |</l>
	<l>bāhumūle ca keyūraṃ grīvāyām asthimālikā || 3 ||</l>
</lg>

<lg xml:id="HT1.6.4">
	<l>paridhānaṃ vyāghracarma bhakṣaṇaṃ daśārdhāmṛtam |</l>
	<l>herukayogasya puṃso vihāraḥ pañcavarṇeṣu || 4 ||</l>
</lg>

<lg xml:id="HT1.6.5">
	<l><pb n="61"/>pañcavarṇasamāyuktam ekavarṇan tu kalpitam |</l>
	<l>anekenaikavarṇena yasmād bhedo na lakṣyate || 5 ||</l>
</lg>

<lg xml:id="HT1.6.6">
	<l>ekavṛkṣe śmaśāne vā bhāvanā kathyate śubhā |</l>
	<l>mātṛgṛhe tathā ramye 'thavā vijane prāntare || 6 ||</l>
</lg>

<lg xml:id="HT1.6.7">
	<l>kiñcid ūṣme tu saṃprāpte caryāṃ kartuṃ yadīṣyate |</l>
	<l>siddhiṃ gantuṃ yadīcchāsti caryayā tv anayā caret || 7 ||</l>
</lg>

<lg xml:id="HT1.6.8">
	<l><pb n="62"/>cāruvaktrāṃ viśālākṣīṃ rūpayauvanamaṇḍitām |</l>
	<l>nīlotpalaśyāmāṅgīṃ ca svābhiṣiktāṃ kṛpāvatīm |</l>
	<l>vajrakanyām imāṃ gṛhya caryāṃ kartuṃ vibudhyate || 8 ||</l>
</lg>

<p xml:id="HT1.6.9">
	vajrakulābhāvāt sveṣṭadevakulena krīyate |<lb/> athavānyakulodbhavā <lb/>bodhibījanikṣepeṇa saṃskṛtām imāṃ gṛhṇīyat || 9 ||
</p>

<lg xml:id="HT1.6.10">

	<l>tarhi vajrapadaiḥ nāṭyaṃ kuryād yogī samāhitaḥ || 10 ||</l>
</lg>

<lg xml:id="HT1.6.11">

</lg>

<lg xml:id="HT1.6.12">

</lg>

<lg xml:id="HT1.6.13">

</lg>

<lg xml:id="HT1.6.14">

</lg>

<lg xml:id="HT1.6.15">

</lg>

<lg xml:id="HT1.6.16">

</lg>

<lg xml:id="HT1.6.17">

</lg>

<lg xml:id="HT1.6.18">

</lg>

<lg xml:id="HT1.6.19">

</lg>

<lg xml:id="HT1.6.20">

</lg>

<lg xml:id="HT1.6.21">

</lg>

<lg xml:id="HT1.6.22">
