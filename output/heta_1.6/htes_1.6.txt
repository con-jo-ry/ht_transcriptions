<div n="6" type="paṭala">
<lg xml:id="HT1.6.1">
	<l>ataḥ paraṃ pravakṣyāmi caryāṃ pāraṅgatāṃ varāṃ ||</l>
	<l>gamyate yena siddhāntaṃ hevajre siddhihetunā || (1)</l>
</lg>

<lg xml:id="HT1.6.2">
	<l>bhāvakena vidhartavyaṃ karṇayor divyakuṇḍalaṃ ||</l>
	<l>śirasi cakrī dhartavyā hastayo rucakadvayaṃ || (2)</l>
</lg>

<lg xml:id="HT1.6.3">
	<l>kaṭyāṃ vā mekhalaṃ caiva pādayor nūpuran tathā ||</l>
	<l>bāhumūle ca keyūraṃ grīvāyām asthimālikā || (3)</l>
</lg>

<lg xml:id="HT1.6.4">
	<l>paridhānaṃ vyāghracarma bhakṣaṇaṃ daśārdhāmṛtaṃ ||</l>
	<l>herukayogasya puṃso vihāraḥ pañcavarṇeṣu || (4)</l>
</lg>

<lg xml:id="HT1.6.5">
	<l>pañcavarṇasamāyuktam ekavarṇan tu kalpitaṃ ||</l>
	<l>anekenaikavarṇena yasmād bhedo na lakṣyate || (5)</l>
</lg>

<lg xml:id="HT1.6.6">
	<l>ekavṛkṣe śmaśāne vā bhāvanā kathyate śubhā ||</l>
	<l>mātṛgṛhe tathā ramye 'thavā vijane prāntare || (6)</l>
</lg>

<lg xml:id="HT1.6.7">
	<l>kiñcid uṣme tu saṃprāpte caryāṃ kartuṃ yadīṣyate ||</l>
	<l>siddhiṃ gantuṃ yadīcchāsti caryayā tv anayā caret || (7)</l>
</lg>

<lg xml:id="HT1.6.8">
	<l>cāruvaktrāṃ viśālākṣīṃ rūpayauvanamaṇḍitāṃ ||</l>
	<l>nīlotpalaśyāmāṅgīṃ ca svābhiṣiktāṃ kṛpāvatīṃ ||</l>
	<l>vajrakanyām imāṃ gṛhya caryāṃ kartuṃ vibudhyate || (8)</l>
</lg>

<lg xml:id="HT1.6.9">
	<l>vajrakulābhāvāt sveṣṭadevakulena krīyate ||</l>
	<l>athavānyakulodbhavā</l>
	<l>bodhibījanikṣepeṇa saṃskṛtām &lt;imāṃ&gt; gṛh&lt;ṇīyat&gt; || (9)</l>
</lg>

<lg xml:id="HT1.6.10">
	<l>yadi [gītaṃ] gīyata ānandāt tarhi vajrānvitaṃ paraṃ ||</l>
	<l>yady ānande samutpanne nṛtyate mokṣahetunā ||</l>
	<l>tarhi vajrapade nāṭyaṃ kuryād yogī samāhitaḥ || (10)</l>
</lg>

<lg xml:id="HT1.6.11">
	<l>akṣobhyaś cakrīrupeṇāmitābhaḥ kuṇḍalātmakaḥ ||</l>
	<l>ratneśaḥ kaṇṭhamālāyāṃ haste vairocanaḥ smṛtaḥ || (11)</l>
</lg>

<lg xml:id="HT1.6.12">
	<l>mekhalāyāṃ sthito 'moghaḥ prajñā khaṭvāṅgarūpiṇī ||</l>
	<l>ḍamarūpāyarūpeṇa yogī dveṣaviśuddhitaḥ || (12)</l>
</lg>

<lg xml:id="HT1.6.13">
	<l><pb n="20"/>mantra[vi]śuddhyā sthitā gītā nartanā bhāvanā smṛtā ||</l>
	<l>tasmād gītañ ca nāṭyañ ca kuryād yogī sadā sadā || (13)</l>
</lg>

<lg xml:id="HT1.6.14">
	<l>bhakṣitavyan tu bhaiṣajyaṃ pātavyaṃ vāri nityatām ||</l>
	<l>jarāmṛtyur na bādheta rakṣābhūtaḥ sadā bhavet || (14)</l>
</lg>

<lg xml:id="HT1.6.15">
	<l>cauryakeśakṛtāṃ mukuṭīṃ tatra hūṃbhavo yojayet ||</l>
	<l>pañcabuddhakapālāni dhartavyaṃ yogacaryayā || (15)</l>
</lg>

<lg xml:id="HT1.6.16">
	<l>pañcāṅgulakapālakhaṇḍaṃ [kṛtvā] mukuṭyāṃ dhriyate sadā ||</l>
	<l>kacaḍorī dvivetā ca prajñopāyasvabhāvataḥ ||</l>
	<l>bhasmakeśapavitrañ ca yogī vibharti caryayā || (16)</l>
</lg>

<lg xml:id="HT1.6.17">
	<l>jāpo ḍamarukaśabdaḥ prajñākhaṭvāṅgo bhāvanā ||</l>
	<l>jāpyaṃ bhāvyaṃ bhaved etad vajrakapālacaryayā || (17)</l>
</lg>

<lg xml:id="HT1.6.18">
	<l>lobhaṃ mohaṃ bhayaṃ krodhaṃ vrīḍākāryañ ca varjayet ||</l>
	<l>nidrām ātmānam utsṛjya caryāṃ kuryān na saṃśayaḥ || (18)</l>
</lg>

<lg xml:id="HT1.6.19">
	<l>śarīraṃ dānaṃ datvā ca paścāc caryāṃ samārabhet ||</l>
	<l>bhāgābhāgavicāreṇa tasmād dānaṃ na dīyate || (19)</l>
</lg>

<lg xml:id="HT1.6.20">
	<l>bhakṣyaṃ bhojyaṃ tathā pānaṃ yathāprāptaṃ tu bhakṣayet ||</l>
	<l>grahaṇaṃ nātra kartavyaṃ iṣṭāniṣṭavikalpataḥ || (20)</l>
</lg>

<lg xml:id="HT1.6.21">
	<l>bhakṣyābhakṣyavicāran tu peyāpeyaṃ tathaiva ca ||</l>
	<l>gamyāgamyan tathā mantrī vikalpan naiva kārayet || (21)</l>
</lg>

<lg xml:id="HT1.6.22">
	<l>siddhilabdho 'pi yaḥ śiṣyaḥ samyagjñānāvabhāsakaḥ ||</l>
	<l>abhivandayati guruṃ siddho 'vīcyātyājyahetunā || (22)</l>
</lg>

<lg xml:id="HT1.6.23">
	<l>śikṣādikṣāvinirmukto lajjakāryaṃ tathaiva ca ||</l>
	<l>sarvabhāvasvabhāvena [vi]cared yogī mahākṛpaḥ || (23)</l>
</lg>

<lg xml:id="HT1.6.24">
	<l>homatyāgatapo'tīto mantradhyānavivarjitaḥ ||</l>
	<l>samayasamvaravinirmuktaś caryāṃ kurute suyogavān || (24)</l>
</lg>

<lg xml:id="HT1.6.25">
	<l>śakratulyo 'pi yo daityaḥ purato bhavati niścitaṃ ||</l>
	<l>bhayaṃ tatra na kurvīta siṃharūpeṇa paryaṭet || (25)</l>
</lg>
