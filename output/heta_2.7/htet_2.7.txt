	<l><pb n="218"/>athāha tatra sā devī bolakakkolayogataḥ |</l>
	<l>oṣṭhaṃ dantena saṃpīḍya kathaṃ bhavati pustakam ||</l>
	<l>vajrapadmasamāyogāt tuṣṭo devaḥ prakāśate || 1 ||</l>

<lg xml:id="HT2.7.2">
	<l>bhagavān āha—</l>
	<l>śṛṇu devi mahābhāge pustakaṃ kathayāmy aham |</l>
	<l>bhūrjapatre likhet samayī dvādaśāṅgulapustakam |</l>
	<l>mahāmadhumasiṃ kṛtvā lekhanyāṃ mānuṣāsthibhiḥ || 2 ||</l>
</lg>

<lg xml:id="HT2.7.3">
	<l>pustakañ ca paṭañ caiva yadi vā dunduraḥ paśyati |</l>
	<l>iha janmani na siddhiḥ syān na vā paralokagocare || 3 ||</l>
</lg>

<lg xml:id="HT2.7.4">
	<l>sampradāyaprayuktasya darśanañ ca kadācana |</l>
	<l>gopitavyaṃ kace kakṣe pustakam adhvagocare || 4 ||</l>
</lg>

<lg xml:id="HT2.7.5">

</lg>

<lg xml:id="HT2.7.6">

</lg>

<lg xml:id="HT2.7.7">

</lg>

<lg xml:id="HT2.7.8">

</lg>

<lg xml:id="HT2.7.9">

</lg>

<lg xml:id="HT2.7.10">
