<lg xml:id="HT2.7.1">
	<l><pb n="88"/>athāha tatra sā devī bolakakkolayogataḥ ||</l>
	<l>oṣṭhaṃ dantena saṃpīḍya kathaṃ bhavati pustakaṃ ||</l>
	<l>vajrapadmasamāyogāt tuṣṭo devaḥ prakāśate || (1)</l>
</lg>

<lg xml:id="HT2.7.2">
	<l>bhagavān āha |</l>
	<l>śṛṇu devi mahābhāge pustakaṃ kathayāmy ahaṃ |</l>
	<l>bhūrjapatre likhet samayī dvādaśāṅgulapustakaṃ ||</l>
	<l>mahāmadhumasiṃ kṛtvā lekhanyāṃ mānuṣāsthibhiḥ || (2)</l>
</lg>

<lg xml:id="HT2.7.3">
	<l>pustakañ ca paṭañ caiva yadi vā dunduraḥ paśyati |</l>
	<l>iha janmani na siddhiḥ syān na vā paralokagocare || (3)</l>
</lg>

<lg xml:id="HT2.7.4">
	<l>saṃpradāyaprayuktasya darśanañ ca kadācana |</l>
	<l>gopitavyam kace kakṣe pustakam adhvagocare || (4)</l>
</lg>

<lg xml:id="HT2.7.5">
	<l>bhage liṅgaṃ pratiṣṭhāpya cumbayitvā muhurmuhuḥ |</l>
	<l>mahāsukhaṃ samāsādhya vajrī bhojanam ādiśet || (5)</l>
</lg>

<lg xml:id="HT2.7.6">
	<l>sṛṇu devi viśālākṣi bhojanaṃ gaṇamaṇḍale |</l>
	<l>yatrabhukte bhavet siddhiḥ sarvakāmārthasādhakī || (6)</l>
</lg>

<lg xml:id="HT2.7.7">
	<l>śmaśāne girikuñje vāmānuṣa[sya] pure tathā |</l>
	<l>athavā vijane prānte idaṃ bhojanam ārabhet || (7)</l>
</lg>

<lg xml:id="HT2.7.8">
	<l>kalpayed āsanaṃ tatra navākhyaṃ śavarūpiṇaṃ |</l>
	<l>athavā vyāghracarmañ ca śmaśānakarpaṭaṃ tathā || (8)</l>
</lg>

<lg xml:id="HT2.7.9">
	<l>madhye hevajrarūpātmā yoginīnāṃ tato nyaset |</l>
	<l>sthānaṃ jñātvā yathāpūrvaṃ diśāsu vidiśāsu ca || (9)</l>
</lg>

<lg xml:id="HT2.7.10">

	<l>bhakṣañ ca bhakṣayet tatra rājaśāliṃ prayatnataḥ |</l>
	<l>bhuktvā bhuktvā punas tatra pūjyante tatra mātaraḥ || (10)</l>
</lg>

<lg xml:id="HT2.7.11">

</lg>
