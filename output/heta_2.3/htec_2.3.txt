<div n="3" type="paṭala">
<lg xml:id="HT2.3.1">
	<l>atha vajrī sarvatantranidānaṃ nāmopāyaṃ yoginīnāṃ kathayām āsa |</l>
	<l>saṃvaraṃ cābhiṣekaṃ ca sandhyābhāṣaṃ tathaiva ca |</l>
	<l>ānandakṣaṇabhedaṃ ca anyac ca bhojanādikam || 1 ||</l>
</lg>

<lg xml:id="HT2.3.2">
	<l>tatra saṃvaram āha—</l>
	<l>saṃvaraṃ sarvabuddhānām evaṃkāre pratiṣṭhitam |</l>
	<l>abhiṣekāj jñāyate samyag ekākāraṃ mahat sukham || 2 ||</l>
</lg>

<lg xml:id="HT2.3.3">
	<l>atha bhagavantaṃ vajrasattvaṃ yoginya evam āhuḥ—</l>
	<l>evaṃkāraṃ kim ucyate ḍākinīnāṃ tu saṃvaram |</l>
	<l>deśayantu yathānyāyaṃ bhagavāṃs chāstā jagadguruḥ || 3 ||</l>
</lg>

<lg xml:id="HT2.3.4">
	<l>bhagavān āha—</l>
	<l>ekārākṛti yad divyaṃ madhye vaṃkārabhūṣitam |</l>
	<l>ālayaḥ sarvasaukhyānāṃ buddharatnakaraṇḍakam || 4 ||</l>
</lg>

<lg xml:id="HT2.3.5">
	<l>ānandās tatra jāyante kṣaṇabhedena bheditāḥ |</l>
	<l>kṣaṇajñānāt sukhajñānam evaṃkāre pratiṣṭhitam || 5 ||</l>
</lg>

<lg xml:id="HT2.3.6">
	<l>vicitraṃ ca vipākaṃ ca vimardo vilakṣaṇas tathā |</l>
	<l>catuḥkṣaṇaṃ samāgamya evaṃ jānanti yoginaḥ || 6 ||</l>
</lg>

<lg xml:id="HT2.3.7">
	<l>vicitraṃ vividhaṃ khyātaṃ āliṅganacumbanādikam |</l>
	<l>vipākaṃ tadviparyāsaṃ sukhajñānasya bhuñjanam || 7 ||</l>
</lg>

<lg xml:id="HT2.3.8">
	<l>vimardam ālocanaṃ proktaṃ sukhaṃ bhuktaṃ mayeti ca |</l>
	<l>vilakṣaṇaṃ tribhyo 'nyad rāgārāgavivarjitam || 8 ||</l>
</lg>

<lg xml:id="HT2.3.9">
	<l>vicitre prathamānadaḥ paramānando vipākake |</l>
	<l>viramānando vimarde ca sahajānando vilakṣaṇe || 9 ||</l>
</lg>

<lg xml:id="HT2.3.10">
	<l>ācārya guhya prajñā ca caturthaṃ tat punas tathā |</l>
	<l>ānandādyāḥ kramaśo jñeyāś catuḥsecanasaṃkhyayā || 10 ||</l>
</lg>

<lg xml:id="HT2.3.11">
	<l>hasitaśuddhyā tv ācārya īkṣaṇe guhyakas tathā |</l>
	<l>prajñā hi pāṇyāvāptau ca tat punar dvandvatantrake || 11 ||</l>
</lg>

<lg xml:id="HT2.3.12">
	<l>sekaṃ caturvidhaṃ khyātaṃ sattvānāṃ siddhihetave |</l>
	<l>sicyate snāpyate aneneti sekas tenābhidhīyate || 12 ||</l>
</lg>

<lg xml:id="HT2.3.13">
	<l>pāṇibhyāṃ tu samāliṅgya prajñāṃ vai ṣoḍaśābdikām |</l>
	<l>ghaṇṭāvajrasamāyogād ācāryasecanaṃ matam || 13 ||</l>
</lg>

<lg xml:id="HT2.3.14">
	<l>cāruvaktrāṃ viśālākṣīṃ rūpayauvanamaṇḍitām |</l>
	<l>jyeṣṭhānāmikābhyāṃ tu śiṣyavaktre nipātayet || 14 ||</l>
</lg>

<lg xml:id="HT2.3.15">
	<l>kāritavyaṃ ca tatraiva samarasaṃ śiṣyagocaram |</l>
	<l>prajñāṃ pūjayec chāstā arcayitvā samarpayet || 15 ||</l>
</lg>

<lg xml:id="HT2.3.16">
	<l>śāstā brūyān mahāsattva gṛhṇa mudrāṃ sukhāvahām |</l>
	<l>jñātvā śiṣyaṃ mahadbhūtaṃ nirīrṣyaṃ krodhavarjitam || 16 ||</l>
</lg>

<lg xml:id="HT2.3.17">
	<l>śāstā tam ājñāpayati kunduruṃ kuru vajradhṛk |</l>
	<l>śiṣyakṛtyaṃ pravakṣyāmi sekam anunāthayed yathā || 17 ||</l>
</lg>

<lg xml:id="HT2.3.18">
	<l>mudrāyuktaṃ guruṃ dṛṣṭvā stutipūjāṃ kared yathā |</l>
	<l>he bhagavan mahāśānta vajrayogaikatatparaḥ || 18 ||</l>
</lg>

<lg xml:id="HT2.3.19">
	<l>mudrāprasādhakābhedyavajrayogasamudbhavaḥ |</l>
	<l>yathā yūyaṃ mahātmāno mamāpi kuru tad vibho || 19 ||</l>
</lg>

<lg xml:id="HT2.3.20">
	<l>saṃsārapaṅkasaṃghāte magno 'haṃ trāhy aśaraṇam |</l>
	<l>miṣṭānnapānakhādyaṃ ca madanaṃ balaṃ mahattaram || 20 ||</l>
</lg>

<lg xml:id="HT2.3.21">
	<l>dhūpanaivedyamālyaṃ ca ghaṇṭādhvajavilepanaiḥ |</l>
	<l>ābhiḥ pūjābhiḥ śiṣyaḥ pūjayed vajradhāriṇam || 21 ||</l>
</lg>

<lg xml:id="HT2.3.22">
	<l>paramānande saṃprāpte nānātvavarjite kṣaṇe |</l>
	<l>śāstā brūyān mahāsattva dhāraṇīyaṃ mahat sukham || 22 ||</l>
</lg>

<lg xml:id="HT2.3.23">
	<l>yāvad ābodhiparyantaṃ sattvārthaṃ kuru vajradhṛk |</l>
	<l>ity evaṃ vadate vajrī śiṣyaṃ vīkṣya kriyāparam || 23 ||</l>
</lg>

<lg xml:id="HT2.3.24">
	<l>etad eva mahājñānaṃ sarvadehe vyavasthitam |</l>
	<l>advayaṃ dvayarūpaṃ ca bhāvābhāvātmakaṃ prabhum || 24 ||</l>
</lg>

<lg xml:id="HT2.3.25">
	<l>sthiracalaṃ vyāpya saṃtiṣṭhen māyārūpīva bhāti ca |</l>
	<l>maṇḍalacakrādyupāyena sātatyaṃ yāti niścayam || 25 ||</l>
</lg>

<p xml:id="HT2.3.26">
	<l>atha sarvayoginīnāṃ bhaginīnāṃ mṛṣitvā vajragarbho bhagavantam evam āha—</l>
	<l>maṇḍalacakraṃ kim ucyeta sarvabuddhātmakaṃ param |</l>
	<l>deśayantu yathānyāyaṃ bhagavanto bhrāntir me 'bhūt || 26 ||</l>
</p>

<lg xml:id="HT2.3.27">
	<l>bhagavān āha—</l>
	<l>maṇḍalaṃ sāram ity uktaṃ bodhicittaṃ mahat sukham |</l>
	<l>ādānaṃ taṃ karotīti maṇḍalaṃ mīlanaṃ matam || 27 ||</l>
</lg>

<lg xml:id="HT2.3.28">
	<l>cakraṃ nivahaṃ khadhātvākhyaṃ viṣayādīnāṃ viśodhanam |</l>
	<l>bolakakkolayogena tasya saukhyaṃ pratīyate || 28 ||</l>
</lg>

<lg xml:id="HT2.3.29">
	<l>vajragarbha uvāca—</l>
	<l>kena samayena sthātavyaṃ kena saṃvareṇeti |</l>
	<l>bhagavān āha—</l>
	<l>prāṇinaś ca tvayā ghātyā vaktavyaṃ ca mṛṣāvacaḥ |</l>
	<l>adattaṃ ca tvayā grāhyaṃ sevanaṃ parayoṣitaḥ || 29 ||</l>
</lg>

<lg xml:id="HT2.3.30">
	<l>ekacittaṃ prāṇivadhaṃ prāṇaṃ cittaṃ yato matam |</l>
	<l>lokān uttārayiṣyāmīti mṛṣāvādaṃ ca śabditam |</l>
	<l>yoṣicchukram adattaṃ ca paradārā svābhasundarī || 30 ||</l>
</lg>

<lg xml:id="HT2.3.31">
	<l>atha sarvayoginyo bhagavantam evam āhuḥ—</l>
	<l>ke punas te viṣayāḥ | kānīndriyāṇi | kim āyatanam | katamāḥ skandhāḥ | ke punas te dhātavaḥ | eṣāṃ kiṃ svabhāvam || 31 ||</l>
</lg>

<lg xml:id="HT2.3.32">
	<l>bhagavān āha ṣaḍ viṣayāḥ |</l>
	<l>rūpa śabdas tathā gandho rasaḥ sparśas tathaiva ca |</l>
	<l>dharmadhātusvabhāvaś ca ṣaḍ ete viṣayā matāḥ || 32 ||</l>
</lg>

<lg xml:id="HT2.3.33">
	<l>indriyāṇi ṣaṭ |</l>
	<l>cakṣuḥ śrotraṃ ca ghrāṇaṃ ca jihvā kāyo manas tathā |</l>
	<l>mohavajrādibhir yuktāni ṣaḍ etānīndryāṇi ca || 33 ||</l>
</lg>

<lg xml:id="HT2.3.34">
	<l>viṣayaviṣayillakābhyāṃ tu dvādaśāyatanaṃ bhavet |</l>
	<l>pañcaskandhāś ca rūpādyāḥ saṃskārāntā mahākṛpa || 34 ||</l>
</lg>

<lg xml:id="HT2.3.35">
	<l>indriyaṃ viṣayaṃ caiva indriyavijñānam eva ca |</l>
	<l>dhātavo 'ṣṭādaśākhyātā yoginīnāṃ tu bodhaye || 35 ||</l>
</lg>

<lg xml:id="HT2.3.36">
	<l>svabhāvaṃ caivādyanutpannaṃ na satyaṃ na mṛṣā tathā |</l>
	<l>udakacandropamaṃ sarvaṃ yoginyo jānatecchayā || 36 ||</l>
</lg>

<p xml:id="HT2.3.37">
	tad yathā kāṇḍaṃ ca mathanīyaṃ ca puruṣahastavyāyāmaṃ ca pratītyākasmād agnir upajāyate | asāv agnir na kāṇḍe tiṣṭhati na mathanīye na puruṣahastayoḥ | sarvākārataḥ parigaveṣyamāṇa ekasminn api nāsti | sa cāgnir na satyaṃ na mṛṣā | evaṃ sarvadharmān yoginyo manasikuruta || 37 ||
</p>

<p xml:id="HT2.3.38">
	atha nairātmyayoginīpramukhāḥ sarvavajraḍākinyaḥ pañcāmṛtaṃ samayadravyaṃ ca gṛhītvā bhagavantaṃ vajrasattvaṃ pūjayanti | kunduruyogenānurāgayanti pibāvayanti ca vajrāmṛtarasam || 38 ||
</p>

<lg xml:id="HT2.3.39">
	<l>tataḥ paścād bhagavāṃs tuṣṭe saty adhiṣṭhānaṃ darśayati—</l>
	<l>bho bho vajraḍākinyaḥ |</l>
	<l>mayā guptīkṛtaṃ tattvaṃ sarvabuddhair namaskṛtam |</l>
	<l>pūjāvajraprabhāvena kathayāmi śṛṇutecchayā || 39 ||</l>
</lg>

<p xml:id="HT2.3.40">
	utsāhaprāptāḥ harṣaprāptāḥ sarvā devatyo dakṣiṇaṃ jānumaṇḍalaṃ pṛthivyām pratiṣṭhāpya yena bhagavāṃs tenāñjaliṃ praṇamya bhagavato bhāṣitaṃ śṛṇvanti || 40 ||
</p>

<lg xml:id="HT2.3.41">
	<l>bhagavān āha—</l>
	<l>khānaṃ pānaṃ yathāprāptaṃ gamyāgamyaṃ na varjayet |</l>
	<l>snānaṃ śaucaṃ na kurvīta grāmyadharmaṃ na varjayet || 41 ||</l>
</lg>

<lg xml:id="HT2.3.42">
	<l>mantraṃ naiva japed dhīmān dhyānaṃ naivāvalambayet |</l>
	<l>nidrātyāgaṃ na kurvīta nendriyāṇāṃ nivāraṇam || 42 ||</l>
</lg>

<lg xml:id="HT2.3.43">
	<l>bhakṣaṇīyaṃ balaṃ sarvaṃ pañcavarṇaṃ samācaret |</l>
	<l>ramate sarvayoṣitāṃ tu nirviśaṅkena cetasā || 43 ||</l>
</lg>

<lg xml:id="HT2.3.44">
	<l>mitre snehaṃ na kartavyaṃ dviṣṭe dveṣaṃ tathā na ca |</l>
	<l>na vandayed imān devān kāṣṭhapāṣāṇamṛnmayān |</l>
	<l>satataṃ devatāmūrtyā sthātavyaṃ yoginā yataḥ || 44 ||</l>
</lg>

<lg xml:id="HT2.3.45">
	<l>ḍombacaṇḍālacarmārahaḍḍikādyān duḥspṛśān |</l>
	<l>brahmakṣatriyaviṭśūdrādyān ātmadeham iva spṛśet || 45 ||</l>
</lg>

<lg xml:id="HT2.3.46">
	<l>pañcāmṛtaṃ guḍaṃ madyaṃ viṣaṃ nimbaṃ prasūtajam |</l>
	<l>amlamadhurakaṣāyādi tiktaṃ lavaṇakaṭukaṃ tathā || 46 ||</l>
</lg>

<lg xml:id="HT2.3.47">
	<l>pūtisurabhi jalāsṛg vā bodhicittena bhakṣayet |</l>
	<l>nābhakṣaṃ vidyate kiñcid advayajñānacetasā || 47 ||</l>
</lg>

<lg xml:id="HT2.3.48">
	<l>svayambhūkusumaṃ prāpya padmabhāṇḍe niveśayet |</l>
	<l>śleṣmasiṅghāṇakābhyāṃ ca miśrīkṛtya pibed vratī || 48 ||</l>
</lg>

<lg xml:id="HT2.3.49">
	<l>kaupīnaṃ viśvavarṇaṃ ca mṛcchārair bhūṣaṇaṃ tathā |</l>
	<l>puṣpaṃ pretālaye prāpya bandhayen mūrdhajaṃ varam || 49 ||</l>
</lg>

<lg xml:id="HT2.3.50">
	<l>atha vajragarbha āha—</l>
	<l>indriyāṇy aviśuddhāni ṣaṭsaṃkhyākṛtāni vai |</l>
	<l>śuddhiḥ sarvaviṣayasya bhagavatā kathitā purā || 50 ||</l>
</lg>

<lg xml:id="HT2.3.51">
	<l>bhagavān āha—</l>
	<l>cakṣuṣor mohavajrī tu śrotrayor dveṣavajrikā |</l>
	<l>ghrāṇe mātsaryakī khyātā vaktre ca rāgavajrikā || 51 ||</l>
</lg>

<lg xml:id="HT2.3.52">
	<l>sparśe īrṣyāvajrī ca mano nairātmyayoginī |</l>
	<l>kavacam ebhir mahāsattva indriyāṇāṃ viśuddhaye || 52 ||</l>
</lg>

<lg xml:id="HT2.3.53">
	<l>vajragarbha uvāca—</l>
	<l>sandhyābhāṣaṃ kim ucyeta bhagavanto brūhi niścitam |</l>
	<l>yoginīnāṃ mahāsamayaṃ śrāvakādyair na cchidritam || 53 ||</l>
</lg>

<lg xml:id="HT2.3.54">
	<l>hasitekṣaṇābhyāṃ ca āliṅganadvandvakais tathā |</l>
	<l>tantreṇāpi caturṇāṃ ca sandhyābhāṣaṃ na śabditam || 54 ||</l>
</lg>

<lg xml:id="HT2.3.55">
	<l>bhagavān āha—</l>
	<l>vajragarbha ahaṃ vakṣye śṛṇu tvam ekacetasā |</l>
	<l>sandhyābhāṣaṃ mahābhāṣaṃ samayasaṃketavistaram || 55 ||</l>
</lg>

<lg xml:id="HT2.3.56">
	<l>madanaṃ madyaṃ balaṃ māṃsaṃ malayajaṃ mīlanaṃ tathā |</l>
	<l>gatiḥ kheṭaḥ śavaḥ śrāpaḥ asthyābharaṇaṃ niraṃśukam || 56 ||</l>
</lg>

<lg xml:id="HT2.3.57">
	<l>āgatiḥ preṅkhaṇaṃ prājña kṛpīṭaṃ ḍamarukaṃ matam |</l>
	<l>abhavyaṃ dunduraṃ khyātaṃ bhavyaṃ kāliñjaraṃ matam || 57 ||</l>
</lg>

<lg xml:id="HT2.3.58">
	<l>asparśaṃ ḍiṇḍimaṃ proktaṃ kapālaṃ padmabhājanam |</l>
	<l>bhaktaṃ tṛptikaraṃ jñeyaṃ vyañjanaṃ mālatīndhanam || 58 ||</l>
</lg>

<lg xml:id="HT2.3.59">
	<l>gūthaṃ catuḥsamaṃ proktaṃ mūtraṃ kastūrikā smṛtā |</l>
	<l>svayaṃbhū sihlakaṃ jñeyaṃ śukraṃ karpūrakaṃ matam || 59 ||</l>
</lg>

<lg xml:id="HT2.3.60">
	<l>mahāmāṃsaṃ śālijaṃ proktaṃ dvendriyayogaṃ tu kundurum |</l>
	<l>vajraṃ bolakaṃ khyātaṃ padmaṃ kakkolakaṃ matam || 60 ||</l>
</lg>

<lg xml:id="HT2.3.61">
	<l>kulaṃ pañcavidhaṃ khyātaṃ varṇabhedena bheditam |</l>
	<l>sandhyābhāṣata eva syur buddhāḥ pañcakaulikāḥ || 61 ||</l>
</lg>

<lg xml:id="HT2.3.62">
	<l>ḍombī vajrakulī khyātā naṭī padmakulī tathā |</l>
	<l>śvapacī ratnakulī caiva dvijā tāthāgatī matā || 62 ||</l>
</lg>

<lg xml:id="HT2.3.63">
	<l>rajakī karmakulī caiva etā mudrāḥ susiddhidāḥ |</l>
	<l>āsāṃ śukraṃ bhaved vajraṃ pūjayitvā pibed vratī || 63 ||</l>
</lg>

<lg xml:id="HT2.3.64">
	<l>vajragarbha mahāsattva yan mayā kathitaṃ tvayi |</l>
	<l>tat sarvaṃ sādaraṃ grāhyaṃ sandhyābhāṣaṃ mahādbhutam || 64 ||</l>
</lg>

<lg xml:id="HT2.3.65">
	<l>yo 'bhiṣikto 'tra hevajre na vadet sandhyābhāṣayā |</l>
	<l>samayavidrohanaṃ tasya jāyate nātra saṃśayaḥ || 65 ||</l>
</lg>

<lg xml:id="HT2.3.66">
	<l>ity upadravacauraiś ca grahajvaraviṣeṇa ca |</l>
	<l>mriyate yadi buddho 'pi sandhyābhāṣaṃ na bhāṣayet || 66 ||</l>
</lg>

<lg xml:id="HT2.3.67">
	<l>svasamayavidāṃ prāpya yadi na bhāṣed idaṃ vacaḥ |</l>
	<l>tadā kṣobhaṃ prakurvanti yoginyaś catuṣpīṭhajāḥ || 67 ||</l>
</lg>

<p xml:id="HT2.3.col">
	hevajre ḍākinījālasaṃvare sarvatantranidhānasandhyābhāṣo nāma tṛtīyaḥ paṭalaḥ || ||
</p>
