<div n="3" type="paṭala">
<lg xml:id="HT2.3.1">
	<l>atha vajrī sarvatantranidānaṃ nāmopayaṃ yoginīnāṃ kathayām āsa |</l>
	<l>saṃvaraṃ cābhiṣekañ ca saṃdhyābhāṣaṃ tathaiva ca ||</l>
	<l>ānandakṣaṇabhedañ ca tathānyaṃ bhojanādikaṃ || (1)</l>
</lg>

<lg xml:id="HT2.3.2">
	<l>tatra saṃvaram āha |</l>
	<l>saṃvaraṃ sarvabuddhānām evaṃkāre pratiṣṭhitaṃ ||</l>
	<l>abhiṣekāj jñāyate samyag evaṃkāraṃ mahat sukhaṃ || (2)</l>
</lg>

<lg xml:id="HT2.3.3">
	<l>atha bhagavantaṃ vajrasattvaṃ yoginya evam āhuḥ |</l>
	<l>evaṃkāraṃ kim ucyate ḍākinīnān tu saṃvaraṃ ||</l>
	<l>deśayantu yathānyāyaṃ bha[aga]vān śāstā jagadguruḥ || (3)</l>
</lg>

<lg xml:id="HT2.3.4">
	<l>bhagavān āha |</l>
	<l>ekārākṛti yad divyaṃ madhye vaṃkārabhūṣitaṃ ||</l>
	<l>ālayaḥ sarvasaukhyānāṃ buddharatnakaraṇḍakaṃ || (4)</l>
</lg>

<lg xml:id="HT2.3.5">
	<l>ānandās tatra jāyante kṣaṇabhedena bheditāḥ ||</l>
	<l>kṣaṇajñānāt sukhajñānam evaṃkāre pratiṣṭhitaṃ || (5)</l>
</lg>

<lg xml:id="HT2.3.6">
	<l>vicitraṃ ca vipākaṃ ca vimardo vilakṣaṇaṃ tathā ||</l>
	<l>catuḥkṣaṇasamāgamyam evaṃ jānanti yoginaḥ || (6)</l>
</lg>

<lg xml:id="HT2.3.7">
	<l>vicitraṃ vividhaṃ khyātaṃ āliṅgacuṃbanādikaṃ ||</l>
	<l><pb n="54"/>vipākaṃ tadviparyāsaṃ sukhajñānasya bhuñjanaṃ || (7)</l>
</lg>

<lg xml:id="HT2.3.8">
	<l>vimardam ālocanaṃ proktaṃ sukhaṃ bhuktaṃ mayeti ca ||</l>
	<l>vilakṣaṇaṃ tribhyo 'nyatra rāgārāgavivarjitaṃ || (8)</l>
</lg>

<lg xml:id="HT2.3.9">
	<l>vicitre prathamānadaḥ paramānando vipākake ||</l>
	<l>viramānando vimarde ca sahajānando vilakṣaṇe || (9)</l>
</lg>

<lg xml:id="HT2.3.10">
	<l>ācārya guhya prajñā ca caturthan tat punas tathā ||</l>
	<l>ānandāḥ kramaśo jñeyāś catuḥsecanasaṃkhyayā || (10)</l>
</lg>

<lg xml:id="HT2.3.11">
	<l>hasitaśuddhyā tv ācārya īkṣaṇe guhyakas tathā ||</l>
	<l>prajñā &lt;hi&gt; pāṇyāvāptau ca tat punar dvandvatantrake || (11)</l>
</lg>

<lg xml:id="HT2.3.12">
	<l>sekaṃ caturvidhaṃ khyātaṃ sattvānāṃ siddhihetave ||</l>
	<l>sicyate snāpyate 'neneti sekas tenābhidhīyate || (12)</l>
</lg>

<lg xml:id="HT2.3.13">
	<l>pāṇibhyāṃ tu samāliṅgya prajñāṃ vai śoḍaśābdikāṃ ||</l>
	<l>ghaṇṭhāvajrasamāyogād ācāryasecanaṃ mataṃ || (13)</l>
</lg>

<lg xml:id="HT2.3.14">
	<l>cāruvaktrā viśālākṣī rūpayauvanamaṇḍitā ||</l>
	<l>jyeṣṭhānāmikābhyañ ca śiṣyavaktre nipātayet || (14)</l>
</lg>

<lg xml:id="HT2.3.15">
	<l>kāritavyan tu tatraiva samarasaṃ śiṣyagocaraṃ ||</l>
	<l>prajñāṃ pūjayec chāstā arcayitvā samarpayet || (15)</l>
</lg>

<lg xml:id="HT2.3.16">
	<l>śāstā brūyāt mahāsattva gṛhna mudrāṃ sukhāvahāṃ ||</l>
	<l>jñātvā śiṣyaṃ mahadbhūtaṃ nirīrṣyaṃ krodhavarjitaṃ || (16)</l>
</lg>

<lg xml:id="HT2.3.17">
	<l>śāstā tam ājñāpayati kunduraṃ kuru vajradhṛk ||</l>
	<l>śiṣyakṛtyaṃ pravakṣyāmy abhiṣekam anunāyayed || (17)</l>
</lg>

<lg xml:id="HT2.3.18">
	<l>mudrāyuktaṃ guruṃ dṛṣṭvā stutipūjāṃ karet yathā ||</l>
	<l>he bhagavan mahāśānta vajrayogaikatatpara || (18)</l>
</lg>

<lg xml:id="HT2.3.19">
	<l>mudrāprasādhakābhedyavajrayogasamudbhava ||</l>
	<l>yathā yūyaṃ mahātmano mamāpi kuru tad vibho || (19)</l>
</lg>

<lg xml:id="HT2.3.20">
	<l>saṃsārapaṅkasaṃghāṭe magno 'haṃ trāhy aśaraṇaṃ ||</l>
	<l>miṣṭānnapānakhādyañ ca madanaṃ balaṃ mahattaraṃ || (20)</l>
</lg>

<lg xml:id="HT2.3.21">
	<l>dhūpan naivedyaṃ mālyañ ca ghaṇṭhādhvajavilepanaiḥ ||</l>
	<l>ābhiḥ pūjādibhiḥ śiṣyaḥ pūjayed vajradhāriṇaṃ || (21)</l>
</lg>

<lg xml:id="HT2.3.22">
	<l>paramānande tu saṃprāpte nānātvavarjite kṣaṇe ||</l>
	<l>śāstā brūyāt mahāsattva dhāraṇīyaṃ mahat sukhaṃ || (22)</l>
</lg>

<lg xml:id="HT2.3.23">
	<l>yāvad ā bodhiparyantaṃ sattvārthaṃ kuru vajradhṛk ||</l>
	<l>ity evaṃ vadate vajrī śiṣyaṃ vīkṣyā kṛpācayaṃ || (23)</l>
</lg>

<lg xml:id="HT2.3.24">
	<l>etad eva mahājñānaṃ sarvadehe vyavasthitaṃ ||</l>
	<l>advayaṃ dvayarūpañ ca bhāvābhāvātmakaṃ prabhuṃ || (24)</l>
</lg>

<lg xml:id="HT2.3.25">
	<l>sthiracalaṃ vyāpya saṃtiṣṭhet māyārūpi ca bhāti ca ||</l>
	<l><pb n="56"/>maṇḍalacakrādyupāyena sātatyaṃ yāti niścayaṃ || (25)</l>
</lg>

<p xml:id="HT2.3.26">
	<l>atha sarvayoginīnāṃ bhaginīnāṃ mṛṣitvā vajragarbo bhagavantaṃ evaṃ āha ||</l>
	<l>maṇḍalacakraṃ kim ucyeta sarvabuddhātmakaṃ puraṃ ||</l>
	<l>deśayantu yathānyāyaṃ bhagavanto bhrāntir me 'bhūt || (26)</l>
</p>

<lg xml:id="HT2.3.27">
	<l>bhagavān āha |</l>
	<l>maṇḍalaṃ sāram ity uktaṃ bodhicittaṃ mahat sukhaṃ ||</l>
	<l>ādānan tat karotīti maṇḍalaṃ malanaṃ mataṃ || (27)</l>
</lg>

<lg xml:id="HT2.3.28">
	<l>cakraṃ nivahaṃ khadhātvākhyaṃ viṣayādīnāṃ viśodhanaṃ ||</l>
	<l>bolakakkolayogena tasya saukhyaṃ pratīyate || (28)</l>
</lg>

<lg xml:id="HT2.3.29">
	<l>vajragarbha āha |</l>
	<l>kena samayena sthātavyaṃ kena saṃvareṇeti |</l>
	<l>bhagavān āha |</l>
	<l>prāṇinaś ca tvayā ghātyā vaktavyaṃ ca mṛṣāvacaḥ ||</l>
	<l>adattañ ca tvayā grāhyaṃ sevanaṃ parayoṣitaḥ || (29)</l>
</lg>

<lg xml:id="HT2.3.30">
	<l>ekacittaṃ prāṇivadhaṃ [proktaṃ] prāṇa cittaṃ yato mataṃ ||</l>
	<l>lokān uttārayiṣyāmi mṛṣāvādañ ca śabditaṃ ||</l>
	<l>yoṣicchrukram adattañ ca paradārāḥ svābhasundarī || (30)</l>
</lg>

<lg xml:id="HT2.3.31">
	<l>atha sarvayoginyo bhagavantaṃ evam āhuḥ |</l>
	<l>ke punas te viṣayāḥ kānīndriyāṇi kim āyatanaṃ kathame skandhāḥ || ke punas te dhātavaḥ || eṣāṃ kiṃ svabhāvaṃ || (31)</l>
</lg>

<lg xml:id="HT2.3.32">
	<l>bhagavān āha | ṣaḍ viṣayāḥ |</l>
	<l>rūpa śabdas tathā gandho rasa sparśas tathaiva ca ||</l>
	<l>dharmadhātusvabhāvaś ca ṣaḍ ete viṣayā matāḥ || (32)</l>
</lg>

<lg xml:id="HT2.3.33">
	<l>indriyāṇi ca ṣaṭ |</l>
	<l>cakṣuḥ śrotraṃ ca ghrāṇañ ca jihvā kāyo manas tathā ||</l>
	<l>mohavajrādibhir yuktāḥ ṣaḍ etānīndryāṇi ca || (33)</l>
</lg>

<lg xml:id="HT2.3.34">
	<l>viṣayavaiṣayikābhyāṃ ca dvādaśāyatanaṃ bhavet ||</l>
	<l>pañcaskandhāś ca rūpādyā vijñānāntā mahākṛpa || (34)</l>
</lg>

<lg xml:id="HT2.3.35">
	<l>indriyaṃ viṣayaṃ caiva indriyavijñānam eva ca ||</l>
	<l>dhātavo 'ṣṭādaśākhyātā yoginīnāṃ tu bodhaye || (35)</l>
</lg>

<lg xml:id="HT2.3.36">
	<l>svabhāvam ādyanutpannaṃ na satyaṃ na mṛṣā tathā ||</l>
	<l>udakacandropamaṃ sarvaṃ yoginyo jānatecchayā || (36)</l>
</lg>

<p xml:id="HT2.3.37">
	tad yathā kāṇḍañ ca mathanīyaṃ ca puruṣahastavyāyāmañ ca pratītyākasmād agnir upajāyate || asāv agnir na kāṇḍe tiṣṭhati || na mathanīye na puruṣahastavyāyāme || sarvākārataḥ parigaveṣyamānaḥ || ekasminn api <pb n="58"/>nāsti sa cāgnir na satyaṃ na mṛṣā || evaṃ sarvadharmān yoginyo manasikuruta || (37)
</p>

<p xml:id="HT2.3.38">
	atha nairātmyayoginīpramukhāḥ sarvavajraḍākinyaḥ pañcāmṛtaṃ gṛhītvā samayadravyañ ca bhagavantaṃ vajrasattvaṃ pūjayanti | kunduruyogenānurāgayanti pibayanti ca vajrāmṛtarasaṃ | (38)
</p>

<lg xml:id="HT2.3.39">
	<l>tataḥ paścād bhagavāṃs tuṣṭe sati adhiṣṭhānaṃ darśayanti |</l>
	<l>bho bho vajraḍākinyo</l>
	<l>mayā guptīkṛtaṃ tattvaṃ sarvair buddhair namaskṛtaṃ ||</l>
	<l>vajrapūjāprabhāvena kathayāmi śṛṇutecchayā || (39)</l>
</lg>

<p xml:id="HT2.3.40">
	atha utsāhaprāptāḥ sarvā devyo dakṣiṇaṃ jānumaṇḍalaṃ pṛthivyām pratiṣṭhāpya yena bhagavāṃs tenāñjaliṃ praṇamya bhagavato bhāṣitaṃ śṛṇvanti | (40)
</p>

<lg xml:id="HT2.3.41">
	<l>bhagavān āha |</l>
	<l>khānaṃ pāṇaṃ yathāprāptaṃ gamyāgamyaṃ na varjayet ||</l>
	<l>snānaṃ śaucaṃ na kurvīta grāmyadharmaṃ na varjayet || (41)</l>
</lg>

<lg xml:id="HT2.3.42">
	<l>mantran naiva japed dhīmān dhyānaṃ naivāvalambayet ||</l>
	<l>nidrātyāgaṃ na kurvīta nendriyāṇāṃ nivāraṇaṃ || (42)</l>
</lg>

<lg xml:id="HT2.3.43">
	<l>bhakṣaṇīyaṃ balaṃ sarvaṃ pañcavarṇaṃ samācaret ||</l>
	<l>ramate sarvayoṣitā nirviśaṅkena cetasā || (43)</l>
</lg>

<lg xml:id="HT2.3.44">
	<l>mitrasnehaṃ na kurvīta dviṣṭe dveṣaṃ tathā na ca ||</l>
	<l>na vandayed imān devān kāṣṭhapāṣāṇamṛnmayān ||</l>
	<l>satataṃ devatāmūrtyā sthātavyaṃ yoginā yataḥ || (44)</l>
</lg>

<lg xml:id="HT2.3.45">
	<l>ḍombacaṇḍālacarmārahaḍḍikādyān tu duḥspṛśān ||</l>
	<l>brahmakṣatrivaisyaśūdrādyān ātmadeham iva spṛśet || (45)</l>
</lg>

<lg xml:id="HT2.3.46">
	<l>pañcāmṛtaṃ guḍaṃ madyaṃ viṣaṃ nimbaṃ prasūtajaṃ ||</l>
	<l>amlamadhurakaṣāyādi tiktalavaṇakaṭukas tathā || (46)</l>
</lg>

<lg xml:id="HT2.3.47">
	<l>pūtisurabhi jalāsṛg bodhicittena bhakṣayet ||</l>
	<l>nābhaktaṃ vidyate kiñcid advayajñānacetasā || (47)</l>
</lg>

<lg xml:id="HT2.3.48">
	<l>svayaṃbhūkusumaṃ prāpya padmabhāṇḍe niveśayet ||</l>
	<l>śleṣmasiṅghāṇakānān tu miśrīkṛtya pibed vratī || (48)</l>
</lg>

<lg xml:id="HT2.3.49">
	<l>kaupinaṃ viśvavarṇañ ca mṛcchārair bhūṣaṇan tathā ||</l>
	<l>puṣpaṃ pretālaye prāpya bandhayen mūrddhajaṃ varaṃ || (49)</l>
</lg>

<lg xml:id="HT2.3.50">
	<l>atha vajragarbha āha |</l>
	<l>indriyāṇy aviśuddhāni ṣaṭsaṃkhyayākṛtāni vai ||</l>
	<l>śuddhiḥ sarvaviṣayasya bhagavatā kathitā purā || (50)</l>
</lg>

<lg xml:id="HT2.3.51">
	<l><pb n="60"/>bhagavān āha |</l>
	<l>cakṣuṣo mohavajrā tu śrotrayor dveṣavajrikā ||</l>
	<l>ghrāṇer mātsaryakī khyātā vaktre ca rāgavajrikā || (51)</l>
</lg>

<lg xml:id="HT2.3.52">
	<l>sparśe īrṣyāvajrā ca mano nairātmyayoginī ||</l>
	<l>kavacam ebhir mahāsatva indriyāṇāṃ viśuddhaye || (52)</l>
</lg>

<lg xml:id="HT2.3.53">
	<l>vajragarbha uvāca</l>
	<l>sandhyābhāṣaṃ kim ucyeta bhagavān bobrūta niścitaṃ ||</l>
	<l>yoginīnāṃ mahāsamayaṃ śrāvakādyair na chidritaṃ || (53)</l>
</lg>

<lg xml:id="HT2.3.54">
	<l>hasitaṃ cekṣaṇābhyān tu āliṅgaṃ dvandakais tathā ||</l>
	<l>tantreṇāpi caturṇāṃ ca saṃdhyābhāṣaṃ na śabditaṃ || (54)</l>
</lg>

<lg xml:id="HT2.3.55">
	<l>bhagavān āha ||</l>
	<l>vakṣyāmy ahaṃ vajragarbha śṛṇu tvam ekacetasā ||</l>
	<l>saṃdhyābhāṣaṃ mahābhāṣaṃ samayasaṃketavistaraṃ || (55)</l>
</lg>

<lg xml:id="HT2.3.56">
	<l>madanaṃ madyaṃ balaṃ māṃsaṃ malayajaṃ milanaṃ mataṃ ||</l>
	<l>gatiḥ kheṭaḥ śavaḥ śrāyo asthyābharaṇaṃ niraṃśukaṃ || (56)</l>
</lg>

<lg xml:id="HT2.3.57">
	<l>āgatiḥ preṅkhaṇaṃ proktaṃ kṛpīṭaṃ ḍamarukaṃ mataṃ ||</l>
	<l>abhavyaṃ dunduraṃ khyātaṃ bhavyaṃ kāliñjaram mataṃ || (57)</l>
</lg>

<lg xml:id="HT2.3.58">
	<l>asparśaṃ diṇḍimaṃ proktaṃ kapālaṃ padmabhājanaṃ ||</l>
	<l>bhakṣaṃ tṛptikaraṃ jñeyaṃ vyañjanaṃ mālatīndhanaṃ || (58)</l>
</lg>

<lg xml:id="HT2.3.59">
	<l>gūthaṃ catuḥsamaṃ proktaṃ mūtraṃ kasturikā smṛtā ||</l>
	<l>svayaṃbhu sihlakaṃ jñeyaṃ śukraṃ karpūrakaṃ mataṃ || (59)</l>
</lg>

<lg xml:id="HT2.3.60">
	<l>mahāmāṃsaṃ sālijaṃ proktaṃ dvīndriyayogaṃ kunduruṃ ||</l>
	<l>vajraṃ bolakaṃ khyātaṃ padma kakkolakaṃ mataṃ || (60)</l>
</lg>

<lg xml:id="HT2.3.61">
	<l>kulaṃ pañcavidhaṃ khyātaṃ varṇabhedena bheditaṃ ||</l>
	<l>saṃdhyābhāṣata evaṃ syur buddhāś &lt;ca&gt; pañcakaulikāḥ || (61)</l>
</lg>

<lg xml:id="HT2.3.62">
	<l>ḍombī vajrakulī khyātā naṭī padmakulī tathā ||</l>
	<l>caṇḍālī ratnakulī caiva dvijā tāthāgatī matā || (62)</l>
</lg>

<lg xml:id="HT2.3.63">
	<l>rajakī karmakulī caiva etā mudrāḥ susiddhidā ||</l>
	<l>āsāṃ śukraṃ bhaved vajraṃ pūjayitvā pibed vratī || (63)</l>
</lg>

<lg xml:id="HT2.3.64">
	<l>vajragarbha mahāsattva yan mayā kathitaṃ tvayi ||</l>
	<l>tat sarvaṃ sādaraṃ grāhyaṃ sandhyābhāṣaṃ mahādbhutaṃ || (64)</l>
</lg>

<lg xml:id="HT2.3.65">
	<l>yo 'bhiṣikto 'tra hevajre na vadet saṃdhyābhāṣayā ||</l>
	<l>samayavidrohanaṃ tasya jāyate nātra saṃśayaḥ || (65)</l>
</lg>

<lg xml:id="HT2.3.66">
	<l>ity upadravacauraiś ca grahajvaraviṣādibhiḥ ||</l>
	<l><pb n="62"/>mṛyate 'sau yadi buddho 'pi saṃdhyābhāṣān na bhāṣayet || (66)</l>
</lg>

<lg xml:id="HT2.3.67">
	<l>svasamayavidāṃ prāpya yadi na bhāṣed idaṃ vacaḥ ||</l>
	<l>tadā kṣobhaṃ prakurvanti yoginyaś catuḥpīṭhajāḥ || (67)</l>
</lg>

	hevajrasarvatantranidhānasandhyābhāṣo nāma tṛtīyaḥ paṭalaḥ ||
