<lg xml:id="HT1.10.1">
	<l>athātaḥ saṃpravakṣyāmi maṇḍalasya yathākramaṃ ||</l>
	<l>śiṣyo 'bhiṣicyate yena vidhiś cāpi pravakṣyate || (1)</l>
</lg>

<lg xml:id="HT1.10.2">
	<l>vasudhāṃ śodhayed yogī prathamaṃ devatātmakaḥ ||</l>
	<l>hūṃvajrīkṛtya yatnena paścān maṇḍalam ālikhet || (2)</l>
</lg>

<lg xml:id="HT1.10.3">
	<l>udyāne vijane deśe bodhisattvagṛheṣu ca ||</l>
	<l>maṇḍalāgāramadhye ca vartayen maṇḍalam varaṃ || (3)</l>
</lg>

<lg xml:id="HT1.10.4">
	<l>divyena rajolekhena athavā madhyamena tu ||</l>
	<l>pañcaratnamayaiś cūrṇair athavā taṇḍulādibhiḥ || (4)</l>
</lg>

<lg xml:id="HT1.10.5">
	<l>trihastaṃ maṇḍalaṃ kāryaṃ trayāṅguṣṭhādhikan tataḥ |||</l>
	<l>vidyā tatra praveṣṭavyā divyā pañcakulodbhavā || (5)</l>
</lg>

<lg xml:id="HT1.10.6">
	<l>athavā yā kā yathālabdhā ṣoḍaśābdā tathaiva ca ||</l>
	<l>tāvad dhi sevyate mudrā yāvac chukravatī bhavet || (6)</l>
</lg>

<lg xml:id="HT1.10.7">
	<l>mudrāyāś ca mukhaṃ baddhvā upāyasya mukhaṃ tathā ||</l>
	<l><pb n="36"/>sevayā tatra yad bhūtaṃ śiṣyavaktre nipātayet || (7)</l>
</lg>

<lg xml:id="HT1.10.8">
	<l>kāritavyaṃ ca tatraiva samarasaṃ śiṣyagocaraṃ ||</l>
	<l>svasaṃvedyād bhaved jñānaṃ svaparavittivarjitaṃ || (8)</l>
</lg>

<lg xml:id="HT1.10.9">
	<l>khasamaṃ virajaṃ śūnyaṃ bhāvābhāvātmakaṃ paraṃ ||</l>
	<l>prajñopāyavyatimiśraṃ rāgārāgavimiśritaṃ || (9)</l>
</lg>

<lg xml:id="HT1.10.10">
	<l>sa eva prāṇināṃ prāṇaḥ sa eva paramākṣaraḥ ||</l>
	<l>sarvavyāpī sa evāsau sarvadehavyavasthitaḥ || (10)</l>
</lg>

<lg xml:id="HT1.10.11">
	<l>sa evāsau mahāprāṇaḥ sa evāsau jaganmayaḥ ||</l>
	<l>bhāvābhāvau tadudbhūtau anyāni yāni tāni ca || (11)</l>
</lg>

<lg xml:id="HT1.10.12">
	<l>sarvaṃ vijñānarūpaṃ ca puruṣaḥ purāṇa īśvaro ||</l>
	<l>ātmā jīvaṃ ca sattvaṃ ca kālaḥ puḍgala eva ca ||</l>
	<l>sarvabhāvasvabhāvo 'sau māyārūpī ca saṃsthitaḥ || (12)</l>
</lg>

<lg xml:id="HT1.10.13">
	<l>prathamānandamātran tu paramānandaṃ dvisaṃkhyataḥ ||</l>
	<l>tṛtīyaṃ viramākhyaṃ ca caturthaṃ sahajaṃ sṛtaṃ || (13)</l>
</lg>

<lg xml:id="HT1.10.14">
	<l>evaṃ śrutvā tu vai sarve vajragarbhādayo budhāḥ ||</l>
	<l>paramavismayāpannā mūrchitāḥ patitāvanau || (14)</l>
</lg>

<lg xml:id="HT1.10.15">
	<l>prathamānandaṃ jagadrūpaṃ paramānandaṃ jagat tathā ||</l>
	<l>viramānandaṃ jagac caiva na vidyate sahajaṃ triṣu || (15) iti ||</l>
</lg>

<lg xml:id="HT1.10.16">
	<l>bhagavān āha hevajraḥ sarvabuddhaikavigrahaḥ ||</l>
	<l>saṃśayāpanayaṃ divyaṃ vajragarbhasya bodhaye || (16)</l>
</lg>

<lg xml:id="HT1.10.17">
	<l>na rāgo na virāgaś ca madhyamaṃ nopalabhyate ||</l>
	<l>trayāṇāṃ varjanād eva sahajaṃ bodhir ucyate || (17)</l>
</lg>

<lg xml:id="HT1.10.18">
	<l>athavā sarvātmakaḥ saivāthavā sarvair vivarjitaḥ ||</l>
	<l>viramādau sa lakṣyata ānandatrayavarjitaḥ || (18)</l>
</lg>

<lg xml:id="HT1.10.19">
	<l>prathamaṃ meghavad bhāti siddhe tu māyāvad bhavet ||</l>
	<l>sahasā svapnavad bhāti svapijāgradabhedavat || (19)</l>
</lg>

<lg xml:id="HT1.10.20">
	<l>abhedalakṣaṇāsiddhau mudrāyogī tu sidhyati ||</l>
	<l>ity āha maṇḍalaṃ śāstā catuṣkoṇaṃ samujjvalaṃ || (20)</l>
</lg>

<lg xml:id="HT1.10.21">
	<l>caturdvāraṃ mahādīptaṃ hārārdhahārabhūṣitaṃ ||</l>
	<l>śrakcitracāmarair yuktaṃ aṣṭastambhopaśobhitaṃ || (21)</l>
</lg>

<lg xml:id="HT1.10.22">
	<l>vajrasūtrair samāyuktaṃ nānāpuṣpopaśobhitaṃ ||</l>
	<l>dhūpaṃ dīpaṃ tathā gandham aṣṭakalaśādibhir yutaṃ || (22)</l>
</lg>

<lg xml:id="HT1.10.23">
	<l>te ca sapallavāgrāḥ syur vastrācchāditakandharāḥ ||</l>
	<l><pb n="38"/>pañcaratnaparikṣiptaṃ dadyād vijayaṃ pūrvataḥ || (23)</l>
</lg>

<lg xml:id="HT1.10.24">
	<l>navena suniyuktena supramāṇena cāruṇā ||</l>
	<l>sūtreṇa sūtrayet prājñaḥ sveṣṭadevatārūpataḥ || (24)</l>
</lg>

<lg xml:id="HT1.10.25">
	<l>cakreśasya japel lakṣaṃ māṇḍaleyasya cāyutaṃ ||</l>
	<l>pūrvoktenaiva mantreṇa śodhayed dharaṇīṃ budhaḥ || (25)</l>
</lg>

<lg xml:id="HT1.10.26">
	<l>baliṃ ca dāpayet tatra prāg akārādimantrataḥ ||</l>
	<l>rakṣāṃ caiva yathādiṣṭāṃ yathā dhyāne tathātra ca || (26)</l>
</lg>

<lg xml:id="HT1.10.27">
	<l>prakāśitās tu ye sekā vidhivad dadyāt svamaṇḍale ||</l>
	<l>pūjā cābhyarcanā caiva yathākhyātā tathātra ca || (27)</l>
</lg>

<lg xml:id="HT1.10.28">
	<l>śuddhaṃ dvipuṭam ālikhya cakraṃ gauryādīnāṃ tatha ||</l>

</lg>

<lg xml:id="HT1.10.29">

</lg>

<lg xml:id="HT1.10.30">

</lg>

<lg xml:id="HT1.10.31">
	<l>snātaḥ śuciḥ sugandhāṅgo citrābharaṇabhūṣitaḥ ||</l>
	<l>hūṃhūṃkārakṛtāṭopo hīhīkārabhayānakaḥ || (31)</l>
</lg>

<lg xml:id="HT1.10.32">
	<l>paścāt tattvaṃ samākhyātaṃ viśuddhaṃ jñānarūpiṇaṃ ||</l>
	<l>saṃsāravyavadānena nāsti bhedo manāg api || (32)</l>
</lg>

<lg xml:id="HT1.10.33">
	<l>paramaratau na ca bhāva na bhāvaka ||</l>
	<l>na ca vigraha na ca grāhya na grāhaka ||</l>
	<l>māṃsa na śoṇita viṣṭa na mūtraṃ  ||</l>
	<l>na charda na moha na śaucapavitraṃ || (33)</l>
</lg>

<lg xml:id="HT1.10.34">
	<l>rāga na dveśa na moha na īrṣyā ||</l>
	<l>na ca paiśunya na ca māna na dṛśyaṃ ||</l>
	<l>bhāva na bhāvaka mitra na śatru ||</l>
	<l>nistaraṅga sahajākhyavicitraṃ || (34)</l>
</lg>

<lg xml:id="HT1.10.40">
