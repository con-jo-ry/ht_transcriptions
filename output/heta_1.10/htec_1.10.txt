<div n="10" type="paṭala">
<lg xml:id="HT1.10.1">
	<l>athātaḥ saṃpravakṣyāmi maṇḍalasya yathākramam |</l>
	<l>śiṣyo 'bhiṣicyate yena vidhiś cāpi pravakṣyate || 1 ||</l>
</lg>

<lg xml:id="HT1.10.2">
	<l>vasudhāṃ śodhayed yogī prathamaṃ devatātmakaḥ |</l>
	<l>hūṁvajrīkṛtya yatnena paścān maṇḍalam ālikhet || 2 ||</l>
</lg>

<lg xml:id="HT1.10.3">
	<l>udyāne vijane deśe bodhisattvagṛheṣu ca |</l>
	<l>maṇḍalāgāramadhye ca vartayen maṇḍalaṃ varam || 3 ||</l>
</lg>

<lg xml:id="HT1.10.4">
	<l>divyena rajolekhena athavā madhyamena tu |</l>
	<l>pañcaratnamayaiś cūrṇair athavā taṇḍulādibhiḥ || 4 ||</l>
</lg>

<lg xml:id="HT1.10.5">
	<l>trihastaṃ maṇḍalaṃ kāryaṃ trayāṅguṣṭhādhikaṃ catuḥ |</l>
	<l>vidyā tatra praveṣṭavyā divyā pañcakulodbhavā || 5 ||</l>
</lg>

<lg xml:id="HT1.10.6">
	<l>athavā yāṃ tāṃ yathālabdhāṃ ṣoḍaśābdāṃ tathaiva ca |</l>
	<l>tāvat sevyate mudrā yāvac chukravatī bhavet || 6 ||</l>
</lg>

<lg xml:id="HT1.10.7">
	<l>mudrāyās tu mukhaṃ baddhvā upāyasya mukhaṃ tathā |</l>
	<l>sevayā tatra yad bhūtaṃ śiṣyavaktre nipātayet || 7 ||</l>
</lg>

<lg xml:id="HT1.10.8">
	<l>kāritavyaṃ ca tatraiva samarasaṃ śiṣyagocaram |</l>
	<l>svasaṃvedyād bhavej jñānaṃ svaparasaṃvittivarjitam || 8 ||</l>
</lg>

<lg xml:id="HT1.10.9">
	<l>khasamaṃ virajaṃ śūnyaṃ bhāvābhāvātmakaṃ param |</l>
	<l>prajñopāyavyatimiśraṃ rāgārāgavimiśritam || 9 ||</l>
</lg>

<lg xml:id="HT1.10.10">
	<l>sa eva prāṇināṃ prāṇaḥ sa eva paramākṣaraḥ |</l>
	<l>sarvavyāpī sa evāsau sarvadehe vyavasthitaḥ || 10 ||</l>
</lg>

<lg xml:id="HT1.10.11">
	<l>sa evāsau jaganmayaḥ |</l>
	<l>bhāvābhāvau tadudbhūtau anyāni yāni tāni ca || 11 ||</l>
</lg>

<lg xml:id="HT1.10.12">
	<l>sarvavijñānarūpaṃ ca puruṣaṃ purāṇam īśvaram |</l>
	<l>ātmā jīvaṃ ca sattvaṃ ca kālaḥ pudgala eva ca |</l>
	<l>sarvabhāvasvabhāvo 'sau māyārūpīva saṃsthitaḥ || 12 ||</l>
</lg>

<lg xml:id="HT1.10.13">
	<l>prathamānandamātraṃ tu paramānandaṃ dvisaṃkhyataḥ |</l>
	<l>tṛtīyaṃ viramākhyaṃ tu caturthaṃ sahajaṃ sṛtam || 13 ||</l>
</lg>

<lg xml:id="HT1.10.14">
	<l>evaṃ śrutvā tu vai sarve vajragarbhādayo budhāḥ |</l>
	<l>paramavismayam āpannā mūrchitāḥ patitāvanau || 14 ||</l>
</lg>

<lg xml:id="HT1.10.15">
	<l>prathamānandaṃ jagadrūpaṃ paramānandaṃ jagat tathā |</l>
	<l>viramānandaṃ jagac caiva na vidyet sahajaṃ triṣu || 15 || iti ||</l>
</lg>

<lg xml:id="HT1.10.16">
	<l>bhagavān āha hevajraḥ sarvabuddhaikavigrahaḥ |</l>
	<l>saṃśayāpanayaṃ divyaṃ vajragarbhasya bodhaye || 16 ||</l>
</lg>

<lg xml:id="HT1.10.17">
	<l>na rāgo na virāgaś ca madhyamā nopalabhyate |</l>
	<l>trayāṇāṃ varjanād eva sahajaḥ saṃbodhir ucyate || 17 ||</l>
</lg>

<lg xml:id="HT1.10.18">
	<l>athavā sarvātmakaḥ saiva athavā sarvair vivarjitaḥ |</l>
	<l>viramādau lakṣayet tac ca ānandatrayavarjitam || 18 ||</l>
</lg>

<lg xml:id="HT1.10.19">
	<l>prathamaṃ meghavad bhāti siddhe tu māyāvad bhavet |</l>
	<l>sahasā svapnavad bhāti svapijāgradabhedavat || 19 ||</l>
</lg>

<lg xml:id="HT1.10.20">
	<l>abhedalakṣaṇāsiddhau mudrāyogīti sidhyati |</l>
	<l>ity āha maṇḍalaṃ śāstā catuṣkoṇaṃ samujjvalam || 20 ||</l>
</lg>

<lg xml:id="HT1.10.21">
	<l>caturdvāraṃ mahādīptaṃ hārārdhahārabhūṣitam |</l>
	<l>srakcitracāmarair yuktam aṣṭastambhopaśobhitam || 21 ||</l>
</lg>

<lg xml:id="HT1.10.22">
	<l>vajrasūtrasamāyuktaṃ nānāpuṣpopaśobhitam |</l>
	<l>dhūpaṃ dīpaṃ tathā gandham aṣṭakalaśādibhir yutam || 22 ||</l>
</lg>

<lg xml:id="HT1.10.23">
	<l>te ca pallavāgrāḥ syur vastrācchāditakandharāḥ |</l>
	<l>pañcaratnaparikṣiptaṃ dadyād vijayaṃ ca pūrvataḥ || 23 ||</l>
</lg>

<lg xml:id="HT1.10.24">
	<l>navena suniyuktena supramāṇena cāruṇā |</l>
	<l>sūtreṇa sūtrayet prājñaḥ sveṣṭadevatarūpataḥ || 24 ||</l>
</lg>

<lg xml:id="HT1.10.25">
	<l>cakreśasya japel lakṣaṃ māṇḍaleyasya tathāyutam |</l>
	<l>pūrvoktenaiva mantreṇa śodhayed dharaṇīṃ budhaḥ || 25 ||</l>
</lg>

<lg xml:id="HT1.10.26">
	<l>baliṃ ca dāpayet tatra prāg akārādimantrataḥ |</l>
	<l>rakṣāṃ caiva yathoddiṣṭāṃ yathā dhyāne tathātra ca || 26 ||</l>
</lg>

<lg xml:id="HT1.10.27">
	<l>prakāśitās tu ye sekā vidhivad dadyāt svamaṇḍale |</l>
	<l>pūjā cābhyarthanā caiva yathākhyātā tathātra ca || 27 ||</l>
</lg>

<lg xml:id="HT1.10.28">
	<l>śuddhaṃ dvipuṭam vilikhyāśu gauryādīnāṃ likhet tataḥ |</l>
	<l>pūrveṇa likhet kartiṃ dakṣiṇe paścime tathā || 28 ||</l>
</lg>

<lg xml:id="HT1.10.29">
	<l>uttare agnikoṇe ca nairṛtye vāyave tathā |</l>
	<l>aiśāne ca yathākathitam adhordhve ca tathā tathā || 29 ||</l>
</lg>

<lg xml:id="HT1.10.30">
	<l>vajrasattvakṛtāṭopaḥ kṛtam ālīḍha pādayoḥ |</l>
	<l>praviśen maṇḍalācāryo dvibhujahevajrayogataḥ || 30 ||</l>
</lg>

<lg xml:id="HT1.10.31">
	<l>snātaḥ śuciḥ sugandhāṅgo vicitrābharaṇabhūṣitaḥ |</l>
	<l>hūṁhūṁkārakṛtāṭopo hīhīkārabhayānakaḥ || 31 ||</l>
</lg>

<lg xml:id="HT1.10.32">
	<l>paścāt tattvaṃ samākhyāti viśuddhaṃ jñānarūpiṇam |</l>
	<l>saṃsāravyavadānena nāsti bhedo manāg api || 32 ||</l>
</lg>

<lg xml:id="HT1.10.33">
	<l>paramaratau na ca bhāva na bhāvaka |</l>
	<l>na ca vigraha na ca grāhya na grāhaka |</l>
	<l>māṃsa na śoṇita viṣṭa na mūtraṃ  |</l>
	<l>na ca ghṛṇa moha na śaucapavitram || 33 ||</l>
</lg>

<lg xml:id="HT1.10.34">
	<l>rāga na dveṣa na moha na īrṣyā</l>
	<l>na ca paiśunya na māna na dṛśyam |</l>
	<l>bhāva na bhāvaka mitra na śatruṃ</l>
	<l>nistaraṅga sahajākhya vicitram || 34 ||</l>
</lg>

<lg xml:id="HT1.10.35">
	<l>ity āha vajragarbhākhyaḥ kasmād bhūtātmakaṃ bhavet |</l>
	<l>dehaṃ svabhāvataḥ śuddham ādāv evāsvabhāvakam || 35 ||</l>
</lg>

<lg xml:id="HT1.10.36">
	<l>tatrāha bhagavān vajrī ḍākinīnāṃ sukhaṃdadaḥ |</l>
	<l>nistaraṅgasvarūpātmā sarvadehe vyavasthitaḥ || 36 ||</l>
</lg>

<p xml:id="HT1.10.37">
	he bhagavan | kasmān mahābhautikaḥ skandhaḥ || 37 || 
</p>

<lg xml:id="HT1.10.38">
	<l>bhagavān āha—</l>
	<l>bolakakkolayogena |</l>
	<l>sparśāt kāṭhinyadharmeṇa pṛthivī tatra jāyate || 38 ||</l>
</lg>

<lg xml:id="HT1.10.39">
	<l>bodhicittadravākārād abdhātoś ca sambhavaḥ |</l>
	<l>tajo jāyate gharṣaṇād gamanād vāyuḥ prakīrtitaḥ || 39 ||</l>
</lg>

<lg xml:id="HT1.10.40">
	<l>saukhyam ākāśadhātuś ca pañcabhiḥ pariveṣṭitaḥ |</l>
	<l>tasmāt saukhyaṃ na tattvākhyaṃ mahābhūtaṃ yataḥ sukham || 40 ||</l>
</lg>

<lg xml:id="HT1.10.41">
	<l>sahajābhyāṃ yad utpannaṃ sahajaṃ tan nigadyate |</l>
	<l>svabhāvaṃ sahajam ity uktaṃ sarvākāraikasaṃvaram || 41 ||</l>
</lg>

<lg xml:id="HT1.10.42">
	<l>kṛpopāyo bhaved yogī mudrā hetuviyogataḥ |</l>
	<l>śūnyatākaruṇābhinnaṃ bodhicittam iti smṛtam || 42 ||</l>
</lg>

<lg xml:id="HT1.10.43">
	<l>na mantrajāpo na tapo na homo</l>
	<l>na māṇḍaleyaṃ na ca maṇdalaṃ ca |</l>
	<l>sa mantrajāpaḥ sa tapaḥ sa homas</l>
	<l>taṃ māṇḍaleyaṃ taṃ maṇḍalaṃ</l>
	<l>samāsataś cittasamājarūpī || 44 ||</l>
</lg>

<p xml:id="HT1.10.col">
	abhiṣekapaṭalo daśamaḥ ||
</p>
