<div n="10" type="paṭala">
<lg xml:id="HT1.10.1">
	<l><pb n="112"/>athātaḥ saṃpravakṣyāmi maṇḍalasya yathākramam |</l>
	<l>śiṣyo 'bhiṣicyate yena vidhiś cāpi pravakṣyate || 1 ||</l>
</lg>

<lg xml:id="HT1.10.2">
	<l>vasudhāṃ śodhayed yogī prathamaṃ devatātmakaḥ |</l>
	<l>hūṃvajrīkṛtya yatnena paścān maṇḍalam ālikhet || 2 ||</l>
</lg>

<lg xml:id="HT1.10.3">
	<l>udyāne vijane deśe bodhisattvagṛheṣu ca |</l>
	<l>maṇḍalāgāramadhye ca vartayen maṇḍalam varam || 3 ||</l>
</lg>

<lg xml:id="HT1.10.4">
	<l>divyena rajolekhena athavā madhyamena tu |</l>
	<l>pañcaratnamayaiś cūrṇair athavā taṇḍulādibhiḥ || 4 ||</l>
</lg>

<lg xml:id="HT1.10.5">
	<l><pb n="113"/>trihastaṃ maṇḍalaṃ kāryaṃ trayāṅguṣṭhādhikaṃ tataḥ |</l>
	<l>vidyā tatra praveṣṭavyā divyā pañcakulodbhavā || 5 ||</l>
</lg>

<lg xml:id="HT1.10.6">
	<l>athavā yā kā yathālabdhā ṣoḍaśābdā tathaiva ca |</l>
	<l>tāvad dhi sevyate mudrā yāvac chukravatī bhavet || 6 ||</l>
</lg>

<lg xml:id="HT1.10.7">
	<l>mudrāyāś ca mukhaṃ baddhvā upāyasya mukhaṃ tathā |</l>
	<l>sevayā tatra yad bhūtaṃ śiṣyavaktre nipātayet || 7 ||</l>
</lg>

<lg xml:id="HT1.10.8">
	<l>kāritavyaṃ ca tatraiva samarasaṃ śiṣyagocaram |</l>
	<l>svasaṃvedyād bhaved jñānaṃ svaparavittivarjitam || 8 ||</l>
</lg>

<lg xml:id="HT1.10.9">
	<l><pb n="114"/>khasamaṃ virajaṃ śūnyaṃ bhāvābhāvātmakaṃ param |</l>
	<l>prajñopāyavyatimiśraṃ rāgārāgavimiśritam || 9 ||</l>
</lg>

<lg xml:id="HT1.10.10">
	<l>sa eva prāṇināṃ prāṇaḥ sa eva paramākṣaraḥ |</l>
	<l>sarvavyāpī sa evāsau sarvadehavyavasthitaḥ || 10 ||</l>
</lg>

<lg xml:id="HT1.10.11">
	<l><pb n="115"/>sa evāsau mahāprāṇaḥ sa evāsau jaganmayaḥ |</l>
	<l>bhāvābhāvau tadudbhūtau anyāni yāni tāni ca || 11 ||</l>
</lg>

<lg xml:id="HT1.10.12">
	<l>sarvaṃ vijñānarūpaṃ ca puruṣaḥ purāṇa īśvaraḥ |</l>
	<l>ātmā jīvaṃ ca sattvaṃ ca kālaḥ pudgala eva ca |</l>
	<l>sarvabhāvasvabhāvo 'sau māyārūpī ca saṃsthitaḥ || 12 ||</l>
</lg>

<lg xml:id="HT1.10.13">
	<l><pb n="116"/>prathamānandamātraṃ tu paramānandaṃ dvisaṃkhyataḥ |</l>
	<l>tṛtīyaṃ viramākhyaṃ ca caturthaṃ sahajaṃ sṛtam || 13 ||</l>
</lg>

<lg xml:id="HT1.10.14">
	<l>evaṃ śrutvā tu vai sarve vajragarbhādayo budhāḥ |</l>
	<l>paramavismayāpannā mūrchitāḥ patitāvanau || 14 ||</l>
</lg>

<lg xml:id="HT1.10.15">
	<l>prathamānandaṃ jagadrūpaṃ paramānandaṃ jagat tathā |</l>
	<l>viramānandaṃ jagac caiva na vidyate sahajaṃ triṣu || 15 || iti ||</l>
</lg>

<lg xml:id="HT1.10.16">
	<l><pb n="117"/>bhagavān āha hevajraḥ sarvabuddhaikavigrahaḥ |</l>
	<l>saṃśayāpanayaṃ divyaṃ vajragarbhasya bodhaye || 16 ||</l>
</lg>

<lg xml:id="HT1.10.17">
	<l>na rāgo na virāgaś ca madhyamaṃ nopalabhyate |</l>
	<l>trayāṇāṃ varjanād eva sahajaṃ bodhir ucyate || 17 ||</l>
</lg>

<lg xml:id="HT1.10.18">
	<l><pb n="118"/>athavā sarvātmakaḥ saivāthavā sarvair vivarjitaḥ |</l>
	<l>viramādau sa lakṣyata ānandatrayavarjitaḥ || 18 ||</l>
</lg>

<lg xml:id="HT1.10.19">
	<l><pb n="119"/>prathamaṃ meghavad bhāti siddhe tu māyāvad bhavet |</l>
	<l>sahasā svapnavad bhāti svapijāgradabhedavat || 19 ||</l>
</lg>

<lg xml:id="HT1.10.20">
	<l>abhedalakṣaṇāsiddhau mudrāyogī tu sidhyati |</l>
	<l>ity āha maṇḍalaṃ śāstā catuṣkoṇaṃ samujjvalam || 20 ||</l>
</lg>

<lg xml:id="HT1.10.21">
	<l><pb n="120"/>caturdvāraṃ mahādīptaṃ hārārdhahārabhūṣitam |</l>
	<l>srakcitracāmarair yuktam aṣṭastambhopaśobhitam || 21 ||</l>
</lg>

<lg xml:id="HT1.10.22">
	<l>vajrasūtraiḥ samāyuktaṃ nānāpuṣpopaśobhitam |</l>
	<l>dhūpaṃ dīpaṃ tathā gandham aṣṭakalaśādibhir yutam || 22 ||</l>
</lg>

<lg xml:id="HT1.10.23">
	<l>te ca sapallavāgrāḥ syur vastrācchāditakandharāḥ |</l>
	<l>pañcaratnaparikṣiptaṃ dadyād vijayaṃ pūrvataḥ || 23 ||</l>
</lg>

<lg xml:id="HT1.10.24">
	<l>navena suniyuktena supramāṇena cāruṇā |</l>
	<l>sūtreṇa sūtrayet prājñaḥ sveṣṭadevatārūpataḥ || 24 ||</l>
</lg>

<lg xml:id="HT1.10.25">
	<l>cakreśasya japel lakṣaṃ māṇḍaleyasya cāyutam |</l>
	<l>pūrvoktenaiva mantreṇa śodhayed dharaṇīṃ budhaḥ || 25 ||</l>
</lg>

<lg xml:id="HT1.10.26">
	<l><pb n="121"/>baliṃ ca dāpayet tatra prāg akārādimantrataḥ |</l>
	<l>rakṣāṃ caiva yathādiṣṭāṃ yathā dhyāne tathātra ca || 26 ||</l>
</lg>

<lg xml:id="HT1.10.27">
	<l>prakāśitās tu ye sekā vidhivad dadyāt svamaṇḍale |</l>
	<l>pūjā cābhyarcanā caiva yathākhyātā tathātra ca || 27 ||</l>
</lg>

<lg xml:id="HT1.10.28">
	<l>śuddhaṃ dvipuṭam ālikhya cakraṃ gauryādīnāṃ tatha |</l>
	<l>pūrveṇa likhet kartrikāṃ dakṣiṇe paścime tathā || 28 ||</l>
</lg>

<lg xml:id="HT1.10.29">
	<l>uttare cāgnikoṇe ca nairṛtye vāyave tathā |</l>
	<l>aiśāne ca yathākhyātam adhaś cordhvaṃ tathā tathā || 29 ||</l>
</lg>

<lg xml:id="HT1.10.30">
	<l>vajrasattvakṛtāṭopaḥ kṛtālīḍhaś ca pādayoḥ |</l>
	<l>praviśen maṇḍalācāryo dvibhujahevajrayogataḥ || 30 ||</l>
</lg>

<lg xml:id="HT1.10.31">
	<l><pb n="122"/>snātaḥ śuciḥ sugandhāṅgo citrābharaṇabhūṣitaḥ |</l>
	<l>hūṁhūṁkāra kṛtāṭopo hīhīkāra bhayānakaḥ || 31 ||</l>
</lg>

<lg xml:id="HT1.10.32">
	<l>paścāt tattvaṃ samākhyātaṃ viśuddhaṃ jñānarūpiṇam |</l>
	<l>saṃsāravyavadānena nāsti bhedo manāg api || 32 ||</l>
</lg>

<lg xml:id="HT1.10.33">
	<l>paramaratau na ca bhāva na bhāvaka |</l>
	<l>na ca vigraha na ca grāhya na grāhaka |</l>
	<l>māṃsa na śoṇita viṣṭa na mūtra |</l>
	<l>na charda na moha na śaucapavitram || 33 ||</l>
</lg>

<lg xml:id="HT1.10.34">
	<l><pb n="123"/>rāga na dveṣa na moha na īrṣyā |</l>
	<l>na ca paiśunya na ca māna na dṛśyam |</l>
	<l>bhāva na bhāvaka mitra na śatru |</l>
	<l>nistaraṅga sahajākhyavicitram || 34 ||</l>
</lg>

<lg xml:id="HT1.10.35">
	<l>ity āha vajragarbhākhyaḥ kasmād bhūtātmakaṃ bhavet |</l>
	<l>dehaṃ svabhāvataḥ śuddham ādāv evāsvabhāvakam || 35 ||</l>
</lg>

<lg xml:id="HT1.10.36">
	<l><pb n="124"/>tatrāha bhagavān vajrī ḍākinīnāṃ sukhaṃdadaḥ |</l>
	<l>nistaraṅgaḥ svarūpātmā sarva dehe vyavasthitaḥ || 36 ||</l>
</lg>

<p xml:id="HT1.10.37">
	vajragarbha āha—he bhagavan, kasmān mahābhautikaskandhaḥ || 37 || 
</p>

<lg xml:id="HT1.10.38">
	<l>bhagavān āha—</l>
	<l>bolakakkolayogena kunduruṃ kurute vratī |</l>
	<l>sparśāt kāṭhinyadharmeṇa pṛthivī tatra jāyate || 38 ||</l>
</lg>

<lg xml:id="HT1.10.39">
	<l>bodhicittadravākārād abdhātoś caiva saṃbhavaḥ |</l>
	<l>gharṣaṇād jāyate tejo gamanād vāyuḥ prakīrtitaḥ || 39 ||</l>
</lg>

<lg xml:id="HT1.10.40">
	<l><pb n="125"/>saukhyam ākāśadhātuś ca pañcabhiḥ pariveṣṭitaḥ |</l>
	<l>tasmāt saukhyaṃ na tattvākhyaṃ mahābhūtaṃ yataḥ sukham || 40 ||</l>
</lg>

<lg xml:id="HT1.10.41">
	<l>sahajātyāṃ yad utpannaṃ sahajaṃ tat prakīrtitam |</l>
	<l>svabhāvaṃ sahajaṃ proktaṃ sarvākāraikasaṃvaram || 41 ||</l>
</lg>

<lg xml:id="HT1.10.42">
	<l><pb n="126"/>kṛpopāyo bhaved yogī mudrā hetuviyogataḥ |</l>
	<l>śūnyatākaruṇābhinnaṃ bodhicittam iti smṛtam || 42 ||</l>
</lg>

<lg xml:id="HT1.10.43">
	<l><pb n="127"/>na mantrajāpo na tapo na homo</l>
	<l>na maṇḍaleyaṃ na ca maṇdalaṃ ca |</l>
	<l>sa mantrajāpaḥ sa tapaḥ sa homas</l>
	<l>tan maṇḍaleyaṃ tan maṇḍalaṃ ca || 43 ||</l>
	<l>samāsataś cittasamājarūpī || 44 ||</l>
</lg>

	abhiṣekapaṭalo daśamaḥ ||
