	<lg>

	</lg>
</p>

<lg xml:id="HT2.1.3">

</lg>

<lg xml:id="HT2.1.4">

</lg>

<lg xml:id="HT2.1.7">

</lg>

<lg xml:id="HT2.1.9">

</lg>

</lg>
