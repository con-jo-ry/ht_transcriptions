<div n="8" type="paṭala">
<p xml:id="HT1.8.1">
	<pb n="74"/>atha yoginīcakraṃ vyākhyāsyāmaḥ—
	<lg>
		<l>khadhātau bhagaṃ dhyātvā madhye kurvīta bhāvanām |</l>
		<l>cakraṃ pūrvaṃ yathānyāyaṃ devatānāṃ yathodayam || 1 ||</l>
	</lg>
</p>

<lg xml:id="HT1.8.2">
	<l>cakraṃ kṣoṇī jalaṃ pūrvaṃ yathānyāyaṃ hutāśanam |</l>
	<l>devatānāṃ mahāvāyur bhāvakaś ca yathodayam || 2 ||</l>
</lg>

<lg xml:id="HT1.8.3">
	<l><pb n="75"/>dharmodayodbhavaṃ cakraṃ dvipuṭaṃ hi nirāmayam |</l>
	<l>kiñjalkena bhaved ekaṃ trikoṇenāparaṃ śrutam || 3 ||</l>
</lg>

<lg xml:id="HT1.8.4">
	<l>tanmadhye cintayen mṛtakaṃ pañcadaśāsanātmakam |</l>
	<l>tasyopari bhavec candraś candrasyopari bījakam || 4 ||</l>
</lg>

<lg xml:id="HT1.8.5">
	<l><pb n="76"/>paścān mārtaṇḍam ākrāntam dvayor melā mahat sukham |</l>
	<l>sthitāliś candrarūpeṇa kālirūpeṇa bhāskaraḥ || 5 ||</l>
</lg>

<lg xml:id="HT1.8.6">
	<l>candrasūryadvayor melād gauryādyas te prakīrtitāḥ |</l>
	<l>ādarśajñānavāṃś candraḥ samatāvān saptāśvikaḥ || 6 ||</l>
</lg>

<lg xml:id="HT1.8.7">
	<l><pb n="77"/>bījaiś cihnaiḥ svadevasya pratyavekṣaṇam ucyate |</l>
	<l>sarvair ekam anusṭhānaṃ niṣpattiḥ śuddhidharmatā || 7 ||</l>
</lg>

<lg xml:id="HT1.8.8">
	<l>ākārān bhāvayet pañca vidhānaiḥ kathitair budhaḥ |</l>
	<l>ālikālisamāyogo vajrasattvasya viṣṭaraḥ || 8 ||</l>
</lg>

<lg xml:id="HT1.8.9">
	<l>akṣarodbhavapiṇḍasya hūṁphaṭkārau na ceṣyete |</l>
	<l>sattvabimbasamudbhūtaṃ maṇḍaleśaṃ vibhāvayet || 9 ||</l>
</lg>

<lg xml:id="HT1.8.10">
	<l><pb n="78"/>pūrvavad vaktracihnādyaiś candrakāntimaṇiprabham |</l>
	<l>evaṃ sarve ca niṣpannāḥ prajñopāyasvabhāvataḥ || 10 ||</l>
</lg>

<lg xml:id="HT1.8.11">
	<l>prajñālikālyupāyeti candrārkasya prabhedanāt |</l>
	<l>gauryādyāś ca bhavanty asmād varṇabhedaṃ paraṃ pṛthak || 11 ||</l>
</lg>

<lg xml:id="HT1.8.12">
	<l><pb n="79"/>adhyātmakapuṭe tāvat sthitā vai pañcayoginyaḥ |</l>
	<l>pañcaskandhasvabhāvena bhāvayed yogavit sadā || 12 ||</l>
</lg>

<lg xml:id="HT1.8.13">
	<l>indre vajrā yame gaurī vāruṇyāṃ vāriyoginī |</l>
	<l>kauveryāṃ vajraḍākī ca madhye nairātmyayoginī || 13 ||</l>
</lg>

<lg xml:id="HT1.8.14">
	<l>bāhyapuṭe punaḥ <lb/>gaurī caurī vetālī ca ghasmarī pukkasī tathā |</l>
	<l>śavarī caṇḍālī caiva aṣṭamī ḍombinī matā || 14 ||</l>
</lg>

<lg xml:id="HT1.8.15">
	<l>adhovaty ūrdhvavaty eva khecarī bhūcarī smṛtā |</l>
	<l>bhavanirvāṇasvabhāvena sthitāv etau dvidevate || 15 ||</l>
</lg>

<lg xml:id="HT1.8.16">
	<l><pb n="80"/>sarvā devatyaḥ </l>
	<l>kṛṣṇavarṇā mahāraudrā pañcamudrāvibhūṣitāḥ |</l>
	<l>ekavaktrāś ca raktākṣāḥ kartrikapāladhṛkkarāḥ || 16 ||</l>
</lg>

<lg xml:id="HT1.8.17">
	<l>cakrī kuṇḍalakaṇṭhī ca haste rūcakamekhalāḥ |</l>
	<l>pañcabuddhaviśuddhyā ca pañcaite śuddhamudrakāḥ || 17 ||</l>
</lg>

<lg xml:id="HT1.8.18">
	<l>sarvā etādṛśāḥ khyātā yathā nairātmyayoginī |</l>
	<l>kapālaikakaravyagrā dakṣiṇe kartridhārikāḥ || 18 ||</l>
</lg>

<lg xml:id="HT1.8.19">
	<l>khaṭvāṅgaṃ caiva vāmena vyāghracarmāvṛtā kaṭiḥ |</l>
	<l>śavārūḍhā jvaladdīptā dvibhujāḥ piṅgamūrdhajāḥ || 19 ||</l>
</lg>

<lg xml:id="HT1.8.20">
	<l>tathā mānādi ṣaḍdoṣān kartituṃ kartrikā sthitā |</l>
	<l>bhāvābhāvavikalpasya śirasā padmabhājanam || 20 ||</l>
</lg>

<lg xml:id="HT1.8.21">
	<l><pb n="81"/>raktaṃ ca caturmārāṇāṃ pīyate siddhihetave |</l>
	<l>kaṭvāṅgaśūnyatākāraiḥ śavopāyena kalpitam || 21 ||</l>
</lg>

<lg xml:id="HT1.8.22">
	<l>etena bhāvayec cakraṃ laghu siddhim avāpnuyāt |</l>
	<l>prathame bhāvayet kṛṣṇāṃ dvitīye raktam eva ca || 22 ||</l>
</lg>

<lg xml:id="HT1.8.23">
	<l>tṛtīye bhāvayet pītāṃ caturthe haritāṃ tathā |</l>
	<l>pañcame nīlavarṇāñ ca ṣaṣṭhame śukladehikām || 23 ||</l>
</lg>

<lg xml:id="HT1.8.24">
	<l>ṣaḍaṅgaṃ bhāvayed yogī viramāntaṃ punas tathā |</l>
	<l>kramam utpattikaṃ caiva utpannakramam eva ca || 24 ||</l>
</lg>

<lg xml:id="HT1.8.25">
	<l><pb n="87"/>kramadvayaṃ samāśritya vajriṇā dharmadeśanā |</l>
	<l>utpattibhāgaṃ kathitam utpannaṃ kathayāmy aham || 25 ||</l>
</lg>

<lg xml:id="HT1.8.26">
	<l>khadhātāv iti padmeṣu jñānaṃ bhagam iti smṛtam |</l>
	<l>bhāvaneti samāpattis tatsukhaṃ cakram ucyate || 26 ||</l>
</lg>

<lg xml:id="HT1.8.27">
	<l><pb n="88"/>yathānyāyaṃ svasaṃvedyaṃ bodhicittaṃ tu devatā |</l>
	<l>yathodayaṃ bhavec chukraṃ dvaividyaṃ sahajaṃ tataḥ || 27 ||</l>
</lg>

<lg xml:id="HT1.8.28">
	<l>yoṣit tāvad bhavet prajñā upāyaḥ puruṣaḥ smṛtaḥ |</l>
	<l>paścād anayor dvaividhyaṃ vivṛtisaṃvṛtibhedataḥ || 28 ||</l>
</lg>

<lg xml:id="HT1.8.29">
	<l>puṃsi tāvad dhi dvaividhyaṃ śukraṃ tasya sukhañ ca vā |</l>
	<l>prajñāyāṃ ca yathā puṃsi śukraṃ tasya sukhañ ca vā || 29 ||</l>
</lg>

<lg xml:id="HT1.8.30">
	<l><pb n="89"/>atraivāpi hy ānandānāṃ catasṛṇāṃ prabhedanam |</l>
	<l>sahajaṃ caturvidhaṃ yasmād utpannakramapakṣataḥ || 30 ||</l>
</lg>

<lg xml:id="HT1.8.31">
	<l>ānandaṃ prathamaṃ vīraṃ paramānandaṃ yoginī |</l>
	<l>suratānandaṃ samastaṃ tatsukhopāyaḥ sarvavit || 31 ||</l>
</lg>

<lg xml:id="HT1.8.32">
	<l>ānandena sukhaṃ kiñcit paramānandaṃ tato 'dhikam |</l>
	<l>viramena virāgaḥ syāt sahajānandaṃ śeṣataḥ || 32 ||</l>
</lg>

<lg xml:id="HT1.8.33">
	<l><pb n="90"/>prathamaṃ sparśākāṅkṣayā dvitīyaṃ sukhavāñchayā |</l>
	<l>tṛtīyaṃ rāganāśatvāc caturthaṃ tena bhāvyate || 33 ||</l>
</lg>

<lg xml:id="HT1.8.34">
	<l>paramānandaṃ bhavaṃ proktaṃ nirvāṇaṃ ca virāgataḥ |</l>
	<l>madhyamānandamātran tu sahajam ebhir vivarjitam || 34 ||</l>
</lg>

<lg xml:id="HT1.8.35">
	<l><pb n="91"/>na rāgo na virāgaś ca madhyamaṃ nopalabhyate |</l>
	<l>nātra prajñā na copāyaḥ samyaktattvāvabodhataḥ || 35 ||</l>
</lg>

<lg xml:id="HT1.8.36">
	<l>nānyena kathyate sahajaṃ na kasminn api labhyate |</l>
	<l>ātmanā jñāyate puṇyād guruparvopasevayā || 36 ||</l>
</lg>

<lg xml:id="HT1.8.37">
	<l><pb n="93"/>hīnamadhyamotkṛṣṭāny evānyāni yāni tāni ca |</l>
	<l>sarvāṇy etāni samānīti draṣṭavyaṃ tattvabhāvanaiḥ || 37 ||</l>
</lg>

<lg xml:id="HT1.8.38">
	<l>hīnaṃ sūkṣmapadārthaṃ tu utkṛṣṭaṃ bhāvam ucyate |</l>
	<l>madhyamaṃ varjitaṃ dvābhyām anyānīti ṣaḍ indriyam || 38 ||</l>
</lg>

<lg xml:id="HT1.8.39">
	<l>sthiracalaṃ yāni tāni sarvāṇy etānīty evāham |</l>
	<l>samāni tulyaceṣṭāni samarasais tattvabhāvanaiḥ || 39 ||</l>
</lg>

<lg xml:id="HT1.8.40">
	<l><pb n="94"/>samaṃ tulyam iti proktaṃ tasya cakro rasaḥ smṛtaḥ |</l>
	<l>samarasaṃ tv ekabhāvam etenārthena bhaṇyate || 40 ||</l>
</lg>

<lg xml:id="HT1.8.41">
	<l>madbhavaṃ hi jagat sarvaṃ madbhavaṃ bhuvanatrayam |</l>
	<l>madvyāpitam idaṃ sarvaṃ nānyamayaṃ dṛṣṭaṃ jagat || 41 ||</l>
</lg>

<lg xml:id="HT1.8.42">
	<l><pb n="95"/>evaṃ matvā tu vai yogī yo 'bhyāse susamāhitaḥ |</l>
	<l>sa sidhyati na sandeho mandapuṇyo 'pi mānavaḥ || 42 ||</l>
</lg>

<lg xml:id="HT1.8.43">
	<l>khānapāne yathā snāne jāgrat supto 'pi cintayet |</l>
	<l>sātatyaṃ tu tato yāti mahāmudrābhikāṅkṣakaḥ || 43 ||</l>
</lg>

<lg xml:id="HT1.8.44">
	<l>bhāvyate hi jagat sarvaṃ manasā yasmān na bhāvyate |</l>
	<l>sarvadharmaparijñānaṃ bhāvanā naiva bhāvanā || 44 ||</l>
</lg>

<lg xml:id="HT1.8.45">
	<l><pb n="96"/>sthiracalāś ca ye bhāvās tṛṇagulmalatādayaḥ |</l>
	<l>bhāvyante vai paraṃ tattvam ātmabhāvasvarūpakam || 45 ||</l>
</lg>

<lg xml:id="HT1.8.46">
	<l>teṣām ekaṃ paraṃ nāsti svasaṃvedyaṃ mahat sukham |</l>
	<l>svasaṃvedyā bhavet siddhiḥ svasaṃvedyā hi bhāvanā || 46 ||</l>
</lg>

<lg xml:id="HT1.8.47">
	<l><pb n="97"/>svasaṃvedyamayaṃ karma bādhanāt karma jāyate |</l>
	<l>svayaṃ hartā svayaṃ kartā svayaṃ rājā svayaṃ prabhuḥ || 47 ||</l>
</lg>

<lg xml:id="HT1.8.48">
	<l><pb n="98"/>rāgo dveṣas tathaiverṣyā moho mānas tathaiva ca |</l>
	<l>sarve te tatpade ramye kalāṃ nārghanti ṣoḍaśīm || 48 ||</l>
</lg>

<lg xml:id="HT1.8.49">
	<l><pb n="99"/>dharmodayodbhavaṃ jñānaṃ khasamaṃ sopāyānvitam |</l>
	<l>trailokyaṃ tatra jātaṃ hi prajñopāyasvabhāvataḥ || 49 ||</l>
</lg>

<lg xml:id="HT1.8.50">
	<l><pb n="100"/>śukrākāro bhaved bhagavān tatsukhaṃ kāminī smṛtam |</l>
	<l>ekānekaviyogo 'sau kṣaṇād ekā parā ratiḥ || 50 ||</l>
</lg>

<lg xml:id="HT1.8.51">
	<l><pb n="101"/>svasaṃvedyam idaṃ jñānaṃ vākpathātītagocaram |</l>
	<l>adhiṣṭhānakramo hy eṣaḥ sarvajñajñānatanmayaḥ || 51 ||</l>
</lg>

<lg xml:id="HT1.8.52">
	<l>pṛthivy āpaś ca vāyuś ca teja ākāśam eva ca |</l>
	<l>kṣaṇāt sarve na bādhante svaparasaṃvidvedanam || 52 ||</l>
</lg>

<lg xml:id="HT1.8.53">
	<l>svargamartyaiś ca pātālair ekamūrtir bhavet kṣaṇāt |</l>
	<l>svaparabhāgavikalpena bādhituṃ naiva śakyate || 53 ||</l>
</lg>

<lg xml:id="HT1.8.54">
	<l><pb n="102"/>samastavedasiddhāntaiḥ karmaprasaraṇādibhiḥ |</l>
	<l>siddhir na syād bhavec chuddhyā punarjanma bhavāntare || 54 ||</l>
</lg>

<lg xml:id="HT1.8.55">
	<l>na ca tena vinā siddhir iha loke paratra ca |</l>
	<l>na jñātaṃ yena hevajraṃ vyarthas tasya pariśramaḥ || 55 ||</l>
</lg>

<lg xml:id="HT1.8.56">
	<l><pb n="103"/>nadīsrotaḥpravāhena dīpajyotiḥprabandhavat |</l>
	<l>satataṃ tattvayogena sthātavyam ahorātrataḥ || 56 ||</l>
</lg>

<p xml:id="HT1.8.col">
	yoginīcakro nāma mahāyoginīnāṃ melāpakapaṭalo 'ṣṭamaḥ ||
</p>
