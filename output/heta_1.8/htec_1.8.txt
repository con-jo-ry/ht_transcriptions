<div n="8" type="paṭala">
<p xml:id="HT1.8.1">
	atha yoginīcakraṃ vyākhyāsyāmaḥ —
	<lg>
		<l>khadhātau bhagaṃ dhyātvā madhye kurvīta bhāvanām |</l>
		<l>cakraṃ pūrvaṃ yathānyāyaṃ devatānāṃ yathodayam || 1 ||</l>
	</lg>
</p>

<lg xml:id="HT1.8.2">
	<l>cakraṃ kṣoṇījalaṃ pūrvaṃ yathānyāyaṃ hutāśanaḥ |</l>
	<l>devatānāṃ mahāvāyur bhāvakaś ca yathodayam || 2 ||</l>
</lg>

<lg xml:id="HT1.8.3">
	<l>dharmodayodbhavaṃ cakraṃ dvipuṭaṃ śuddhaṃ nirāmayam |</l>
	<l>kiñjalkena bhaved ekaṃ trikoṇenāparaṃ smṛtam || 3 ||</l>
</lg>

<lg xml:id="HT1.8.4">
	<l>tanmadhye cintayen mṛtakaṃ pañcadaśāsanātmakam |</l>
	<l>tasyopari bhavec candraś candrasyopari bījakam || 4 ||</l>
</lg>

<lg xml:id="HT1.8.5">
	<l>paścān mārtaṇḍākrāntaṃ dvayor melā mahat sukham |</l>
	<l>sthitāliś candrarūpeṇa kālirūpeṇa bhāskaraḥ || 5 ||</l>
</lg>

<lg xml:id="HT1.8.6">
	<l>candrasūryadvayor melā gauryādyāḥ prakīrtitāḥ |</l>
	<l>ādarśajñānavāṃś candraḥ samatāvān saptasaptikaḥ || 6 ||</l>
</lg>

<lg xml:id="HT1.8.7">
	<l>bījaiś cihnaṃ svadevasya pratyavekṣaṇam ucyate |</l>
	<l>sarvair aikyam anusṭhānaṃ niṣpattiḥ śuddhadharmatā || 7 ||</l>
</lg>

<lg xml:id="HT1.8.8">
	<l>ākārān bhāvayet pañca vidhānaiḥ kathitair budhaḥ |</l>
	<l>ālikālisamāyogo vajrasattvasya viṣṭaraḥ || 8 ||</l>
</lg>

<lg xml:id="HT1.8.9">
	<l>akṣarodbhava piṇḍasya hūṁphaṭkāro na ceṣyete |</l>
	<l>sattvabimbasamudbhūtaṃ maṇḍaleśaṃ vibhāvayet || 9 ||</l>
</lg>

<lg xml:id="HT1.8.10">
	<l>pūrvavad vaktracihnādyaiś candrakāntimaṇiprabham |</l>
	<l>evaṃ sarvaiva niṣpannāḥ prajñopāyasvabhāvataḥ || 10 ||</l>
</lg>

<lg xml:id="HT1.8.11">
	<l>prajñāli kālyupāyeti candrārkasya prabhedanāt |</l>
	<l>gauryādyā bhavanty asmād varṇabhedaḥ paraṃ pṛthak || 11 ||</l>
</lg>

<lg xml:id="HT1.8.12">
	<l>adhyātmapuṭe tāvat sthitā vai pañca yoginīḥ |</l>
	<l>pañcaskandhasvabhāvena bhāvayed yogavit sadā || 12 ||</l>
</lg>

<lg xml:id="HT1.8.13">
	<l>indre vajrā yame gaurī vāruṇyāṃ vāriyoginī |</l>
	<l>kauveryāṃ vajraḍākī ca madhye nairātmyayoginī || 13 ||</l>
</lg>

<lg xml:id="HT1.8.14">
	<l>bāhyapuṭe — gaurī caurī ca vettālī ghasmarī pukkasī tathā |</l>
	<l>śabarī caṇḍālinī caiva aṣṭamī ḍombinī matā || 14 ||</l>
</lg>

<lg xml:id="HT1.8.15">
	<l>adhavatī ūrdhvavatī caiva khecarī bhūcarī tathā |</l>
	<l>bhavanirvāṇasvabhāvena sthitāv etau mahākṛpa || 15 ||</l>
</lg>

<lg xml:id="HT1.8.16">
	<l>sarvā devatyaḥ—</l>
	<l>kṛṣṇavarṇā mahāraudrāḥ pañcamudrāvibhūṣitāḥ |</l>
	<l>ekavaktrā raktākṣāḥ kartikapālavṛtau karau || 16 ||</l>
</lg>

<lg xml:id="HT1.8.17">
	<l>cakrī kuṇḍala kaṇṭhā ca haste rūcaka mekhalam |</l>
	<l>pañcabuddhaviśuddhena pañcaite śuddha mudrakāḥ || 17 ||</l>
</lg>

<lg xml:id="HT1.8.18">
	<l>sarvā etādṛśāḥ khyātā yathā nairātmyayoginī |</l>
	<l>kapālaikakaravyagrā dakṣiṇe kartidhārikāḥ || 18 ||</l>
</lg>

<lg xml:id="HT1.8.19">
	<l>khaṭvāṅgaṃ caiva vāmena vyāghracarmāvṛtā kaṭiḥ |</l>
	<l>śavārūḍhā jvaladdīptā dvibhujāḥ piṅgordhvamūrdhajāḥ || 19 ||</l>
</lg>

<lg xml:id="HT1.8.20">
	<l>tathā mānādiṣaḍdoṣān kartituṃ kartikā sthitā |</l>
	<l>bhāvābhāvavikalpasya śirasā padmabhājanam || 20 ||</l>
</lg>

<lg xml:id="HT1.8.21">
	<l>raktaṃ caturmārāṇāṃ pīyate siddhihetunā |</l>
	<l>kaṭvāṅga śūnyatākāraiḥ śavopāyena kalpitam || 21 ||</l>
</lg>

<lg xml:id="HT1.8.22">
	<l>etena bhāvayec cakraṃ laghu siddhir avāpyate |</l>
	<l>prathamaṃ bhāvayet kṛṣṇāṃ dvitīye raktāṃ vibhāvayet || 22 ||</l>
</lg>

<lg xml:id="HT1.8.23">
	<l>tṛtīye bhāvayet pītāṃ caturthe haritakāṃ tathā |</l>
	<l>pañcame nīlavarṇāṃ ca ṣaṣṭhame śukladehikām || 23 ||</l>
</lg>

<lg xml:id="HT1.8.24">
	<l>ṣaḍ aṅgaṃ bhāvayed yogī viramāntaṃ punas tathā |</l>
	<l>krama utpattikaṃ caiva utpannakramam eva ca || 24 ||</l>
</lg>

<lg xml:id="HT1.8.25">
	<l>kramadvayaṃ samāśritya vajriṇāṃ dharmadeśanā || 25 ||</l>
	<l>utpattibhāgaḥ kathitaḥ ||</l>
</lg>

<lg xml:id="HT1.8.26">
	<l>khadhātāv iti padmeṣu jñānaṃ bhagam iti smṛtam |</l>
	<l>bhāvaneti samāpattis tatsukhaṃ cakram ucyate || 26 ||</l>
</lg>

<lg xml:id="HT1.8.27">
	<l>yathānyāyaṃ svasaṃvedyaṃ bodhicittaṃ tu devatā |</l>
	<l>yathodayaṃ bhavec chukraṃ dvaividhyaṃ sahajaṃ tataḥ || 27 ||</l>
</lg>

<lg xml:id="HT1.8.28">
	<l>yoṣit tāvad bhavet prajñā upāyaḥ puruṣaḥ smṛtaḥ |</l>
	<l>paścād anayor dvaividhyaṃ vivṛtisaṃvṛtibhedataḥ || 28 ||</l>
</lg>

<lg xml:id="HT1.8.29">
	<l>puṃsi tāvad dvaividhyaṃ śukraṃ tasya sukhaṃ ca vā |</l>
	<l>prajñāyām api yathā puṃsi śukraṃ tasya sukhaṃ ca vā || 29 ||</l>
</lg>

<lg xml:id="HT1.8.30">
	<l>ata eva hy ānandānāṃ catasṛṇāṃ prabhedanam |</l>
	<l>sahajaṃ caturvidhaṃ yasmād utpannakramapakṣataḥ || 30 ||</l>
</lg>

<lg xml:id="HT1.8.31">
	<l>ānandaṃ prathamaṃ vīraṃ paramānandaṃ tu yoginī |</l>
	<l>suratānandaṃ samastaṃ vai tatsukhopāya sarvavit || 31 ||</l>
</lg>

<lg xml:id="HT1.8.32">
	<l>ānandena sukhaṃ kiñcit paramānandaṃ tato 'dhikam |</l>
	<l>viramānandair virāgaḥ syāt sahajānandaṃ tu śeṣataḥ || 32 ||</l>
</lg>

<lg xml:id="HT1.8.33">
	<l>prathamaṃ sparśākāṅkṣaṃ dvitīyaṃ sukhavāñchayā |</l>
	<l>tṛtīyaṃ rāganāśatvāc caturthaṃ tena bhāvyate || 33 ||</l>
</lg>

<lg xml:id="HT1.8.34">
	<l>paramānandaṃ bhavaṃ proktaṃ nirvāṇaṃ ca virāgataḥ |</l>
	<l>madhyamānandamātraṃ tu sahajam ebhir vivarjitam || 34 ||</l>
</lg>

<lg xml:id="HT1.8.35">
	<l>na rāgo na virāgaś ca madhyamaṃ nopalabhyate |</l>
	<l>nātra prajñā na copāyaḥ samyaktattvāvabodhataḥ || 35 ||</l>
</lg>

<lg xml:id="HT1.8.36">
	<l>nānyena kathyate sahajaṃ na kasminn api labhyate |</l>
	<l>ātmanā jñāyate puṇyād guruparvopasevayā || 36 ||</l>
</lg>

<lg xml:id="HT1.8.37">
	<l>hīnamadhyotkṛṣṭāni anyāni yāni tāni ca |</l>
	<l>sarve tāni samānīti draṣṭavyaṃ tattvabhāvanaiḥ || 37 ||</l>
</lg>

<lg xml:id="HT1.8.38">
	<l>hīnaṃ sūkṣmapadārthaṃ tu utkṛṣṭaṃ bhāvam ucyate |</l>
	<l>madhyamaṃ varjitaṃ dvābhyām anyānīti ṣaḍ indriyam || 38 ||</l>
</lg>

<lg xml:id="HT1.8.39">
	<l>sthiracalaṃ yāni tānīti sarvaitānīti naivāham |</l>
	<l>samāni tulyaceṣṭāni samarasais tattvabhāvanaiḥ || 39 ||</l>
</lg>

<lg xml:id="HT1.8.40">
	<l>samaṃ tulyam ity uktaṃ tasya cakro rasaḥ smṛtaḥ |</l>
	<l>samarasaṃ tv ekabhāvaṃ tu anenārthena bhaṇyate || 40 ||</l>
</lg>

<lg xml:id="HT1.8.41">
	<l>madbhavaṃ hi jagat sarvaṃ madbhavaṃ bhuvanatrayam |</l>
	<l>mayā vyāpitam idaṃ sarvaṃ nānyamayaṃ dṛṣṭaṃ jagat || 41 ||</l>
</lg>

<lg xml:id="HT1.8.42">
	<l>evaṃ matvā tu vai yogī yo 'bhyaset susamāhitaḥ |</l>
	<l>sa sidhyati na sandeho mandapuṇyo 'pi mānavaḥ || 42 ||</l>
</lg>

<lg xml:id="HT1.8.43">
	<l>khāne pāne tathā snāne jāgrat supto 'pi cintayet |</l>
	<l>sātatyaṃ tu tato yāti mahāmudrābhikāṅkṣakaḥ || 43 ||</l>
</lg>

<lg xml:id="HT1.8.44">
	<l>bhāvyate hi jagat sarve manasā yasmān na bhāvyate |</l>
	<l>sarvadharmaparijñānaṃ bhāvanā naiva bhāvanā || 44 ||</l>
</lg>

<lg xml:id="HT1.8.45">
	<l>sthiracalāś ca ye bhāvās tṛṇagulmalatādayaḥ |</l>
	<l>bhāvyante vai paraṃ tattvam ātmabhāvasvarūpakam || 45 ||</l>
</lg>

<lg xml:id="HT1.8.46">
	<l>teṣām ekaṃ paraṃ nāsti svasaṃvedyaṃ mahat sukham |</l>
	<l>svasaṃvedyād bhavet siddhiḥ svasaṃvedyaṃ hi bhāvanā || 46 ||</l>
</lg>

<lg xml:id="HT1.8.47">
	<l>svasaṃvedyamayaṃ karma bādhanāt karma jāyate |</l>
	<l>svayaṃ hartā svayaṃ kartā svayaṃ rājā svayaṃ prabhuḥ || 47 ||</l>
</lg>

<lg xml:id="HT1.8.48">
	<l>rāgaṃ dveṣaṃ tathā īrṣyāṃ mohaṃ mānaṃ tathaiva ca |</l>
	<l>sarve te tatpade ramye kalāṃ nārghanti ṣoḍaśīm || 48 ||</l>
</lg>

<lg xml:id="HT1.8.49">
	<l>dharmodayodbhavaṃ jñānaṃ khasamaṃ sopāyānvitam |</l>
	<l>trailokyas tatra jāto hi prajñopāyasvabhāvataḥ || 49 ||</l>
</lg>

<lg xml:id="HT1.8.50">
	<l>śukrākāro bhaved bhagavāṃs tatsukhaṃ kāminī smṛtam |</l>
	<l>ekānekaviyogo 'sau kṣaṇād ekāt parā ratiḥ || 50 ||</l>
</lg>

<lg xml:id="HT1.8.51">
	<l>svasaṃvedyam idaṃ jñānaṃ vākpathātītagocaram |</l>
	<l>adhiṣṭhānakramo hy eṣa sarvajñajñāna tanmayaḥ || 51 ||</l>
</lg>

<lg xml:id="HT1.8.52">
	<l>pṛthivī āpa vāyuś ca teja ākāśa eva ca |</l>
	<l>kṣaṇāt sarve na bādhyante svaparasaṃvittivedanam || 52 ||</l>
</lg>

<lg xml:id="HT1.8.53">
	<l>svargamartyaiś ca pātālair ekamūrtir bhavet kṣaṇāt |</l>
	<l>svaparabhāgavikalpena bādhituṃ naiva śakyate || 53 ||</l>
</lg>

<lg xml:id="HT1.8.54">
	<l>samastavedasiddhāntaiḥ karmaprasarādibhis tathā |</l>
	<l>siddhir na syād bhavec chuddhā punarjanmabhavāntare || 54 ||</l>
</lg>

<lg xml:id="HT1.8.55">
	<l>na ca tena vinā siddhir iha loke paratra ca |</l>
	<l>na jñātaṃ yena hevajraṃ vṛthā tasya pariśramam || 55 ||</l>
</lg>

<lg xml:id="HT1.8.56">
	<l>nadīsrotaḥpravāhena dīpajyotiḥprabandhavat |</l>
	<l>satataṃ tattvayogena sthātavyam ahorātrataḥ || 56 ||</l>
</lg>

<p xml:id="HT1.8.col">
	yoginīcakro nāma mahāyoginīnāṃ melāpakapaṭalo 'ṣṭamaḥ ||  ||
</p>
