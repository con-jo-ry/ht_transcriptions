<div n="2" type="paṭala">
<lg xml:id="HT2.2.1">
	<l>vajragarbha āha—</l>
	<l>gaganavat sarvadharmeṣu sāgare tumbikā yathā |</l>
	<l>sattvāḥ kathaṃ sidhyante sveṣṭadevatarūpataḥ || 1 ||</l>
</lg>

<lg xml:id="HT2.2.2">
	<l>bhagavān āha—</l>
	<l>nairātmyāyogayuktena athavā śrīherukodyataḥ |</l>
	<l>kṣaṇam apy anyacittaḥ san na tiṣṭhet siddhikāṅkṣakaḥ || 2 ||</l>
</lg>

<lg xml:id="HT2.2.3">
	<l>prathamābhyāsakālasya sthānaṃ vai kalpitaṃ śubham |</l>
	<l>yatrasthaḥ sidhyate mantrī ekacittaḥ samāhitaḥ || 3 ||</l>
</lg>

<lg xml:id="HT2.2.4">
	<l>svagṛheṣu niśākāle siddho 'ham iti cetasā |</l>
	<l>bhāvayed yoginīṃ prājño 'thavā śrīherukākṛtim || 4 ||</l>
</lg>

<lg xml:id="HT2.2.5">
	<l>aṅghriṃ prakṣālayan bhuñjann ācaman pūga bhakṣayan |</l>
	<l>candanair hastaṃ mardayan kaupinaiś chādayan kaṭim || 5 ||</l>
</lg>

<lg xml:id="HT2.2.6">
	<l>niḥsaran bhāṣayan bhāṣāṃ gacchaṃs tiṣṭhan ruṣan hasan |</l>
	<l>bhagavatīṃ sevayan prājño yoginīṃ bhāvayed vratī || 6 ||</l>
</lg>

<lg xml:id="HT2.2.7">
	<l>kṣaṇam apy anyarūpeṇa avidyāduṣṭacetasā |</l>
	<l>na sthātavyaṃ budhair yatnāt siddhyarthaṃ siddhikāṅkṣibhiḥ || 7 ||</l>
</lg>

<lg xml:id="HT2.2.8">
	<l>vajragarbha mayākhyātaṃ dhyānaṃ kilbiṣanāśanam |</l>
	<l>siddhyarthaṃ kautukenāpi pakṣam ekaṃ parīkṣatām || 8 ||</l>
</lg>

<lg xml:id="HT2.2.9">
	<l>sarvacintāṃ parityajya devatāmūrticetasā |</l>
	<l>dinam ekam avicchinnaṃ bhāvayitvā parīkṣatha || 9 ||</l>
</lg>

<lg xml:id="HT2.2.10">
	<l>nānyopāyo 'sti saṃsāre svaparārthaprasiddhaye |</l>
	<l>sakṛd abhyāsitā vidyā sadyaḥ pratyayakāriṇī || 10 ||</l>
</lg>

<lg xml:id="HT2.2.11">
	<l>bhayonmādais tathā duḥkhaiḥ śokapīḍādyupadravaiḥ |</l>
	<l>rāgadveṣamahāmohaiḥ sādhako naiva kliśyate || 11 ||</l>
</lg>

<lg xml:id="HT2.2.12">
	<l>evaṃ vimṛṣyamāṇā vai hitāhitaphalodayam |</l>
	<l>kathaṃ te kṣaṇam apy ekaṃ yoginaḥ santi raurave || 12 ||</l>
</lg>

<lg xml:id="HT2.2.13">
	<l>pañcānantaryakārī ca prāṇivadhe ratāś ca ye |</l>
	<l>api tu ye janmahīnā ye mūrkhāḥ krūrakarmiṇaḥ || 13 ||</l>
</lg>

<lg xml:id="HT2.2.14">
	<l>kurūpā vikalagātrāś ca sidhyante te 'pi cintayā |</l>
	<l>daśakuśalābhyāsī ca gurubhakto jitendriyaḥ || 14 ||</l>
</lg>

<lg xml:id="HT2.2.15">
	<l>mānakrodhavinirmuktaḥ sa tāvat sidhyate dhruvam |</l>
	<l>satatābhyāsayogena siddhilabdhaḥ samāhitaḥ || 15 ||</l>
</lg>

<lg xml:id="HT2.2.16">
	<l>māsam ekaṃ cared guptaṃ yāvan mudrā na labhyate |</l>
	<l>ādeśaṃ labhate mantrī yoginībhir ādiśyate || 16 ||</l>
</lg>

<lg xml:id="HT2.2.17">
	<l>gṛhītvā amukīṃ mudrāṃ sattvārthaṃ kuru vajradhṛk |</l>
	<l>tāṃ ca prāpya viśālākṣīṃ rūpayauvanamaṇḍitām || 17 ||</l>
</lg>

<lg xml:id="HT2.2.18">
	<l>sihlakarpūrasaṃyuktāṃ bodhicittena saṃskaret |</l>
	<l>daśakuśalād ārabhya tasyāṃ dharmaṃ prakāśayet || 18 ||</l>
</lg>

<lg xml:id="HT2.2.19">
	<l>devatārūpacittaṃ ca samayaṃ caikacittatām |</l>
	<l>māsam ekena bhavyā sā bhaven naivātra saṃśayaḥ || 19 ||</l>
</lg>

<lg xml:id="HT2.2.20">
	<l>varalabdhā yato nārī sarvasaṃkalpavarjitā |</l>
	<l>athavā cātmanaḥ śaktyā kṛṣṭvā mudrāṃ prakalpayet || 20 ||</l>
</lg>

<lg xml:id="HT2.2.21">
	<l>devāsuramanuṣyebhyo yakṣebhyaḥ kinnarād api |</l>
	<l>tāṃ ca gṛhya carec caryām ātmano dhairyapratyayāt || 21 ||</l>
</lg>

<lg xml:id="HT2.2.22">
	<l>na caryā bhogataḥ proktā yā khyātā bhīmarūpiṇī |</l>
	<l>svacittapratyavekṣāya sthiraṃ kiṃ vā calaṃ manaḥ || 22 ||</l>
</lg>

<lg xml:id="HT2.2.23">
	<l>vajragarbha āha—</l>
	<l>nairātmyāyogayuktena mudrātvaṃ viśiṣyate katham |</l>
	<l>mudrayā mudrayā dvābhyāṃ mudrāsiddhiḥ kathaṃ bhavet || 23 ||</l>
</lg>

<lg xml:id="HT2.2.24">
	<l>bhagavān āha—</l>
	<l>strīrūpaṃ vihāyānyad rūpaṃ kuryād bhagavataḥ |</l>
	<l>stanaṃ hṛtvā bhaved bolaṃ kakkolamadhyasaṃsthitam ||</l>
</lg>

<lg xml:id="HT2.2.25">
	<l>tīradvayaṃ bhaved ghaṇṭā kiñjalkaṃ bolakaṃ bhavet |</l>
	<l>śeṣarūpaṃ mahātmāno herukasya mahārateḥ || 25 ||</l>
</lg>

<lg xml:id="HT2.2.26">
	<l>herukayogasya puṃsaḥ puṃstvam āyāty ayatnataḥ |</l>
	<l>mudrāsiddhir bhavet tasmād vyaktaśaktasya yoginaḥ || 26 ||</l>
</lg>

<lg xml:id="HT2.2.27">
	<l>utpattipralayābhyāṃ ca prajñopāyaṃ na bādhyate |</l>
	<l>upāyaḥ sambhavo yasmāl layaṃ prajñā bhavāntakī || 27 ||</l>
</lg>

<lg xml:id="HT2.2.28">
	<l>tena pralayaṃ nāsyāsti utpādo naiva tattvataḥ |</l>
	<l>pralayāl līyate kaścil layābhāvān na ca kṣayaḥ || 28 ||</l>
</lg>

<lg xml:id="HT2.2.29">
	<l>utpattikramayogena prapañcaṃ bhāvayed vratī |</l>
	<l>prapañcaṃ svapnavat kṛtvā prapañcair niṣprapañcayet || 29 ||</l>
</lg>

<lg xml:id="HT2.2.30">
	<l>yathā māyā yathā svapnaṃ yathā syād antarābhavam |</l>
	<l>tathaiva maṇḍalaṃ bhāti sātatyābhyāsayogataḥ || 30 ||</l>
</lg>

<lg xml:id="HT2.2.31">
	<l>mahāmudrābhiṣekeṣu yathā jñātaṃ mahat sukham |</l>
	<l>tasyaivaitat prabhāvaḥ syān maṇḍalaṃ nānyasambhavam || 31 ||</l>
</lg>

<lg xml:id="HT2.2.32">
	<l>sukhaṃ kṛṣṇaṃ sukhaṃ pītaṃ sukhaṃ raktaṃ sukhaṃ sitam |</l>
	<l>sukhaṃ śyāmaṃ sukhaṃ nīlaṃ sukhaṃ kṛtsnaṃ carācaram || 32 ||</l>
</lg>

<lg xml:id="HT2.2.33">
	<l>sukhaṃ prajñā sukhopāyaḥ sukhaṃ kundurujaṃ tathā |</l>
	<l>sukhaṃ bhāvaḥ sukhābhāvo vajrasattvaḥ sukhaḥ smṛtaḥ || 33 ||</l>
</lg>

<lg xml:id="HT2.2.34">
	<l>vajragarbha āha—</l>
	<l>utpannakramayogo 'yaṃ satsukhaṃ mahāsukhaṃ matam |</l>
	<l>utpanno bhāvanāhīno utpattyā kiṃ prayojanam || 34 ||</l>
</lg>

<lg xml:id="HT2.2.35">
	<l>bhagavān āha—</l>
	<l>aho śraddhāvegena naṣṭo mahābodhisattva iti |</l>
	<l>dehābhāve kutaḥ saukhyaṃ saukhyaṃ vaktuṃ na śakyate |</l>
	<l>vyāpyavyāpakarūpeṇa sukhena vyāpitaṃ jagat || 35 ||</l>
</lg>

<lg xml:id="HT2.2.36">
	<l>yathā puṣpāśritaṃ gandhaṃ puṣpābhāvān na gamyate |</l>
	<l>tathā rūpādyabhāvena saukhyaṃ naivopalabhyate || 36 ||</l>
</lg>

<lg xml:id="HT2.2.37">
	<l>bhāvo 'haṃ naiva bhāvo 'haṃ buddho 'haṃ vastubodhanāt |</l>
	<l>māṃ na jānanti ye mūḍhāḥ kausīdyopahatāś ca ye || 37 ||</l>
</lg>

<lg xml:id="HT2.2.38">
	<l>vihare 'haṃ sukhāvatyāṃ sadvajrayoṣidbhage |</l>
	<l>ekārākṛtirūpe tu buddharatnakaraṇḍake || 38 ||</l>
</lg>

<lg xml:id="HT2.2.39">
	<l>vyākhyātāham ahaṃ dharmaḥ śrotāhaṃ svagaṇair yutaḥ |</l>
	<l>sādhyo 'haṃ jagataḥ śāstā loko 'haṃ laukiko 'py aham || 39 ||</l>
</lg>

<lg xml:id="HT2.2.40">
	<l>sahajānandasvabhāvo 'haṃ paramāntaviramādikam |</l>
	<l>tathā ca pratyayaṃ putra andhakāre pradīpavat || 40 ||</l>
</lg>

<lg xml:id="HT2.2.41">
	<l>dvātriṃśallakṣaṇī śāstā aśītyanuvyañjanī prabhuḥ |</l>
	<l>yoṣidbhage sukhāvatyāṃ śukranāmnā vyavasthitaḥ || 41 ||</l>
</lg>

<lg xml:id="HT2.2.42">
	<l>vinā tena na saukhyaṃ syāt saukhyaṃ hitvā bhaven na saḥ |</l>
	<l>sāpekṣam asamarthatvād devatāyogataḥ sukham || 42 ||</l>
</lg>

<lg xml:id="HT2.2.43">
	<l>tasmād buddho na bhāvaḥ syād abhāvarūpo 'pi naiva saḥ |</l>
	<l>bhujamukhākārarūpī ca arūpī paramasaukhyataḥ || 43 ||</l>
</lg>

<lg xml:id="HT2.2.44">
	<l>tasmāt sahajaṃ jagat sarvaṃ sahajaṃ svarūpam ucyate |</l>
	<l>svarūpam eva nirvāṇaṃ viśuddhyākāracetasā || 44 ||</l>
</lg>

<lg xml:id="HT2.2.45">
	<l>devatārūpayogaṃ tu jātamātre vyavasthitam |</l>
	<l>bhujamukhavarṇasaṃsthānāt kiṃtu prākṛtavāsanā || 45 ||</l>
</lg>

<lg xml:id="HT2.2.46">
	<l>yenaiva viṣakhaṇḍena mriyante sarvajantavaḥ |</l>
	<l>tenaiva viṣatattvajño viṣeṇa sphoṭayed viṣam || 46 ||</l>
</lg>

<lg xml:id="HT2.2.47">
	<l>yathā vātagṛhītasya māṣabhakṣyaṃ pradīyate |</l>
	<l>vātena hanyate vātaṃ viparītauṣadhikalpanāt |</l>
	<l>bhavaḥ śuddho bhavenaiva vikalpaṃ pratikalpataḥ || 47 ||</l>
</lg>

<lg xml:id="HT2.2.48">
	<l>karṇe toyaṃ yathā viṣṭaṃ pratitoyena kṛṣyate |</l>
	<l>tathā bhavavikalpo 'pi ākāraiḥ śodhyate khalu || 48 ||</l>
</lg>

<lg xml:id="HT2.2.49">
	<l>yathā pāvakadagdhāś ca svidyante vahninā punaḥ |</l>
	<l>tathā rāgāgnidagdhāś ca svidyante rāgavahninā || 49 ||</l>
</lg>

<lg xml:id="HT2.2.50">
	<l>yena yena hi badhyante jantavo raudrakarmaṇā |</l>
	<l>sopāyena tu tenaiva mucyante bhavabandhanāt || 50 ||</l>
</lg>

<lg xml:id="HT2.2.51">
	<l>rāgeṇa badhyate loko rāgeṇaiva vimucyate |</l>
	<l>viparītabhāvanā hy eṣā na jñātā buddhatīrthikaiḥ || 51 ||</l>
</lg>

<lg xml:id="HT2.2.52">
	<l>kundureṣu bhavet pañca pañcabhūtasvarūpataḥ |</l>
	<l>eka eva mahānandaḥ pañcatāṃ yānti bhedanaiḥ || 52 ||</l>
</lg>

<lg xml:id="HT2.2.53">
	<l>bolakakkolayogena sparśāt kāṭhinyavāsanā |</l>
	<l>kaṭhinasya mohadharmatvān moho vairocano mataḥ || 53 ||</l>
</lg>

<lg xml:id="HT2.2.54">
	<l>bodhicittaṃ dravaṃ yasmād dravam abdhātukaṃ matam |</l>
	<l>apām akṣobhyarūpatvād dveṣo akṣobhyanāyakaḥ || 54 ||</l>
</lg>

<lg xml:id="HT2.2.55">
	<l>dvayor gharṣaṇasaṃyogāt tejo jāyate sadā |</l>
	<l>rāgo 'mitābhavajraḥ syād rāgas tejasi sambhavet || 55 ||</l>
</lg>

<lg xml:id="HT2.2.56">
	<l>kakkolake yac cittaṃ tat samīraṇarūpakam |</l>
	<l>īrṣyā amoghasiddhiḥ syād amogho vāyusambhavaḥ || 56 ||</l>
</lg>

<lg xml:id="HT2.2.57">
	<l>sukhaṃ rāgaṃ bhaved raktaṃ raktir ākāśalakṣaṇā |</l>
	<l>ākāśaṃ piśunavajraḥ syāt piśunam ākāśasambhavam || 57 ||</l>
</lg>

<lg xml:id="HT2.2.58">
	<l>ekam eva mahac cittaṃ pañcarūpeṇa lakṣitam |</l>
	<l>pañcasu kuleṣūtpannās tatrānekasahasraśaḥ || 58 ||</l>
</lg>

<lg xml:id="HT2.2.59">
	<l>tasmād ekasvabhāvo 'sau mahāsukhaḥ paramaśāśvataḥ |</l>
	<l>pañcatāṃ yāti bhedena rāgādipañcacetasā || 59 ||</l>
</lg>

<lg xml:id="HT2.2.60">
	<l>daśagaṅgānadivālukatulyā</l>
	<l>ekakuleṣu tathāgatasaṅghāḥ |</l>
	<l>saṅghakuleṣu anekakulāni</l>
	<l>teṣu kuleṣu kulāni śatāni || 60 ||</l>
</lg>

<lg xml:id="HT2.2.61">
	<l>tāni ca lakṣakulāni mahānti</l> 
	<l>koṭikuleṣu asaṃkhya bhavanti |</l>
	<l>tatra kuleṣu asaṃkhyakulāni</l>
	<l>paramānandakulodbhavāni || 61 ||</l>
</lg>

<p xml:id="HT2.2.col">
	hevajre ḍākinījālaśaṃvare siddhinirṇayanāmā dvitīyaḥ paṭalaḥ || ||
