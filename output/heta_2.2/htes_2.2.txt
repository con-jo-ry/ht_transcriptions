<lg xml:id="HT2.2.1">
	<l>vajragarbha āha |</l>
	<l>gaganavatsarvadharmeṣu sāgare tumbikā yathā ||</l>
	<l>sattvāḥ kathaṃ prasidhyanti sveṣṭadevatārūpataḥ || (1)</l>
</lg>

<lg xml:id="HT2.2.2">
	<l>bhagavān āha |</l>
	<l>nairātmyāyogayuktātmā 'thavā herukayogataḥ ||</l>
	<l>kṣaṇam apy anyacittaḥ san na tiṣṭhet siddhikāṅkṣakaḥ || (2)</l>
</lg>

<lg xml:id="HT2.2.3">
	<l>prathamābhyāsakālasya sthānaṃ vai kalpitaṃ śubhaṃ ||</l>
	<l>yatrasthaḥ siddhyate mantrī ekacittaḥ samāhitaḥ || (3)</l>
</lg>

<lg xml:id="HT2.2.4">
	<l>svagṛheṣu niśākāle siddho 'ham iti cetasā ||</l>
	<l>bhāvayed yoginīṃ prājño 'thavā śrīherukākṛtiṃ || (4)</l>
</lg>

<lg xml:id="HT2.2.5">
	<l>aṅghriṃ prakṣālayan bhuñjan ācamapūgaṃ bhakṣayan ||</l>
	<l>candanair hastaṃ mardayan kaupinaiś chādayan kaṭiṃ || (5)</l>
</lg>

<lg xml:id="HT2.2.6">
	<l>niḥsaraṃ bhāṣayan bhāṣāṃ gacchan tiṣṭhan ruṣan hasan ||</l>
	<l><pb n="46"/>bhagavatīṃ sevayet prājño yoginīṃ bhāvayed vratī || (6)</l>
</lg>

<lg xml:id="HT2.2.7">
	<l>kṣaṇam apy anyarūpeṇa avidyāduṣṭacetasā ||</l>
	<l>na sthātavyaṃ buddhair yatnāt siddhyarthaṃ siddhikāṅkṣibhiḥ || (7)</l>
</lg>

<lg xml:id="HT2.2.8">
	<l>vajragarbha mayākhyātaṃ dhyānaṃ kilbiṣanāśanaṃ ||</l>
	<l>siddhyarthaṃ kautukenāpi pakṣam ekaṃ parikṣethāḥ || (8)</l>
</lg>

<lg xml:id="HT2.2.9">
	<l>sarvacintāṃ parityajya devatāmurticetasā ||</l>
	<l>dinam ekam avicchinnaṃ bhāvayitvā parikṣethāḥ || (9)</l>
</lg>

<lg xml:id="HT2.2.10">
	<l>nānyopāyo 'sti saṃsāre svaparārthaprasiddhaye ||</l>
	<l>sakṛd abhyāsitā vidyā sadyaḥ pratyayakāriṇī || (10)</l>
</lg>

<lg xml:id="HT2.2.11">
	<l>bhayonmādais tathā duḥkhaiḥ śokapīḍādyupadravaiḥ ||</l>
	<l>rāgadveṣamahāmohaiḥ sādhako naiva kliśyate || (11)</l>
</lg>

<lg xml:id="HT2.2.12">
	<l>evaṃ vimṛṣyamāṇā vai hitāhitaphalodayaṃ ||</l>
	<l>kathaṃ te kṣaṇam apy ekaṃ yoginaḥ santi raurave || (12)</l>
</lg>

<lg xml:id="HT2.2.13">
	<l>pañcānantaryakāriṇaḥ prāṇivadharatāś ca ye ||</l>
	<l>api tu ye janmahīnā ye mūrkhāḥ krūrakarmiṇaḥ || (13)</l>
</lg>

<lg xml:id="HT2.2.14">
	<l>kurūpā vikalagātrāś ca siddhyante te 'pi cintayā ||</l>
	<l>daśakuśalābhyāsī ca gurubhakto jitendriyaḥ || (14)</l>
</lg>

<lg xml:id="HT2.2.15">
	<l>mānakrodhavinirmuktaḥ sa tāvat siddhyate dhruvaṃ ||</l>
	<l>sātatyābhyāsayogena siddhilabdhaḥ samāhitaḥ || (15)</l>
</lg>

<lg xml:id="HT2.2.16">
	<l>māsam ekaṃ cared guptaṃ yāvat mudrā na labhyate ||</l>
	<l>ādeśaṃ labhate mantrī yoginībhir ādiśyate || (16)</l>
</lg>

<lg xml:id="HT2.2.17">
	<l>gṛhītvā amukīṃ mudrāṃ sattvārthaṃ kuru vajradhṛk ||</l>
	<l>tāñ ca prāpya viśālākṣīṃ rūpayauvanamaṇḍitāṃ || (17)</l>
</lg>

<lg xml:id="HT2.2.18">
	<l>sihlakarpūrasaṃyuktāṃ bodhicittena saṃskaret ||</l>
	<l>daśakuśalād ārabhya tasyām dharmaṃ prakāśayet || (18)</l>
</lg>

<lg xml:id="HT2.2.19">
	<l>devatārūpacittañ ca samayañ caikacittatāṃ ||</l>
	<l>māsam ekena bhavyā sā bhaven naivātra saṃśayaḥ || (19)</l>
</lg>

<lg xml:id="HT2.2.20">
	<l>varalabdhā yato nārī sarvasaṃkalpavarjitā ||</l>
	<l>athavā cātmanaḥ śaktyā kṛṣṭvā mudrāṃ prakalpayet || (20)</l>
</lg>

<lg xml:id="HT2.2.21">
	<l>devāsuramanuṣyebhyo yakṣebhyaḥ kinnarād api ||</l>
	<l>tāñ ca gṛhya carec caryām ātmano dhairyapratyayāt || (21)</l>
</lg>

<lg xml:id="HT2.2.22">
	<l>na caryā bhogataḥ proktā yā khyātā bhīmarūpiṇī ||</l>
	<l>svacittapratyavekṣāya sthiraṃ kiṃ vācalaṃ manaḥ || (22)</l>
	<l>[satataṃ devatāmūrtyā sthātavyaṃ yoginā yataḥ]</l>
</lg>

<lg xml:id="HT2.2.23">
	<l>vajragarbha āha ||</l>
	<l>nairātmyāyogayuktena mudrārthaṃ viśiṣyate kathaṃ ||</l>
	<l><pb n="48"/>mudrayā mudrayā dhvābhyāṃ mudrāsiddhiḥ kathaṃ bhavet || (23)</l>
</lg>

<lg xml:id="HT2.2.24">
	<l>bhagavān āha |</l>
	<l>strīrūpaṃ vihāyānyad rūpaṃ kuryād bhagavataḥ ||</l>
	<l>stanaṃ hitvā bhaved bolaṃ kakkolamadhyasaṃsthitaṃ || (24)</l>
</lg>

<lg xml:id="HT2.2.25">
	<l>tīradvayaṃ bhavet ghaṇṭhā kiñjalkaṃ bolakaṃ bhavet ||</l>
	<l>śeṣaṃ rūpaṃ mahātmano herukasya mahārateḥ || (25)</l>
</lg>

<lg xml:id="HT2.2.26">
	<l>herukayogasya punsaḥ puṃstvam āyāty ayatnataḥ ||</l>
	<l>mudrāsiddhir bhaved yasmād vyaktaśaktasya yoginaḥ || (26)</l>
</lg>

<lg xml:id="HT2.2.27">
	<l>utpattipralayābhyāṃ ca prajñopāyo na bādhyate ||</l>
	<l>upāyaḥ saṃbhavo yasmāl layaṃ prajñā bhavāntakī || (27)</l>
</lg>

<lg xml:id="HT2.2.28">
	<l>tena pralayan nāsyāsti utpādo naiva tatvataḥ ||</l>
	<l>pralayāntīyate kaścil layābhāvān na ca kṣayaḥ || (28)</l>
</lg>

<lg xml:id="HT2.2.29">
	<l>utpattikramayogena prapañcaṃ bhāvayed vratī ||</l>
	<l>prapañcaṃ svapnavat kṛtvā prapañcair niḥprapañcayet || (29)</l>
</lg>

<lg xml:id="HT2.2.30">
	<l>yathā māyā yathā svapnaṃ yathā syād antarābhavaṃ ||</l>
	<l>tathaiva maṇḍalaṃ bhāti sātatyābhyāsayogataḥ || (30)</l>
</lg>

<lg xml:id="HT2.2.31">
	<l>mahāmudrābhiṣekeṣu yathājñātaṃ mahat sukhaṃ ||</l>
	<l>tasyaiva tatprabhāvaḥ syān maṇḍalaṃ nānyasaṃbhavaṃ || (31)</l>
</lg>

<lg xml:id="HT2.2.32">
	<l>sukhaṃ kṛṣṇaṃ sukhaṃ pītaṃ sukhaṃ raktaṃ sukhaṃ sitaṃ ||</l>
	<l>sukhaṃ śyāmaṃ sukhaṃ nīlaṃ sukhaṃ kṛtsnaṃ carācaraṃ || (32)</l>
</lg>

<lg xml:id="HT2.2.33">
	<l>sukhaṃ prajñā sukhopāyaḥ sukhaṃ kundurujaṃ tathā ||</l>
	<l>sukhaṃ bhāvaḥ sukhābhāvo vajrasattvaḥ sukhasmṛtaḥ || (33)</l>
</lg>

<lg xml:id="HT2.2.34">
	<l>vajragarbha āha ||</l>
	<l>utpannakramayogo 'yaṃ [sat]sukhaṃ mahāsukhaṃ mataṃ |  </l>
	<l>utpannabhāvanāhīno utpattyā kiṃ prayojanaṃ || (34)</l>
</lg>

<lg xml:id="HT2.2.35">
	<l>bhagavān āha | aho</l>
	<l>śraddhāvegena naṣṭo 'yaṃ mahābodhisattva iti ||</l>
	<l>dehābhāve kutaḥ saukhyaṃ saukhyaṃ vaktuṃ na śakyate ||</l>
	<l>vyāpyavyāpakarūpeṇa sukhena vyāpitaṃ jagat || (35)</l>
</lg>

<lg xml:id="HT2.2.36">
	<l>yathā puṣpāśritaṃ gandhaṃ puṣpābhāvān na gamyate ||</l>
	<l>tathā rūpādyabhāvena saukhyaṃ naivopalabhyate || (36)</l>
</lg>

<lg xml:id="HT2.2.37">
	<l>bhāvo 'haṃ naiva bhāvo 'haṃ buddho 'haṃ vastubodhanāt ||</l>
	<l>māṃ na jānanti ye mugdhāḥ kausīdyopahatāś ca ye || (37)</l>
</lg>

<lg xml:id="HT2.2.38">
	<l>vihare 'haṃ sukhāvatyāṃ sadvajrayoṣito bhage ||</l>
	<l>ekārākṛtirūpe tu buddharatnakaraṇḍake || (38)</l>
</lg>

<lg xml:id="HT2.2.39">
	<l>vyākhyātāham ahaṃ dharmaḥ śrotāhaṃ sugaṇair yutaḥ ||</l>
	<l><pb n="50"/>sādhyo 'haṃ jagataḥ śāstā loko 'haṃ laukiko 'py ahaṃ || (39)</l>
</lg>

<lg xml:id="HT2.2.40">
	<l>sahajānandasvabhāvo 'haṃ paramāntaṃ viramādikaṃ ||</l>
	<l>tathā ca pratyayaṃ putra andhakāre pradīpavat || (40)</l>
</lg>

<lg xml:id="HT2.2.41">
	<l>dvātriṃśallakṣaṇī śāstā aśītivyañjanī prabhuḥ ||</l>
	<l>yoṣidbhage sukhāvatyāṃ śukranāmna vyavasthitaḥ || (41)</l>
</lg>

<lg xml:id="HT2.2.42">
	<l>vinā tena na saukhyaṃ syāt sukhaṃ hitvā bhaven na saḥ ||</l>
	<l>sāpekṣam asamarthatvād devatāyogataḥ sukhaṃ || (42)</l>
</lg>

<lg xml:id="HT2.2.43">
	<l>tasmād buddho na bhāvaḥ syād abhāvarūpo 'pi naiva saḥ ||</l>
	<l>bhujamukhākārarūpī cārūpī paramasaukhyataḥ || (43)</l>
</lg>

<lg xml:id="HT2.2.44">
	<l>tasmāt sahajaṃ jagat sarvaṃ sahajaṃ svarūpam ucyate ||</l>
	<l>svarūpam eva nirvāṇaṃ viśuddhyākāracetasā || (44)</l>
</lg>

<lg xml:id="HT2.2.45">
	<l>devatāyogarūpaṃ tu jātamātre vyavasthitaḥ ||</l>
	<l>bhujamukhavarṇasthānāt kiṃ tu prākṛtavāsanā || (45)</l>
</lg>

<lg xml:id="HT2.2.46">
	<l>yenaiva viṣakhaṇḍena mriyante sarvajantavaḥ ||</l>
	<l>tenaiva viṣatattvajño viṣeṇa sphoṭayed viṣaṃ || (46)</l>
</lg>

<lg xml:id="HT2.2.47">

<lg xml:id="HT2.2.48">
