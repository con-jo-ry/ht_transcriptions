<div n="9" type="paṭala">
<p xml:id="HT1.9.1">
	ataḥ paraṃ viśuddhipaṭalaṃ vyākhyāsyāmaḥ |
	<lg>
		<l>sarveṣāṃ khalu vastūnāṃ viśuddhis tathatā smṛtā |</l>
		<l>paścād ekaikabhedena devatānāṃ tu kathyate || 1 ||</l>
	</lg>
</p>

<lg xml:id="HT1.9.2">
	<l>ṣaḍ indriyaṃ pañca skandhaṃ ṣaḍ āyatanaṃ pañca mahābhūtam |</l>
	<l>svabhāvena viśuddham ajñānakleśair āvṛtaṃ viśodhyate || 2 ||</l>
</lg>

<lg xml:id="HT1.9.3">
	<l>svasaṃvedyātmikā śuddhir nānyaśuddhyā vimucyate |</l>
	<l>viṣayāṇāṃ śuddhabhāvatvāt svasaṃvedyaṃ paraṃ sukham || 3 ||</l>
</lg>

<lg xml:id="HT1.9.4">
	<l>rūpaviṣayādi ye 'py anye pratibhāsante hi yoginaḥ |</l>
	<l>sarve te śuddhabhāvā hi yasmād buddhamayaṃ jagat || 4 ||</l>
</lg>

<p xml:id="HT1.9.5">
	he bhagavan ke te 'viśuddhāḥ |
	bhagavān āha—rūpādayaḥ | kasmāt | grāhyagrāhakabhāvāt |
	vajragarbha āha—ke te grāhakāḥ grāhyāś ceti || 5 ||
</p>

<p xml:id="HT1.9.6">
	bhagavān āha—
	<lg>
		<l>cakṣuṣā gṛhyate rūpaṃ śabdaḥ karṇena gṛhyate |</l>
		<l>gandhaṃ nāsikayā ceti jihvayā svādanaṃ viduḥ || 6 ||</l>
	</lg>
</p>

<lg xml:id="HT1.9.7">
	<l>kāyena spṛśyate vastu manaḥ sukhādim āpnute |</l>
	<l>sevitavyā ime sevyā nirviṣīkṛtya śuddhitaḥ || 7 ||</l>
</lg>

<lg xml:id="HT1.9.8">
	<l>rūpaskandhe bhaved vajrā gaurī ca vedanāyāṃ smṛtā |</l>
	<l>saṃjñāyāṃ vāriyoginī saṃskāre vajraḍākinī || 8 ||</l>
</lg>

<p xml:id="HT1.9.9">
	<lg>
		<l>vijñānaskandharūpeṇa sthitā nairātmyayoginī</l>
	</lg>
	—adhyātmapuṭe |
	<lg>
		<l>sadā hy āsāṃ viśuddhyā vai sidhyante tattvayoginaḥ ||</l>
	</lg>
</p>

<p xml:id="HT1.9.10">
	bāhyapuṭe—
	<lg>
		<l>aiśānyāṃ pukkasī khyātā agnau śabarī tathaiva ca |</l>
		<l>nairṛtye sthāpya caṇḍālī vāyave ḍombinī sthitā || 10 ||</l>
	</lg>
</p>

<lg xml:id="HT1.9.11">
	<l>paścād bāhyapuṭaṃ vakṣye aparagauryādidevatīm |</l>
	<l>indre gaurī yame caurī vettālī vāruṇe diśi |</l>
	<l>kauvere ghasmarī caiva adhare bhūcarī smṛtā || 11 ||</l>
</lg>

<lg xml:id="HT1.9.12">
	<l>ūrdhve khecarī proktā utpattikramapakṣataḥ |</l>
	<l>bhavanirvāṇasvabhāvena sthitāv etau dvidevatī || 12 ||</l>
</lg>

<lg xml:id="HT1.9.13">
	<l>rūpe gaurī samākhyātā śabde caurī prakīrtitā |</l>
	<l>vettālī gandhabhāge ca rase ghasmarī prakīrtitā || 13 ||</l>
</lg>

<lg xml:id="HT1.9.14">
	<l>sparśe bhūcarī khyātā khecarī dharmadhātutaḥ |</l>
	<l>sadā hy āsāṃ viśuddhyā tu sidhyanti tattvayoginaḥ || 14 ||</l>
</lg>

<lg xml:id="HT1.9.15">
	<l>bhujānāṃ śūnyatā śuddhiś caraṇo māraśuddhitaḥ |</l>
	<l>mukhāṣṭavimokṣeṇa netraśuddhis trivajriṇā || 15 ||</l>
</lg>

<lg xml:id="HT1.9.16">
	<l>pṛthivī pukkasī khyātā abdhātuḥ śabarī smṛtā |</l>
	<l>tejaś caṇḍālinī jñeyā vāyur ḍombī prakīrtitā || 16 ||</l>
</lg>

<lg xml:id="HT1.9.17">
	<l>dveṣākhyāpitā cakrī rāgā ca vāriyoginī |</l>
	<l>īrṣyā vajraḍākī paiśunyā guptagaurikā || 17 ||</l>
</lg>

<lg xml:id="HT1.9.18">
	<l>moha tathākhyātā dveṣādīnāṃ viśodhanam |</l>
	<l>etena śodhyate skandha utpattikramapakṣataḥ || 18 ||</l>
</lg>
     
<lg xml:id="HT1.9.19">
	<l>yena tu yena badhyati lokas</l>
	<l>tena tu tena tu bandhana muñce |</l>
	<l>loko muhyati vetti na tattvaṃ</l>
	<l>tattvavivarjita siddhi na lapse |</l>
</lg>
     
<lg xml:id="HT1.9.20">
	<l>tasmād gandha na śabda na rūpaṃ</l>
	<l>naiva rasaṃ na ca cittaviśuddhiḥ |</l>
	<l>sparśa na dharma na sarvaviśuddhyā</l>
	<l>śuddhasahāva jago jaga manye || 20 ||</l>
</lg>

<p xml:id="HT1.9.col">
	viśuddhipaṭalo navamaḥ ||  ||
</p>
